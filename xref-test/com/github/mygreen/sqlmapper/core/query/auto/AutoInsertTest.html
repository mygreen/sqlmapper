21  @ExtendWith(SpringExtension.class)
22  @ContextConfiguration(classes=TestConfig.class)
23  public class AutoInsertTest {
24  
25      @Autowired
26      SqlMapper sqlMapper;
27  
28      @Test
29      void testInsert() {
30  
31          Customer entity = new Customer();
32          entity.setId("00001");
33          entity.setFirstName("Taro");
34          entity.setLastName("Yamada");
35          entity.setBirthday(LocalDate.of(2010, 10, 1));
36  
37          int count = sqlMapper.insert(entity)
38              .execute();
39  
40          assertThat(count).isEqualTo(1);
41  
42          Customer result = sqlMapper.selectFrom(MCustomer.customer)
43                  .id(entity.getId())
44                  .getSingleResult();
45  
46          assertThat(result.getId()).isEqualTo("00001");
47          assertThat(result.getVersion()).isEqualTo(0L);
48  
49      }
50  
51      @Test
52      void testInsertWithGenerateIdentity() {
53  
54          Employee entity = new Employee();
55          entity.setName("Suzuki Hanako");
56          entity.setAge(20);
57          entity.setRole(Role.MANAGER);
58          entity.setHireDate(LocalDate.of(2021, 1, 1));
59          entity.setSectionCode("021");
60          entity.setBusinessEstablishmentCode(1);
61  
62          int execCount = sqlMapper.insert(entity)
63              .execute();
64  
65          assertThat(execCount).isEqualTo(1);
66  
67          long recordCount = sqlMapper.selectFrom(MEmployee.employee).getCount();
68  
69          Employee result = sqlMapper.selectFrom(MEmployee.employee)
70                  .id(entity.getId())
71                  .getSingleResult();
72  
73          assertThat(result.getId()).isEqualTo(recordCount);
74          assertThat(result.getVersion()).isEqualTo(0L);
75  
76      }
77  }
