48          SqlTemplateContext templateContext = new MapSqlTemplateContext(Map.of("customerId", "005"));
49  
50          int count = txNew().execute(action ->
51                  sqlMapper.updateBySqlFile("/sqltemplate/customer_deleteById.sql", templateContext)
52                      .execute()
53                  );
54  
55          assertThat(count).isEqualTo(1);
56  
57          MCustomer m_ = MCustomer.customer;
58  
59          Optional<Customer> result = sqlMapper.selectFrom(m_)

61                  .getOptionalResult();
