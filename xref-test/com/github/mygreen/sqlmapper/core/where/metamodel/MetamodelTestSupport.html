14  /**
15   * メタモデルのテストサポートをするクラス
16   *
17   *
18   * @author T.TSUCHIE
19   *
20   */
21  public abstract class MetamodelTestSupport {
22  
23      @Autowired
24      protected EntityMetaFactory entityMetaFactory;
25  
26      @Autowired
27      protected Dialect dialect;
28  
29      /**
30       * エンティティパスから、{@link MetamodelWhereVisitor}を作成します。
31       * @param entityPaths 条件式に含まれるエンティティパス。
32       * @return {@link MetamodelWhereVisitor}のインスタンス
33       */
34      protected MetamodelWhereVisitor createVisitor(EntityPath<?>... entityPaths) {
35          Map<Class<?>, EntityMeta> entityMetaMap = new HashMap<>();
36          TableNameResolver tableNameResolver = new TableNameResolver();
37  
38          for(EntityPath<?> entityPath : entityPaths) {
39              EntityMeta entityMeta = entityMetaFactory.create(entityPath.getType());
40              tableNameResolver.prepareTableAlias(entityPath);
41              entityMetaMap.put(entityPath.getType(), entityMeta);
42          }
43  
44          MetamodelWhereVisitor visitor = new MetamodelWhereVisitor(entityMetaMap, dialect, entityMetaFactory, tableNameResolver);
45  
46          return visitor;
47  
48      }
49  }
