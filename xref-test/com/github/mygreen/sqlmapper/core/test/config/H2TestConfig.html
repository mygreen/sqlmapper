14  
15  /**
16   * H2接続用の設定
17   *
18   *
19   * @author T.TSUCHIE
20   *
21   */
22  @EnableTransactionManagement
23  @Configuration
24  public class H2TestConfig extends SqlMapperConfigurationSupport {
25  
26      @Override
27      public DataSource dataSource() {
28          return new EmbeddedDatabaseBuilder()
29                  .generateUniqueName(true)
30                  .setType(EmbeddedDatabaseType.H2)
31                  .setScriptEncoding("UTF-8")
32                  .addScript("script/h2/create_schema.sql")
33                  .build();
34      }
35  
36      @Override
37      public Dialect dialect() {
38          return new H2Dialect();
39      }
40  }
