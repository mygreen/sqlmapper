18      @Test
19      void testPathMeta() {
20  
21          MSampleEntity entity = MSampleEntity.sampleEntity;
22  
23          // エンティティのチェック
24          assertThat(entity.getType()).isEqualTo(SampleEntity.class);
25          assertThat(entity.getPathMeta().getParent()).isNull();
26          assertThat(entity.getPathMeta().getElement()).isEqualTo("sampleEntity");
27          assertThat(entity.getPathMeta().getType()).isEqualTo(PathType.ROOT);
28  
29          // プロパティのチェック - 埋め込みプロパティ
30          {
31              EntityPath<PK> prop = entity.id;
32              assertThat(prop.getType()).isEqualTo(PK.class);
33              assertThat(prop.getPathMeta().getParent()).isEqualTo(entity);
34              assertThat(prop.getPathMeta().getElement()).isEqualTo("id");
35              assertThat(prop.getPathMeta().getType()).isEqualTo(PathType.EMBEDDED);
36  
37          }
38  
39          // プロパティのチェック - 埋め込みエンティティ内のプロパティ
40          {
41              PropertyPath<String> prop = entity.id.key1;
42              assertThat(prop.getType()).isEqualTo(String.class);
43              assertThat(prop.getPathMeta().getParent()).isEqualTo(entity.id);
44              assertThat(prop.getPathMeta().getElement()).isEqualTo("key1");
45              assertThat(prop.getPathMeta().getType()).isEqualTo(PathType.PROPERTY);
46          }
47  
48      }
49  
50      @Test
51      void testBuildCondition() {
52  
53          MSampleEntity entity = MSampleEntity.sampleEntity;
54  
55          Predicate exp = entity.id.key1.eq("100")
56                  .and(entity.name.contains("Taro"));
57          String resultString = exp.toString();
58  
59          assertThat(resultString).isEqualTo("sampleEntity.id.key1 = 100 and contains(sampleEntity.name, Taro)");
60  
61      }
