4   import com.github.mygreen.sqlmapper.metamodel.expression.Expression;

8   /**
9    * 単項演算子({@link UnaryOp})に対する処理を定義します。
10   *
11   *
12   * @author T.TSUCHIE
13   *
14   */
15  public class UnaryOpHandler extends OperationHandler<UnaryOp> {
16  
17      @Override
18      protected void init() {
19          addTemplate(UnaryOp.NOT, "not {0}");
20          addTemplate(UnaryOp.IS_NULL, "{0} is null");
21          addTemplate(UnaryOp.IS_NOT_NULL, "{0} is not null");
22          addTemplate(UnaryOp.EXISTS, "exists {0}");
23          addTemplate(UnaryOp.NOT_EXISTS, "not exists {0}");
24      }
25  
26      @Override
27      public void handle(UnaryOp operator, Operation<?> expr, Visitor<VisitorContext> visitor, VisitorContext context) {
28  
29          Expression<?> left = expr.getArg(0);
30  
31          VisitorContext leftContext = new VisitorContext(context);
32  
33          // 左辺の評価
34          invoke(operator, left, visitor, leftContext);
35  
36          // 評価した結果を親のコンテキストに追加する
37          String sql = formatWithTemplate(operator, leftContext.getCriteria());
38          context.appendSql(sql);
39          context.addParamValues(leftContext.getParamValues());
40  
41      }
42  
43  }
