10  /**
11   * 算術演算子({@link ArithmeticOp})に対する処理を定義します。
12   *
13   *
14   * @author T.TSUCHIE
15   *
16   */
17  public class ArithmeticOpHandler extends OperationHandler<ArithmeticOp> {
18  
19      @Override
20      protected void init() {
21          // 算術演算子
22          addTemplate(ArithmeticOp.MULT, "{0} * {1}");
23          addTemplate(ArithmeticOp.DIV, "{0} / {1}");
24          addTemplate(ArithmeticOp.MOD, "mod({0}, {1})");
25          addTemplate(ArithmeticOp.ADD, "{0} + {1}");
26          addTemplate(ArithmeticOp.SUB, "{0} - {1}");
27  
28      }
29  
30      @Override
31      public void handle(ArithmeticOp operator, Operation<?> expr, Visitor<VisitorContext> visitor, VisitorContext context) {
32  
33          Expression<?> left = expr.getArg(0);
34          Expression<?> right = expr.getArg(1);
35  
36          VisitorContext leftContext = new VisitorContext(context);
37          VisitorContext rightContext = new VisitorContext(context);
38  
39          // 左辺の評価
40          invoke(operator, left, visitor, leftContext);
41  
42          /*
43           * 算術演算子の場合、左変=プロパティパス、右辺=定数で構成される場合、
44           * 定数をプロパティの変換規則に従い変換する。
45           */
46          if(isPropertyPath(left) && right instanceof Constant) {
47              visitConstantWithPropertyPath((Path<?>)left, (Constant<?>)right, rightContext);
48          } else {
49              invoke(operator, right, visitor, rightContext);
50          }
51  
52          // 評価した結果を親のコンテキストに追加する
53          String sql = formatWithTemplate(operator, leftContext.getCriteria(), rightContext.getCriteria());
54          context.appendSql(sql);
55  
56          context.addParamValues(leftContext.getParamValues());
57          context.addParamValues(rightContext.getParamValues());
58      }
59  }
