1   package com.github.mygreen.sqlmapper.core.type;
2   
3   import java.sql.ResultSet;
4   import java.sql.SQLException;
5   
6   import org.springframework.jdbc.core.SqlParameterValue;
7   
8   import com.github.mygreen.splate.type.SqlTemplateValueType;
9   import com.github.mygreen.splate.type.SqlTypeConversionException;
10  
11  /**
12   * SQL(JDBC)とマッピング先の型を表すインタフェースです。
13   *
14   *
15   * @author T.TSUCHIE
16   *
17   * @param <T> マッピング先の型
18   */
19  public interface ValueType<T> extends SqlTemplateValueType<T> {
20  
21      /**
22       * カラムの値を返します。
23       *
24       * @param rs 結果セット
25       * @param columnIndex カラムの位置
26       * @return カラムの値
27       * @throws SQLException 結果セットから値をと出すときにスローされます。
28       * @throws SqlValueConversionException SQLの値の変換に失敗したときにストローされます。
29       */
30      T getValue(ResultSet rs, int columnIndex) throws SQLException, SqlValueConversionException;
31  
32      /**
33       * SQLのパラメータ変数として値を取得します。
34       * <p>JDBCが対応していないタイプの場合は、対応している値に変換します。</p>
35       * <p>{@link SqlParameterValue} として返すことで、特殊な値を対応することができます。</p>
36       *
37       * @param value 変換する値
38       * @return SQLのパラメータ変数。
39       * @throws SqlParameterBindException SQL変数の値へのバインドに失敗した場合にスローされます。
40       */
41      Object getSqlParameterValue(T value) throws SqlParameterBindException;
42  
43      /**
44       * {@inheritDoc}
45       * デフォルト実装として、{@link #getSqlParameterValue(Object)} に委譲されます。
46       */
47      @Override
48      default Object getBindVariableValue(T value) throws SqlTypeConversionException {
49          return getSqlParameterValue(value);
50      }
51  }
