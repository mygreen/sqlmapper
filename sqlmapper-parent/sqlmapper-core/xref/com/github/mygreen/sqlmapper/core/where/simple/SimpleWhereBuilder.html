3   import java.util.ArrayList;
4   import java.util.List;

6   import com.github.mygreen.sqlmapper.core.where.Where;
7   import com.github.mygreen.sqlmapper.core.where.WhereVisitor;
8   
9   import lombok.AccessLevel;
10  import lombok.Getter;
11  
12  /**
13   * SQLのWhere句の条件をANDやORでつなげていく組み立てるためのクラスです。
14   *
15   * @author T.TSUCHIE
16   *
17   */
18  public class SimpleWhereBuilder extends AbstractWhere<SimpleWhereBuilder> implements Where {
19  
20      /**
21       * ORで区切られた塊
22       */
23      @Getter(AccessLevel.PROTECTED)
24      private List<Where> childrenWhere = new ArrayList<>();
25  
26      @Override
27      public void accept(final WhereVisitor visitor) {
28          visitor.visit(this);
29      }
30  
31      /**
32       * これまでに追加された条件とこれから追加される条件を {@literal OR} で結合します。
33       * @return このインスタンス自身
34       */
35      public SimpleWhereBuilder or() {
36          if(!terms.isEmpty()) {
37              this.childrenWhere.add(putAsSimpleWhere());
38          }
39          return this;
40      }
41  
42      /**
43       * これまでに追加された条件と、引数で渡された条件全体を {@literal AND} で結合します。
44       * @param where  {@literal AND}で結合される条件
45       * @return このインスタンス自身
46       */
47      public SimpleWhereBuilder and(final AbstractWhere<?> where) {
48          addTerm(new WhereTerm(where));
49          return this;
50      }
51  }
