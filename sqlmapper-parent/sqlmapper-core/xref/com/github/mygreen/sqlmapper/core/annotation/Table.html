1   package com.github.mygreen.sqlmapper.core.annotation;
2   
3   import java.lang.annotation.Documented;
4   import java.lang.annotation.ElementType;
5   import java.lang.annotation.Retention;
6   import java.lang.annotation.RetentionPolicy;
7   import java.lang.annotation.Target;
8   
9   /**
10   * テーブル情報を指定します。
11   * 本アノテーションが指定されていない場合は、デフォルト値が適用されます。
12   *
13   * @version 0.3
14   * @author T.TSUCHIE
15   *
16   */
17  @Retention(RetentionPolicy.RUNTIME)
18  @Target({ElementType.TYPE})
19  @Documented
20  public @interface Table {
21  
22      /**
23       * (オプション) テーブルの含まれるスキーマ。
24       * デフォルトではユーザーにとっての規定のスキーマです。
25       *
26       * @return スキーマ名。
27       */
28      String schema() default "";
29  
30      /**
31       * (オプション) テーブルの含まれるカタログ。
32       * デフォルトでは既定のカタログです。
33       *
34       * @return カタログ名。
35       */
36      String catalog() default "";
37  
38  
39      /**
40       * (オプション) テーブルの名前。
41       * デフォルトの値はエンティティの名前です。
42       *
43       * @return テーブル名。
44       */
45      String name() default "";
46  
47      /**
48       * (オプション) テーブルが読み取り専用かどうかを指定します。
49       * <p>この属性の値が {@literal true} のときに、挿入／更新／削除 操作が呼ばれたときエラーとなります。
50       *
51       * @since 0.3
52       * @return テーブルが読み取り専用かどうか。
53       */
54      boolean readOnly() default false;
55  
56  }
