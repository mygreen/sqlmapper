101     /**
102      * プロパティのメタ情報を作成します。
103      * @param field フィールド
104      * @param entityMeta エンティティのメタ情報。空の場合はID情報の処理をスキップします。
105      * @param embeddedId 埋め込み型のIDのプロパティかどうか。
106      * @return プロパティのメタ情報
107      */
108     public PropertyMeta create(final Field field, final Optional<EntityMeta> entityMeta, final boolean embeddedId) {
109 
110         final Class<?> declaringClass = field.getDeclaringClass();
111         final PropertyMeta propertyMeta = new PropertyMeta(field.getName(), field.getType());
112         propertyMeta.setEmbeddedableId(embeddedId);
113         doField(propertyMeta, field);
114 
115         // フィールドに対するgetter/setterメソッドを設定します。
116         for(Method method : declaringClass.getMethods()) {
117             ReflectionUtils.makeAccessible(method);
118 
119             int modifiers = method.getModifiers();
120             if(Modifier.isStatic(modifiers)) {
121                 continue;
122             }
123 
124             if(ClassUtils.isSetterMethod(method)) {
125                 doSetterMethod(propertyMeta, method);
126 
127             } else if(ClassUtils.isGetterMethod(method) || ClassUtils.isBooleanGetterMethod(method)) {
128                 doGetterMethod(propertyMeta, method);
129             }
130         }
131 
132         // 永続化対象のプロパティはカラム情報を設定します。
133         if(!propertyMeta.isEmbedded() && !propertyMeta.isTransient()) {
134 
135             doColumnMeta(propertyMeta);
136             entityMeta.ifPresent(em -> doIdGenerator(propertyMeta, em));
137 
138             // プロパティに対する型変換を設定します。
139             ValueType<?> valueType = valueTypeRegistry.findValueType(propertyMeta);
140 
141             // OracleなどBoolean型を純粋にサポートしていない場合は、int型に変換するタイプに変換する。
142             valueType = dialect.getValueType(valueType);
143             propertyMeta.setValueType(valueType);
144 
145 
146             validateColumnProperty(declaringClass, propertyMeta);
147 
148         }
149 
150         return propertyMeta;
151 
152     }

154     /**
155      * プロパティのメタ情報に対する処理を実行します。
156      * @param propertyMeta プロパティのメタ情報
157      * @param field フィールド情報
158      */
159     private void doField(final PropertyMeta propertyMeta, final Field field) {
160 
161         propertyMeta.setField(field);
162 
163         final Annotation[] annos = field.getAnnotations();
164         for(Annotation anno : annos) {
165             if(!isSupportedAnnotation(anno)) {
166                 continue;
167             }
168 
169             final Class<? extends Annotation> annoClass = anno.annotationType();
170             propertyMeta.addAnnotation(annoClass, anno);
171         }
172     }

174     /**
175      * サポートするアノテーションか判定する。
176      * <p>確実に重複するJava標準のアノテーションは除外するようにします。</p>
177      *
178      * @param anno 判定対象のアノテーション
179      * @return tureのときサポートします。
180      */
181     private boolean isSupportedAnnotation(final Annotation anno) {
182 
183         final String name = anno.annotationType().getName();
184         if(name.startsWith("java.lang.annotation.")) {
185             return false;
186         }
187 
188         return true;
189     }

191     /**
192      * setterメソッドの情報を処理する。
193      * @param propertyMeta プロパティのメタ情報
194      * @param method setterメソッド
195      */
196     private void doSetterMethod(final PropertyMeta propertyMeta, final Method method) {
197 
198         final String methodName = method.getName();
199         final String propertyName = NameUtils.uncapitalize(methodName.substring(3));
200 
201         if(!propertyMeta.getName().equals(propertyName)) {
202             // プロパティ名が一致しない場合はスキップする
203             return;
204         }
205 
206         propertyMeta.setWriteMethod(method);
207 
208     }
209 
210     /**
211      * getterメソッドの情報を処理する。
212      * @param propertyMeta プロパティのメタ情報
213      * @param method getterメソッド
214      */
215     private void doGetterMethod(final PropertyMeta propertyMeta, final Method method) {
216 
217         final String methodName = method.getName();
218         final String propertyName;
219         if(methodName.startsWith("get")) {
220             propertyName = NameUtils.uncapitalize(methodName.substring(3));
221         } else {
222             // 「is」から始まる場合
223             propertyName = NameUtils.uncapitalize(methodName.substring(2));
224         }
225 
226         if(!propertyMeta.getName().equals(propertyName)) {
227             // プロパティ名が一致しない場合はスキップする
228             return;
229         }
230 
231         propertyMeta.setReadMethod(method);
232 
233     }

235     /**
236      * カラム情報を処理する。
237      * @param propertyMeta プロパティのメタ情報
238      */
239     private void doColumnMeta(final PropertyMeta propertyMeta) {
240 
241         final ColumnMeta columnMeta = new ColumnMeta();
242 
243         final String defaultColumnName = namingRule.propertyToColumn(propertyMeta.getName());
244 
245         Optional<Column> annoColumn = propertyMeta.getAnnotation(Column.class);
246         if(annoColumn.isPresent()) {
247             if(!annoColumn.get().name().isEmpty()) {
248                 columnMeta.setName(annoColumn.get().name());
249             } else {
250                 columnMeta.setName(defaultColumnName);
251             }
252 
253             columnMeta.setUpdatable(annoColumn.get().updatable());
254 
255         } else {
256             columnMeta.setName(defaultColumnName);
257         }
258 
259         propertyMeta.setColumnMeta(columnMeta);
260 
261     }

263     /**
264      * 主キーの生成情報を処理します。
265      * @param propertyMeta プロパティのメタ情報
266      * @param entityMeta エンティティのメタ情報
267      */
268     private void doIdGenerator(final PropertyMeta propertyMeta, final EntityMeta entityMeta) {
269 
270         if(!propertyMeta.isId()) {
271             return;
272         }
273 
274         Optional<GeneratedValue> annoGeneratedValue = propertyMeta.getAnnotation(GeneratedValue.class);
275         if(annoGeneratedValue.isEmpty()) {
276             return;
277         }
278 
279         final Class<?> propertyType = propertyMeta.getPropertyType();
280 
281         GenerationType generationType = annoGeneratedValue.get().strategy();
282         if(generationType != GenerationType.AUTO && !dialect.supportsGenerationType(generationType)) {
283             throw new InvalidEntityException(entityMeta.getEntityType(), messageFormatter.create("property.anno.attr.notSupportForDialect")
284                     .paramWithClass("classType", entityMeta.getEntityType())
285                     .param("property", propertyMeta.getName())
286                     .paramWithAnno("anno", GeneratedValue.class)
287                     .param("attrName", "strategy")
288                     .param("attrValue", generationType)
289                     .param("dialectName", dialect.getName())
290                     .format());
291         }
292 
293         // 生成戦略の補完
294         if(generationType == GenerationType.AUTO) {
295             generationType = dialect.getDefaultGenerationType();
296         }
297 
298         final IdGenerator idGenerator;
299         if(StringUtils.hasLength(annoGeneratedValue.get().generator())) {
300             /*
301              * 属性「generator」が指定されている場合は、「strategy」の値は無視する。
302              * ただし、IDENTITYの場合はクエリ実行前に処理されてしまうので、
303              * Autoに補完して間違った処理をされないようにする。
304              */
305             generationType = GenerationType.AUTO;
306 
307             final String generatorName = annoGeneratedValue.get().generator();
308             try {
309                 idGenerator = applicationContext.getBean(generatorName, IdGenerator.class);
310 
311             } catch(NoSuchBeanDefinitionException  e) {
312                 throw new InvalidEntityException(entityMeta.getEntityType(), messageFormatter.create("property.anno.attr.noSuchBeanDefinition")
313                         .paramWithClass("classType", entityMeta.getEntityType())
314                         .param("property", propertyMeta.getName())
315                         .paramWithAnno("anno", GeneratedValue.class)
316                         .param("attrName", "generator")
317                         .param("attrValue", generatorName)
318                         .format(), e);
319             } catch(BeanNotOfRequiredTypeException e) {
320                 throw new InvalidEntityException(entityMeta.getEntityType(), messageFormatter.create("property.anno.attr.beanNotOfRequiredType")
321                         .paramWithClass("classType", entityMeta.getEntityType())
322                         .param("property", propertyMeta.getName())
323                         .paramWithAnno("anno", GeneratedValue.class)
324                         .param("attrName", "generator")
325                         .param("attrValue", generatorName)
326                         .paramWithClass("requiredType", IdGenerator.class)
327                         .format(), e);
328             }
329         } else if(generationType == GenerationType.IDENTITY) {
330             IdentityIdGenerator identityIdGenerator = new IdentityIdGenerator(propertyType);
331             idGenerator = identityIdGenerator;
332 
333         } else if(generationType == GenerationType.SEQUENCE) {
334             Optional<SequenceGenerator> annoSequenceGenerator = propertyMeta.getAnnotation(SequenceGenerator.class);
335             final String sequenceName;
336             if(annoSequenceGenerator.isPresent() && !annoSequenceGenerator.get().sequenceName().isEmpty()) {
337                 sequenceName = NameUtils.tableFullName(annoSequenceGenerator.get().sequenceName(),
338                         annoSequenceGenerator.get().catalog(),
339                         annoSequenceGenerator.get().schema());
340             } else {
341                 sequenceName = entityMeta.getTableMeta().getName() + "_" + propertyMeta.getColumnMeta().getName();
342             }
343             SequenceIdGenerator sequenceIdGenerator = new SequenceIdGenerator(
344                     dialect.getSequenceIncrementer(dataSource, sequenceName), propertyType);
345 
346             annoSequenceGenerator.map(a -> a.format()).filter(f -> !f.isEmpty()).ifPresent(f -> {
347                 try {
348                     sequenceIdGenerator.setFormatter(new DecimalFormat(f));
349                 } catch(IllegalArgumentException e) {
350                     throw new InvalidEntityException(entityMeta.getEntityType(), messageFormatter.create("property.anno.attr.wrongFormat")
351                             .paramWithClass("classType", entityMeta.getEntityType())
352                             .param("property", propertyMeta.getName())
353                             .paramWithAnno("anno", TableGenerator.class)
354                             .param("attrName", "format")
355                             .param("attrValue", f)
356                             .format(), e);
357                 }
358             });
359 
360             idGenerator = sequenceIdGenerator;
361 
362         } else if(generationType == GenerationType.TABLE) {
363             Optional<TableGenerator> annoTableGenerator = propertyMeta.getAnnotation(TableGenerator.class);
364 
365             final TableIdContext tableIdContext = new TableIdContext();
366             tableIdContext.setTable(tableIdGeneratorProperties.getTable());
367             tableIdContext.setSchema(tableIdGeneratorProperties.getSchema());
368             tableIdContext.setCatalog(tableIdGeneratorProperties.getCatalog());
369             tableIdContext.setPkColumn(tableIdGeneratorProperties.getPkColumn());
370             tableIdContext.setValueColumn(tableIdGeneratorProperties.getValueColumn());
371             tableIdContext.setAllocationSize(tableIdGeneratorProperties.getAllocationSize());
372             tableIdContext.setInitialValue(tableIdGeneratorProperties.getInitialValue());
373 
374             String sequenceName = entityMeta.getTableMeta().getName() + "_" + propertyMeta.getColumnMeta().getName();
375 
376             annoTableGenerator.ifPresent(a -> {
377                 if(!a.table().isEmpty()) {
378                     tableIdContext.setTable(a.table());
379                 }
380 
381                 if(!a.schema().isEmpty()) {
382                     tableIdContext.setSchema(a.schema());
383                 }
384 
385                 if(!a.catalog().isEmpty()) {
386                     tableIdContext.setCatalog(a.catalog());
387                 }
388 
389                 if(!a.pkColumn().isEmpty()) {
390                     tableIdContext.setPkColumn(a.pkColumn());
391                 }
392 
393                 if(!a.valueColumn().isEmpty()) {
394                     tableIdContext.setValueColumn(a.valueColumn());
395                 }
396 
397                 if(a.initialValue() >= 0L) {
398                     tableIdContext.setInitialValue(a.initialValue());
399                 } else {
400                     throw new InvalidEntityException(entityMeta.getEntityType(), messageFormatter.create("property.anno.attr.min")
401                             .paramWithClass("classType", entityMeta.getEntityType())
402                             .param("property", propertyMeta.getName())
403                             .paramWithAnno("anno", TableGenerator.class)
404                             .param("attrName", "initialValue")
405                             .param("attrValue", a.initialValue())
406                             .param("min", 0)
407                             .format());
408                 }
409 
410                 if(a.allocationSize() >= 1L) {
411                     tableIdContext.setAllocationSize(a.allocationSize());
412                 } else {
413                     throw new InvalidEntityException(entityMeta.getEntityType(), messageFormatter.create("property.anno.attr.min")
414                             .paramWithClass("classType", entityMeta.getEntityType())
415                             .param("property", propertyMeta.getName())
416                             .paramWithAnno("anno", TableGenerator.class)
417                             .param("attrName", "allocationSize")
418                             .param("attrValue", a.allocationSize())
419                             .param("min", 1)
420                             .format());
421                 }
422 
423             });
424 
425 
426             if(annoTableGenerator.isPresent() && annoTableGenerator.get().pkColumn().isEmpty()) {
427                 sequenceName = annoTableGenerator.get().pkColumn();
428             }
429 
430             TableIdGenerator tableIdGenerator = new TableIdGenerator(
431                     new TableIdIncrementer(getJdbcTemplate(), tableIdContext),
432                     propertyMeta.getPropertyType(), sequenceName);
433 
434 
435             annoTableGenerator.map(a -> a.format()).filter(f -> !f.isEmpty()).ifPresent(f -> {
436                 try {
437                     tableIdGenerator.setFormatter(new DecimalFormat(f));
438                 } catch(IllegalArgumentException e) {
439                     throw new InvalidEntityException(entityMeta.getEntityType(), messageFormatter.create("property.anno.attr.wrongFormat")
440                             .paramWithClass("classType", entityMeta.getEntityType())
441                             .param("property", propertyMeta.getName())
442                             .paramWithAnno("anno", TableGenerator.class)
443                             .param("attrName", "format")
444                             .param("attrValue", f)
445                             .format(), e);
446                 }
447             });
448 
449             idGenerator = tableIdGenerator;
450 
451 
452         } else if(generationType == GenerationType.UUID) {
453             idGenerator = new UUIDGenerator(propertyType);
454 
455         } else {
456             throw new InvalidEntityException(entityMeta.getEntityType(), messageFormatter.create("property.anno.attr.notSupportValue")
457                     .paramWithClass("classType", entityMeta.getClass())
458                     .param("property", propertyMeta.getName())
459                     .paramWithAnno("anno", GeneratedValue.class)
460                     .param("attrName", "strategy")
461                     .paramWithEnum("attrValue", generationType)
462                     .format());
463         }
464 
465         if(!idGenerator.isSupportedType(propertyType)) {
466             throw new InvalidEntityException(entityMeta.getClass(), messageFormatter.create("property.anno.notSupportTypeList")
467                     .paramWithClass("classType", entityMeta.getEntityType())
468                     .param("property", propertyMeta.getName())
469                     .paramWithAnno("anno", GeneratedValue.class)
470                     .paramWithClass("actualType", propertyType)
471                     .paramWithClass("expectedTypeList", idGenerator.getSupportedTypes())
472                     .format());
473         }
474 
475         propertyMeta.setIdGenerator(idGenerator);
476         propertyMeta.setIdGeneratonType(generationType);
477 
478         // 生成対象のIDの情報
479         final IdGenerationContext generationContext = new IdGenerationContext();
480         generationContext.setTableMeta(entityMeta.getTableMeta());
481         generationContext.setColumnMeta(propertyMeta.getColumnMeta());
482         generationContext.setEntityType(entityMeta.getEntityType());
483         generationContext.setPropertyType(propertyMeta.getPropertyType());
484         propertyMeta.setIdGenerationContext(generationContext);
485 
486     }

488     /**
489      * カラムとなるプロパティの整合性のチェック。
490      * アノテーションとクラスタイプのチェック
491      *
492      * @param declaringClass プロパティが定義されているクラス
493      * @param propertyMeta チェック対象のプロパティ
494      */
495     private void validateColumnProperty(final Class<?> declaringClass, final PropertyMeta propertyMeta) {
496 
497         final Class<?> propertyType = propertyMeta.getPropertyType();
498 
499         // 主キーのタイプチェック
500         if(propertyMeta.isId()
501                 && propertyType != String.class
502                 && propertyType != UUID.class
503                 && propertyType != Integer.class && propertyType != int.class
504                 && propertyType != Long.class && propertyType != long.class) {
505 
506             throw new InvalidEntityException(declaringClass, messageFormatter.create("property.anno.notSupportTypeList")
507                     .paramWithClass("classType", declaringClass)
508                     .param("property", propertyMeta.getName())
509                     .paramWithAnno("anno", Id.class)
510                     .paramWithClass("actualType", propertyType)
511                     .paramWithClass("expectedTypeList", String.class, Integer.class, int.class, Long.class, long.class)
512                     .format());
513         }
514 
515         // 主キーでないのに値の生成用のアノテーションが付与されている場合
516         if(!propertyMeta.isId() && propertyMeta.hasAnnotation(GeneratedValue.class)) {
517 
518             throw new InvalidEntityException(declaringClass, messageFormatter.create("property.anno.notIdWithGeneratedValue")
519                     .paramWithClass("classType", declaringClass)
520                     .param("property", propertyMeta.getName())
521                     .paramWithAnno("anno", GeneratedValue.class)
522                     .format());
523         }
524 
525 
526         // 列挙型のタイプチェック
527         if(propertyMeta.hasAnnotation(Enumerated.class) && !propertyType.isEnum()) {
528 
529             throw new InvalidEntityException(declaringClass, messageFormatter.create("property.anno.notSupportType")
530                     .paramWithClass("classType", declaringClass)
531                     .param("property", propertyMeta.getName())
532                     .paramWithAnno("anno", Enumerated.class)
533                     .paramWithClass("actualType", propertyType)
534                     .paramWithClass("expectedType", Enum.class)
535                     .format());
536         }
537 
538         // バージョンキーのタイプチェック
539         if(propertyMeta.hasAnnotation(Version.class)
540                 && propertyType != Integer.class && propertyType != int.class
541                 && propertyType != Long.class && propertyType != long.class) {
542 
543             throw new InvalidEntityException(declaringClass, messageFormatter.create("property.anno.notSupportTypeList")
544                     .paramWithClass("classType", declaringClass)
545                     .param("property", propertyMeta.getName())
546                     .paramWithAnno("anno", Version.class)
547                     .paramWithClass("actualType", propertyType)
548                     .paramWithClass("expectedTypeList", Integer.class, int.class, Long.class, long.class)
549                     .format());
550 
551         }
552 
553         // 時制のタイプチェック
554         if(!propertyMeta.hasAnnotation(Temporal.class)
555                 && propertyType == java.util.Date.class) {
556 
557             // 時制の型が不明なプロパティに対して、@Temporalが付与されていない場合
558             throw new InvalidEntityException(declaringClass, messageFormatter.create("property.anno.requiredAnnoTemporal")
559                     .paramWithClass("classType", declaringClass)
560                     .param("property", propertyMeta.getName())
561                     .paramWithAnno("anno", Temporal.class)
562                     .format());
563         }
564 
565     }
