40  /**
41   * メタモデルの式ノードを巡回するVisitorです。
42   *
43   *
44   * @author T.TSUCHIE
45   *
46   */
47  public class ExpressionVisitor implements Visitor<VisitorContext> {
48  
49      private Map<Class<?>, OperationHandler<? extends Operator>> operationHandlerMap;
50      {
51          this.operationHandlerMap = new HashMap<>();
52          operationHandlerMap.put(BooleanOp.class, new BooleanOpHandler());
53          operationHandlerMap.put(UnaryOp.class, new UnaryOpHandler());
54          operationHandlerMap.put(ComparisionOp.class, new ComparisionOpHandler());
55          operationHandlerMap.put(LikeOp.class, new LikeOpHandler());
56          operationHandlerMap.put(FunctionOp.class, new FuncOpHandler());
57          operationHandlerMap.put(ArithmeticOp.class, new ArithmeticOpHandler());
58      }
59  
60      @SuppressWarnings({"rawtypes", "unchecked"})
61      @Override
62      public void visit(final Operation<?> expr, final VisitorContext context) {
63          final Operator operator = expr.getOperator();
64  
65          OperationHandler handler = operationHandlerMap.get(operator.getClass());
66          handler.handle(operator, expr, this, context);
67  
68      }
69  
70      @Override
71      public void visit(final Constant<?> expr, final VisitorContext context) {
72          // 値はプレースホルダーを追加
73          if(expr.isExpandable()) {
74              // IN句などの展開可能な複数要素の場合
75              Collection<?> values = (Collection<?>)expr.getValue();
76              context.addParamValues(values);
77              context.appendSql("(")
78                  .append(QueryUtils.repeat("?", ",", values.size()))
79                  .append(")");
80          } else {
81              context.addParamValue(expr.getValue());
82              context.appendSql("?");
83  
84          }
85  
86      }
87  
88  
89      @Override
90      public void visit(final Path<?> expr, final VisitorContext context) {
91  
92          final PathMeta pathMeta = expr.getPathMeta();
93          if(pathMeta.getType() == PathType.PROPERTY) {
94              Path<?> rootPath = pathMeta.findRootPath();
95              Class<?> rootClassType = rootPath.getType();
96              String propertyName = pathMeta.getElement();
97              Optional<PropertyMeta> propertyMeta = context.getEntityMetaMap().get(rootClassType).findPropertyMeta(propertyName);
98              if(propertyMeta.isEmpty()) {
99                  throw new IllegalQueryException("unknwon property : " + propertyName);
100             }
101 
102             // TODO: Embeddedのネストした場合を考慮する
103             final String tableName = context.getTableNameResolver().getTableAlias(rootPath);
104             final String columnName;
105             // SQL - カラム名を追加
106             if(tableName != null) {
107                 columnName = tableName + "." + propertyMeta.get().getColumnMeta().getName();
108             } else {
109                 columnName = propertyMeta.get().getColumnMeta().getName();;
110             }
111             context.appendSql(columnName);
112 
113         } else {
114             throw new IllegalArgumentException("not support pathType=" + pathMeta.getType());
115         }
116 
117     }
118 
119     @Override
120     public void visit(final SubQueryExpression<?> expr, final VisitorContext context) {
121         final SubQueryMeta queryMeta = expr.getQueryMeta();
122         final Dialect dialect = context.getDialect();
123         final TableNameResolver tableNameResolver = context.getTableNameResolver();
124 
125         // テーブル名の組み立て
126         //TODO: テーブル名は新規に割り当てる必要があるかどうか。
127         final FromClause fromClause = new FromClause();
128         tableNameResolver.prepareTableAlias(queryMeta.getEntityPath());
129         EntityMeta entityMeta = context.getEntityMetaFactory().create(queryMeta.getEntityPath().getType());
130         fromClause.addSql(entityMeta.getTableMeta().getFullName(), tableNameResolver.getTableAlias(queryMeta.getEntityPath()));
131 
132         // 抽出カラムの組み立て
133         final SelectClause selectClause = new SelectClause();
134         for(PropertyMeta propertyMeta : entityMeta.getAllColumnPropertyMeta()) {
135             final String propertyName = propertyMeta.getName();
136             final PropertyPath<?> propertyPath = queryMeta.getEntityPath().findPropertyPath(propertyName);
137 
138             if(propertyMeta.isTransient()) {
139                 continue;
140             }
141 
142             if(!queryMeta.getIncludesProperties().isEmpty()
143                     && !queryMeta.getIncludesProperties().contains(propertyPath)) {
144                 continue;
145             }
146 
147             String tableAlias = tableNameResolver.getTableAlias(queryMeta.getEntityPath());
148             selectClause.addSql(tableAlias, propertyMeta.getColumnMeta().getName());
149         }
150 
151         // 条件の組み立て
152         final WhereClause whereClause = new WhereClause();
153         final List<Object> paramValues = new ArrayList<>();
154         if(queryMeta.getWhere() != null) {
155             MetamodelWhereVisitor visitor = new MetamodelWhereVisitor(context.getEntityMetaMap(), context.getDialect(),
156                     context.getEntityMetaFactory(), tableNameResolver);
157             visitor.visit(new MetamodelWhere(queryMeta.getWhere()));
158 
159             whereClause.addSql(visitor.getCriteria());
160             paramValues.addAll(visitor.getParamValues());
161         }
162 
163         // 並び順の組み立て
164         final OrderByClause orderByClause = new OrderByClause();
165         for(OrderSpecifier order : queryMeta.getOrders()) {
166             String propertyName = order.getPath().getPathMeta().getElement();
167             Optional<PropertyMeta> propertyMeta = entityMeta.findPropertyMeta(propertyName);
168 
169             String tableAlias = tableNameResolver.getTableAlias(order.getPath().getPathMeta().getParent());
170             if(!StringUtils.hasLength(tableAlias)) {
171                 //TODO: 例外処理
172 
173             }
174 
175             propertyMeta.ifPresent(p -> {
176                 String orderBy = String.format("%s.%s %s", tableAlias, p.getColumnMeta().getName(), order.getOrder().name());
177                 orderByClause.addSql(orderBy);
178             });
179         }
180 
181         String sql = "select "
182                 + selectClause.toSql()
183                 + fromClause.toSql()
184                 + whereClause.toSql()
185                 + orderByClause.toSql();
186 
187         // limit句の組み立て
188         if(queryMeta.getLimit() > 0 || queryMeta.getLimit() == 0 && queryMeta.getOffset() > 0) {
189             sql = dialect.convertLimitSql(sql, queryMeta.getOffset(), queryMeta.getLimit());
190         }
191 
192         // 組み立てたSQLとパラメータを格納する
193         context.addParamValues(paramValues);
194         context.appendSql(sql);
195     }
