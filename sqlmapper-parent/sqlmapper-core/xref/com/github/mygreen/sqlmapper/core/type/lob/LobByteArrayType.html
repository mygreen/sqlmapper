10  import org.springframework.jdbc.support.lob.LobHandler;
11  
12  import com.github.mygreen.sqlmapper.core.type.ValueType;
13  
14  import lombok.RequiredArgsConstructor;
15  
16  /**
17   * {@literal byte[]} 型のマッピングを処理します。
18   * <p>JDBCの型としては、{@link Blob} 型として処理を行います。
19   *
20   *
21   * @author T.TSUCHIE
22   *
23   */
24  @RequiredArgsConstructor
25  public class LobByteArrayType implements ValueType<byte[]> {
26  
27      private final LobHandler lobHandler;
28  
29      @Override
30      public byte[] getValue(ResultSet rs, int columnIndex) throws SQLException {
31          return lobHandler.getBlobAsBytes(rs, columnIndex);
32      }
33  
34      @Override
35      public Object getSqlParameterValue(byte[] value) {
36          return new SqlParameterValue(Types.BLOB, new SqlLobValue(value, lobHandler));
37      }
38  
39      /**
40       * {@inheritDoc}
41       *
42       * @return byte配列の値を文字列に変換した値を結合した値を返します。
43       */
44      @Override
45      public String getEmbeddedValue(byte[] value) {
46          if(value == null) {
47              return null;
48          }
49  
50          StringBuilder buff = new StringBuilder();
51          for(byte b : value) {
52              buff.append(b);
53          }
54          return buff.toString();
55      }
56  }
