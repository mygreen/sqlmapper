79      /**
80       * IDのテーブルによる自動採番のキャッシュ情報をクリアします。
81       * クリアすることで、次に採番するときに、最新のDBの情報を反映した状態になります。
82       * @since 0.3
83       */
84      public void refreshTableIdGenerator() {
85  
86          entityMetaMap.values().stream()
87              .flatMap(e -> e.getIdPropertyMetaList().stream())
88              .filter(p -> p.getIdGenerator().isPresent() && (p.getIdGenerator().get() instanceof TableIdGenerator))
89              .map(p -> (TableIdGenerator)(p.getIdGenerator().get()))
90              .forEach(g -> g.clearCache());
91  
92      }

94      /**
95       * エンティティ情報を元にメタ情報を組み立てます。
96       * 組み立ててたメタ情報はキャッシュしており、組み立て済みの場合はキャッシュを返します。
97       *
98       * @param entityClass エンティティクラス
99       * @return エンティティのメタ情報
100      */
101     public EntityMeta create(@NonNull final Class<?> entityClass) {
102         return entityMetaMap.computeIfAbsent(entityClass.getName(), s -> doEntityMeta(entityClass));
103     }

105     /**
106      * エンティティのメタ情報に対応する処理を実行します。
107      * @param entityClass エンティティ情報
108      * @return エンティティのメタ情報
109      */
110     private EntityMeta doEntityMeta(final Class<?> entityClass) {
111 
112         final Entity annoEntity = entityClass.getAnnotation(Entity.class);
113         if(annoEntity == null) {
114             throw new InvalidEntityException(entityClass, messageFormatter.create("entity.anno.required")
115                     .paramWithClass("entityClass", entityClass)
116                     .paramWithAnno("anno", Entity.class)
117                     .format());
118         }
119 
120         final EntityMeta entityMeta = new EntityMeta(entityClass);
121 
122         if(!annoEntity.name().isEmpty()) {
123             entityMeta.setName(annoEntity.name());
124         } else {
125             entityMeta.setName(entityClass.getSimpleName());
126         }
127 
128         doTableMeta(entityMeta, entityClass);
129         doPropertyMeta(entityMeta, entityClass);
130 
131         return entityMeta;
132     }
133 
134     /**
135      * エンティティ情報からテーブルのメタ情報に対応する処理を実行します。
136      * @param entityMeta エンティティメタ情報
137      * @param entityClass エンティティ情報
138      */
139     private void doTableMeta(final EntityMeta entityMeta, final Class<?> entityClass) {
140 
141         final TableMeta tableMeta = new TableMeta();
142         final String defaultName = namingRule.entityToTable(entityClass.getName());
143         final Table annoTable = entityClass.getAnnotation(Table.class);
144 
145         if(annoTable != null) {
146             if(!annoTable.name().isEmpty()) {
147                 tableMeta.setName(annoTable.name());
148             } else {
149                 tableMeta.setName(defaultName);
150             }
151 
152             if(!annoTable.schema().isEmpty()) {
153                 tableMeta.setSchema(annoTable.schema());
154             }
155 
156             if(!annoTable.catalog().isEmpty()) {
157                 tableMeta.setCatalog(annoTable.catalog());
158             }
159 
160             tableMeta.setReadOnly(annoTable.readOnly());
161 
162         } else {
163             tableMeta.setName(defaultName);
164         }
165 
166         entityMeta.setTableMeta(tableMeta);
167 
168     }

170     /**
171      * エンティティ情報からプロパティのメタ情報に対応する処理を実行します。
172      * @param entityMeta エンティティメタ情報
173      * @param entityClass エンティティ情報
174      */
175     private void doPropertyMeta(final EntityMeta entityMeta, final Class<?> entityClass) {
176 
177         final List<PropertyMeta> propertyMetaList = new ArrayList<>();
178         extractProperty(entityClass, propertyMetaList, entityMeta, false);
179         extractSuperClassProperty(entityClass.getSuperclass(), propertyMetaList, entityMeta);
180 
181         // 埋め込みのプロパティを抽出する
182         propertyMetaList.stream()
183             .filter(prop -> prop.isEmbedded())
184             .forEach(prop -> doEmbeddedPropertyMeta(entityClass, prop, entityMeta));
185 
186         validateEntity(entityClass, propertyMetaList);
187 
188         propertyMetaList.stream().forEach(p -> entityMeta.addPropertyMeta(p));
189 
190     }
191 
192     /**
193      * 指定したクラスからプロパティを抽出する。
194      * @param targetClass 抽出対象のクラス。
195      * @param propertyMetaList 抽出したプロパティ一覧
196      * @param entityMeta エンティティのメタ情報
197      * @param embeddedId 埋め込み型のIDのプロパティの処理かどうか。
198      */
199     private void extractProperty(final Class<?> targetClass, final List<PropertyMeta> propertyMetaList,
200             final EntityMeta entityMeta, boolean embeddedId) {
201 
202         for(Field field : targetClass.getDeclaredFields()) {
203 
204             int modifiers = field.getModifiers();
205             if(Modifier.isStatic(modifiers) || Modifier.isFinal(modifiers)) {
206                 continue;
207             }
208 
209             ReflectionUtils.makeAccessible(field);
210 
211             propertyMetaList.add(propertyMetaFactory.create(field, Optional.of(entityMeta), embeddedId));
212 
213         }
214     }

216     /**
217      * アノテーション{@literal MappedSuperclass}が付与されている親クラスのプロパティを抽出する。
218      * <p>親クラスにフィールドやアノテーションが定義してある可能性があるので、それらも取得する。</p>
219      * @param superClass 親クラス
220      * @param propertyMetaList 抽出したプロパティ一覧
221      * @param entityClass エンティティ情報
222      */
223     private void extractSuperClassProperty(final Class<?> superClass, final List<PropertyMeta> propertyMetaList,
224             final EntityMeta entityMeta) {
225 
226         if(superClass.equals(Object.class)) {
227             return;
228         }
229 
230         final MappedSuperclass annoMappedSuperclass = superClass.getAnnotation(MappedSuperclass.class);
231         if(annoMappedSuperclass != null) {
232             extractProperty(superClass, propertyMetaList, entityMeta, false);
233         }
234 
235         // 再帰的に遡っていく
236         extractSuperClassProperty(superClass.getSuperclass(), propertyMetaList, entityMeta);
237 
238     }

240     /**
241      * 指定したプロパティから埋め込み用プロパティを設定する。
242      * @param targetClass 抽出対象のクラス。
243      * @param propertyMeta 処理対象のプロパティ情報
244      * @param entityClass エンティティ情報
245      */
246     private void doEmbeddedPropertyMeta(final Class<?> entityClass, final PropertyMeta propertyMeta,
247             final EntityMeta entityMeta) {
248 
249         final Class<?> embeddableClass = propertyMeta.getPropertyType();
250 
251         boolean embeddedId = propertyMeta.hasAnnotation(EmbeddedId.class);
252 
253         if(embeddableClass.getAnnotation(Embeddable.class) == null) {
254             throw new InvalidEntityException(entityClass, messageFormatter.create("embeddable.anno.required")
255                     .paramWithClass("entityClass", entityClass)
256                     .param("property", propertyMeta.getName())
257                     .paramWithClass("embeddableClass", embeddableClass)
258                     .paramWithAnno("anno", Embeddable.class)
259                     .format());
260         }
261 
262         // 埋め込みクラスのプロパティ情報の抽出
263         final List<PropertyMeta> embeddedPropertyList = new ArrayList<>();
264         extractProperty(embeddableClass, embeddedPropertyList, entityMeta, embeddedId);
265 
266         validatedEmbeddedProperty(entityClass, embeddableClass, embeddedPropertyList);
267 
268         embeddedPropertyList.stream().forEach(p -> propertyMeta.addEmbeddedablePropertyMeta(p));
269 
270     }

272     /**
273      * 埋め込みクラスの情報の整合性などの検証
274      * <p>
275      *  <li></li>
276      * </p>
277      *
278      * @param entityClass
279      * @param embeddableClass
280      * @param embeddedProperties
281      */
282     private void validatedEmbeddedProperty(final Class<?> entityClass, final Class<?> embeddableClass,
283             final List<PropertyMeta> embeddedPropertyList) {
284 
285         // プロパティが存在するかどうか。
286         final long propertyCount = embeddedPropertyList.stream()
287                 .filter(p -> !p.isTransient())
288                 .count();
289         if(propertyCount == 0) {
290             throw new InvalidEntityException(entityClass, messageFormatter.create("entity.prop.empty")
291                     .paramWithClass("classType", embeddableClass)
292                     .format());
293         }
294 
295     }

297     private void validateEntity(final Class<?> entityClass, final List<PropertyMeta> propertyMetaList) {
298 
299         // プロパティが存在するかどうか。
300         final long propertyCount = propertyMetaList.stream()
301                 .filter(p -> !p.isTransient())
302                 .count();
303         if(propertyCount == 0) {
304             throw new InvalidEntityException(entityClass, messageFormatter.create("entity.prop.empty")
305                     .paramWithClass("classType", entityClass)
306                     .format());
307         }
308 
309         // カラム用のプロパティが重複していないかどうか
310         Set<String> existsColumnPropertyNames = new TreeSet<>();
311         Set<String> douplicatedColumnPropertyNames = new TreeSet<>();
312         for(PropertyMeta prop : propertyMetaList) {
313             if(prop.isTransient() || !prop.isColumn()) {
314                 continue;
315             }
316 
317             final String propName = prop.getName();
318             if(!existsColumnPropertyNames.add(propName)) {
319                 douplicatedColumnPropertyNames.add(propName);
320             }
321 
322             // 埋め込みプロパティの場合
323             for(PropertyMeta embeddedProp : prop.getEmbeddedablePopertyMetaList()) {
324                 if(embeddedProp.isTransient() || !prop.isColumn()) {
325                     continue;
326                 }
327                 final String embeddedPropName = embeddedProp.getName();
328                 if(!existsColumnPropertyNames.add(embeddedPropName)) {
329                     douplicatedColumnPropertyNames.add(embeddedPropName);
330                 }
331             }
332 
333         }
334         if(douplicatedColumnPropertyNames.size() > 0) {
335             throw new InvalidEntityException(entityClass, messageFormatter.create("entity.prop.columnDuplicated")
336                     .paramWithClass("classType", entityClass)
337                     .param("propertyNames", douplicatedColumnPropertyNames)
338                     .format());
339         }
340 
341         // EmbeddedIdが複数存在しないかどうか。
342         List<PropertyMeta> embeddedIdList = propertyMetaList.stream()
343                 .filter(prop -> prop.isId())
344                 .filter(prop -> prop.isEmbedded())
345                 .collect(Collectors.toList());
346         if(embeddedIdList.size() >= 2) {
347             List<String> propertyNames = embeddedIdList.stream()
348                     .map(prop -> prop.getName())
349                     .collect(Collectors.toList());
350 
351             throw new InvalidEntityException(entityClass, messageFormatter.create("entity.anno.multiPropertyAnno")
352                     .paramWithClass("classType", entityClass)
353                     .paramWithAnno("anno", EmbeddedId.class)
354                     .param("propertyNames", propertyNames)
355                     .format());
356         }
357 
358         // バージョンキーが複数存在しないかどうか。
359         List<PropertyMeta> versionList = propertyMetaList.stream()
360                 .filter(prop -> prop.isVersion())
361                 .collect(Collectors.toList());
362         if(versionList.size() >= 2) {
363             List<String> propertyNames = versionList.stream()
364                     .map(prop -> prop.getName())
365                     .collect(Collectors.toList());
366 
367             throw new InvalidEntityException(entityClass, messageFormatter.create("entity.anno.multiPropertyAnno")
368                     .paramWithClass("classType", entityClass)
369                     .paramWithAnno("anno", Version.class)
370                     .param("propertyNames", propertyNames)
371                     .format());
372         }
373 
374         // バージョンキーを持つとき主キーが存在するかどうか
375         if(versionList.size() > 0) {
376             boolean existsId = propertyMetaList.stream()
377                     .anyMatch(prop -> prop.isId());
378             if(!existsId) {
379                 throw new InvalidEntityException(entityClass, messageFormatter.create("entity.noIdWhenVersion")
380                         .paramWithClass("classType", entityClass)
381                         .format());
382             }
383         }
384 
385     }
