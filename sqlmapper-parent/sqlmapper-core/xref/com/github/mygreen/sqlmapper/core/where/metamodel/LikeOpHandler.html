12  /**
13   * LIKE演算子({@link LikeOp})に対する処理を定義します。
14   *
15   *
16   * @author T.TSUCHIE
17   *
18   */
19  public class LikeOpHandler extends OperationHandler<LikeOp>{
20  
21      /**
22       * デフォルトのLIKEエスケープ文字
23       */
24      private static char DEFAULT_LIKE_ESCAPE = '\\';
25  
26      @Override
27      protected void init() {
28          // 何もしない
29      }
30  
31      @SuppressWarnings("unchecked")
32      @Override
33      public void handle(LikeOp operator, Operation<?> expr, Visitor<VisitorContext> visitor, VisitorContext context) {
34  
35          Expression<?> left = expr.getArg(0);
36          Expression<?> right = expr.getArg(1);
37          Optional<Expression<?>> escapeExp = expr.getOptArg(2);
38          char escape = escapeExp.map(e -> ((Constant<Character>)e).getValue()).orElse(DEFAULT_LIKE_ESCAPE);
39  
40          if(right instanceof Constant) {
41              String text = ((Constant<?>)right).getValue().toString();
42  
43              switch(operator) {
44                  case CONTAINS:
45                      text = "%" + QueryUtils.escapeLike(text, escape) + "%";
46                      break;
47                  case STARTS:
48                      text = QueryUtils.escapeLike(text, escape) + "%";
49                      break;
50                  case ENDS:
51                      text = "%" + QueryUtils.escapeLike(text, escape);
52                      break;
53                  case LIKE:
54                  default:
55                      // 何もしない
56              }
57  
58              invoke(operator, left, visitor, context);
59              context.appendSql(" like ?");
60              if(escapeExp.isPresent()) {
61                  context.appendSql(" escape '" + escape + "'");
62              }
63              context.addParamValue(text);
64  
65          } else {
66              // 定数以外はただのLIKEとして扱う
67              invoke(operator, left, visitor, context);
68              context.appendSql(" like ");
69              if(escapeExp.isPresent()) {
70                  context.appendSql(" escape '" + escape + "'");
71              }
72              invoke(operator, right, visitor, context);
73          }
74      }
75  
76  }
