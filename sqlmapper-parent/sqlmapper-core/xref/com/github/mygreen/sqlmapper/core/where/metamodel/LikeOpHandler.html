10  /**
11   * LIKE演算子({@link LikeOp})に対する処理を定義します。
12   *
13   *
14   * @author T.TSUCHIE
15   *
16   */
17  public class LikeOpHandler extends OperationHandler<LikeOp>{
18  
19      @Override
20      protected void init() {
21          // 何もしない
22      }
23  
24      @Override
25      public void handle(LikeOp operator, Operation<?> expr, Visitor<VisitorContext> visitor, VisitorContext context) {
26  
27          Expression<?> left = expr.getArg(0);
28          Expression<?> right = expr.getArg(1);
29  
30          if(right instanceof Constant) {
31              String text = ((Constant<?>)right).getValue().toString();
32              switch(operator) {
33                  case CONTAINS:
34                      text = "%" + QueryUtils.escapeLike(text) + "%";
35                      break;
36                  case STARTS:
37                      text = QueryUtils.escapeLike(text) + "%";
38                      break;
39                  case ENDS:
40                      text = "%" + QueryUtils.escapeLike(text);
41                      break;
42                  case LIKE:
43                  default:
44                      // 何もしない
45              }
46  
47              invoke(operator, left, visitor, context);
48              context.appendSql(" like ?");
49              context.addParamValue(text);
50  
51          } else {
52              // 定数以外はただのLIKEとして扱う
53              invoke(operator, left, visitor, context);
54              context.appendSql(" like ");
55              invoke(operator, right, visitor, context);
56          }
57      }
58  
59  }
