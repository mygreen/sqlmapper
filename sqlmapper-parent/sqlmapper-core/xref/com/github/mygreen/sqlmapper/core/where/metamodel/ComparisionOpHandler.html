10  /**
11   * 比較演算子({@link ComparisionOp})に対する処理を定義します。
12   *
13   *
14   * @author T.TSUCHIE
15   *
16   */
17  public class ComparisionOpHandler extends OperationHandler<ComparisionOp> {
18  
19      @Override
20      protected void init() {
21          // 比較演算子 - General
22          addTemplate(ComparisionOp.EQ, "{0} = {1}");
23          addTemplate(ComparisionOp.NE, "{0} <> {1}");
24          addTemplate(ComparisionOp.IN, "{0} in {1}");
25          addTemplate(ComparisionOp.NOT_IN, "{0} not in {1}");
26          addTemplate(ComparisionOp.BETWEEN, "{0} between {1} and {2}");
27          addTemplate(ComparisionOp.GOE, "{0} >= {1}");
28          addTemplate(ComparisionOp.GT, "{0} > {1}");
29          addTemplate(ComparisionOp.LOE, "{0} <= {1}");
30          addTemplate(ComparisionOp.LT, "{0} < {1}");
31      }
32  
33      @Override
34      public void handle(ComparisionOp operator, Operation<?> expr, Visitor<VisitorContext> visitor, VisitorContext context) {
35  
36          if(operator == ComparisionOp.BETWEEN) {
37              handleBetween(operator, expr, visitor, context);
38              return;
39          }
40  
41          Expression<?> left = expr.getArg(0);
42          Expression<?> right = expr.getArg(1);
43  
44          VisitorContext leftContext = new VisitorContext(context);
45          VisitorContext rightContext = new VisitorContext(context);
46  
47          // 左辺の評価
48          invoke(operator, left, visitor, leftContext);
49  
50          /*
51           * 比較演算子の場合、左変=プロパティパス、右辺=定数で構成される場合、
52           * 定数をプロパティの変換規則に従い変換する。
53           */
54          if(isPropertyPath(left) && right instanceof Constant) {
55              visitConstantWithPropertyPath((Path<?>)left, (Constant<?>)right, rightContext);
56          } else {
57              invoke(operator, right, visitor, rightContext);
58          }
59  
60          // 評価した結果を親のコンテキストに追加する
61          String sql = formatWithTemplate(operator, leftContext.getCriteria(), rightContext.getCriteria());
62          context.appendSql(sql);
63  
64          context.addParamValues(leftContext.getParamValues());
65          context.addParamValues(rightContext.getParamValues());
66  
67      }
68  
69      private void handleBetween(ComparisionOp operator, Operation<?> expr, Visitor<VisitorContext> visitor, VisitorContext context) {
70  
71          Expression<?> left = expr.getArg(0);
72          Expression<?> right1 = expr.getArg(1);
73          Expression<?> right2 = expr.getArg(2);
74  
75          VisitorContext leftContext = new VisitorContext(context);
76          VisitorContext rightContext1 = new VisitorContext(context);
77          VisitorContext rightContext2 = new VisitorContext(context);
78  
79          // 左辺の評価
80          invoke(operator, left, visitor, leftContext);
81  
82          /*
83           * 比較演算子の場合、左変=プロパティパス、右辺=定数で構成される場合、
84           * 定数をプロパティの変換規則に従い変換する。
85           */
86          if(isPropertyPath(left) && right1 instanceof Constant) {
87              visitConstantWithPropertyPath((Path<?>)left, (Constant<?>)right1, rightContext1);
88          } else {
89              invoke(operator, right1, visitor, rightContext1);
90          }
91  
92          if(isPropertyPath(left) && right2 instanceof Constant) {
93              visitConstantWithPropertyPath((Path<?>)left, (Constant<?>)right2, rightContext2);
94          } else {
95              invoke(operator, right2, visitor, rightContext2);
96          }
97  
98          // 評価した結果を親のコンテキストに追加する
99          String sql = formatWithTemplate(operator, leftContext.getCriteria(), rightContext1.getCriteria(), rightContext2.getCriteria());
100         context.appendSql(sql);
101 
102         context.addParamValues(leftContext.getParamValues());
103         context.addParamValues(rightContext1.getParamValues());
104         context.addParamValues(rightContext2.getParamValues());
105 
106     }
