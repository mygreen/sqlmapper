22  /**
23   * 更新を行うSQLを自動生成するクエリを実行します。
24   * {@link AutoUpdateImpl}のクエリ実行処理の移譲先です。
25   *
26   * @author T.TSUCHIE
27   *
28   */
29  @Slf4j
30  public class AutoUpdateExecutor {
31  
32      /**
33       * クエリ情報
34       */
35      private final AutoUpdateImpl<?> query;
36  
37      /**
38       * 設定情報
39       */
40      private final SqlMapperContext context;
41  
42      /**
43       * SET句
44       */
45      private final SetClause setClause = new SetClause();
46  
47      /**
48       * WHERE句
49       */
50      private final WhereClause whereClause = new WhereClause();
51  
52      /**
53       * 実行するSQLです
54       */
55      private String executedSql;
56  
57      /**
58       * クエリのパラメータです。
59       */
60      private final List<Object> paramValues = new ArrayList<>();
61  
62      /**
63       * 更新対象のプロパティの個数
64       */
65      private int targetPropertyCount = 0;
66  
67      /**
68       * 組み立てたクエリ情報を指定するコンストラクタ。
69       * @param query クエリ情報
70       */
71      public AutoUpdateExecutor(AutoUpdateImpl<?> query) {
72          this.query = query;
73          this.context = query.getContext();
74      }
75  
76      /**
77       * クエリ実行の準備を行います。
78       */
79      private void prepare() {
80          prepareSetClause();
81          prepareWhereClause();
82  
83          prepareSql();
84  
85      }
86  
87      /**
88       * UPDATE文のSET句を準備します。
89       */
90      @SuppressWarnings({"unchecked", "rawtypes"})
91      private void prepareSetClause() {
92  
93          for(PropertyMeta propertyMeta : query.getEntityMeta().getAllColumnPropertyMeta()) {
94  
95              final Object propertyValue = PropertyValueInvoker.getEmbeddedPropertyValue(propertyMeta, query.getEntity());
96              if(!isTargetProperty(propertyMeta, propertyValue)) {
97                  continue;
98              }
99  
100             this.targetPropertyCount++;
101 
102             // SET句の組み立て
103             setClause.addSql(propertyMeta.getColumnMeta().getName(), "?");
104 
105             final ValueType valueType = propertyMeta.getValueType();
106             paramValues.add(valueType.getSqlParameterValue(propertyValue));
107 
108         }
109 
110         // バージョンカラムの更新用のSET句の準備
111         if(!query.isIncludeVersion() && query.getEntityMeta().hasVersionPropertyMeta()) {
112             final PropertyMeta propertyMeta = query.getEntityMeta().getVersionPropertyMeta().get();
113             final String columnName = propertyMeta.getColumnMeta().getName();
114             setClause.addSql(columnName, columnName + " + 1");
115         }
116     }
117 
118     /**
119      * 更新対象のプロパティか判定します。
120      * @param propertyMeta プロパティ情報
121      * @param propertyValue 更新対象の値
122      * @return 更新対象のとき、{@literal true} を返します。
123      */
124     private boolean isTargetProperty(final PropertyMeta propertyMeta, final Object propertyValue) {
125 
126         // 主キーは検索条件に入れるので対象外
127         if(propertyMeta.isId() || !propertyMeta.getColumnMeta().isUpdatable()) {
128             return false;
129         }
130 
131         /*
132          * バージョンキーは通常は更新対象となるため、通常の条件では対象外。
133          * ただし、includeVersion = true のときは更新対象とする。
134          */
135         if(propertyMeta.isVersion() && !query.isIncludeVersion()) {
136             return false;
137         }
138 
139         if(!propertyMeta.getColumnMeta().isUpdatable()) {
140             return false;
141         }
142 
143         if(propertyMeta.isTransient()) {
144             return false;
145         }
146 
147         final String propertyName = propertyMeta.getName();
148         if(query.getIncludesProperties().contains(propertyName)) {
149             return true;
150         }
151 
152         if(query.getExcludesProperties().contains(propertyName)) {
153             return false;
154         }
155 
156         // nullは対象外とするとき
157         if(query.isExcludesNull() && propertyValue == null) {
158             return false;
159         }
160 
161         // 比較対象と同じ値は更新対象外
162         if(query.getBeforeStates() != null) {
163             final Object oldValue = query.getBeforeStates().get(propertyName);
164             if (propertyValue == oldValue) {
165                 return false;
166             }
167             if (propertyValue != null && propertyValue.equals(oldValue)) {
168                 return false;
169             }
170         }
171 
172         // 更新対象が指定されているときは、その他はすべて更新対象外とする。
173         return query.getIncludesProperties().isEmpty();
174 
175     }
176 
177     /**
178      * UPDATE文のWHERE句を準備します。
179      */
180     @SuppressWarnings({"rawtypes", "unchecked"})
181     private void prepareWhereClause() {
182 
183         final SimpleWhereBuilder where = new SimpleWhereBuilder();
184 
185         // WHERE句の準備 - 主キー
186         for(PropertyMeta propertyMeta : query.getEntityMeta().getIdPropertyMetaList()) {
187             String exp = String.format("%s = ?", propertyMeta.getColumnMeta().getName());
188 
189             Object propertyValue = PropertyValueInvoker.getEmbeddedPropertyValue(propertyMeta, query.getEntity());
190             ValueType valueType = propertyMeta.getValueType();
191             Object value = valueType.getSqlParameterValue(propertyValue);
192 
193             where.exp(exp, value);
194         }
195 
196         // WHERE句の準備 - バージョンキー
197         if(!query.isIncludeVersion() && query.getEntityMeta().hasVersionPropertyMeta()) {
198             final PropertyMeta propertyMeta = query.getEntityMeta().getVersionPropertyMeta().get();
199             String exp = String.format("%s = ?", propertyMeta.getColumnMeta().getName());
200 
201             Object propertyValue = PropertyValueInvoker.getEmbeddedPropertyValue(propertyMeta, query.getEntity());
202             ValueType valueType = propertyMeta.getValueType();
203             Object value = valueType.getSqlParameterValue(propertyValue);
204 
205             where.exp(exp, value);
206         }
207 
208         SimpleWhereVisitor visitor = new SimpleWhereVisitor();
209         where.accept(visitor);
210 
211         this.whereClause.addSql(visitor.getCriteria());
212         this.paramValues.addAll(visitor.getParamValues());
213 
214     }
215 
216     /**
217      * 実行するSQLを組み立てます
218      */
219     private void prepareSql() {
220         final String sql = "update "
221                 + query.getEntityMeta().getTableMeta().getFullName()
222                 + setClause.toSql()
223                 + whereClause.toSql();
224 
225         this.executedSql = sql;
226     }
227 
228     /**
229      * 更新処理を実行します。
230      *
231      * @return 更新したレコード件数です。更新対象のプロパティ（カラム）がない場合は {@literal 0} を返します。
232      * @throws OptimisticLockingFailureException 楽観的排他制御を行う場合に該当するレコードが存在しない場合にスローされます。
233      */
234     public int execute() {
235         prepare();
236 
237         if(targetPropertyCount == 0) {
238             log.warn(context.getMessageFormatter().create("query.skipUpdateWithNoProperty").format());
239             return 0;
240         }
241 
242         final int rows = getJdbcTemplate().update(executedSql, paramValues.toArray());
243         if(isOptimisticLock()) {
244             validateRows(rows);
245         }
246 
247         if(!query.isIncludeVersion() && query.getEntityMeta().hasVersionPropertyMeta()) {
248             incrementVresion();
249         }
250 
251         return rows;
252 
253     }
254 
255     /**
256      * {@link JdbcTemplate}を取得します。
257      * @return {@link JdbcTemplate}のインスタンス。
258      */
259     private JdbcTemplate getJdbcTemplate() {
260         return JdbcTemplateBuilder.create(context.getDataSource(), context.getJdbcTemplateProperties())
261                 .queryTimeout(query.getQueryTimeout())
262                 .build();
263     }
264 
265     /**
266      * 楽観的同時実行制御を行っている場合は<code>true</code>を返します。
267      * @return 楽観的同時実行制御を行っている場合は<code>true</code>
268      */
269     private boolean isOptimisticLock() {
270         return !query.isIncludeVersion() && query.getEntityMeta().hasVersionPropertyMeta();
271     }
272 
273     /**
274      * バージョンキーをインクリメントする。
275      */
276     private void incrementVresion() {
277 
278         final Object entity = query.getEntity();
279 
280         PropertyMeta propertyMeta = query.getEntityMeta().getVersionPropertyMeta().get();
281         Object propertyValue = PropertyValueInvoker.getEmbeddedPropertyValue(propertyMeta, entity);
282 
283         propertyValue = NumberConvertUtils.incrementNumber(propertyMeta.getPropertyType(), propertyValue);
284         PropertyValueInvoker.setEmbeddedPropertyValue(propertyMeta, entity, propertyValue);
285 
286     }
287 
288     /**
289      * 更新対象のレコード数のチェック
290      * @param rows 更新したレコード数
291      */
292     private void validateRows(final int rows) {
293         if(!query.isSuppresOptimisticLockException() && rows == 0) {
294             throw new OptimisticLockingFailureException(context.getMessageFormatter().create("query.alreadyUpdate")
295                     .param("entity", query.getEntity())
296                     .format());
297         }
298     }
