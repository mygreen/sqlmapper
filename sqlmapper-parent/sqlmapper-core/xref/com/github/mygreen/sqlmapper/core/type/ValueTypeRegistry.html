51  /**
52   * {@link ValueType} を管理します。
53   *
54   *
55   * @author T.TSUCHIE
56   *
57   */
58  public class ValueTypeRegistry implements InitializingBean {
59  
60      @Getter
61      @Setter
62      @Autowired
63      private ApplicationContext applicationContext;
64  
65      @Getter
66      @Setter
67      @Autowired
68      private MessageFormatter messageFormatter;
69  
70      @Getter
71      @Setter
72      @Autowired
73      private LobHandler lobHandler;
74  
75      /**
76       * クラスタイプで関連付けられた{@link ValueType}のマップ
77       */
78      protected Map<Class<?>, ValueType<?>> typeMap = new ConcurrentHashMap<>();
79  
80      /**
81       * {@inheritDoc}
82       *
83       * <p>初期化処理として、標準の {@link ValueType} を登録する({@link #registerWithDefaultValueTypes()})。
84       */
85      @Override
86      public void afterPropertiesSet() throws Exception {
87          registerWithDefaultValueTypes();
88      }

90      /**
91       * 標準の {@link ValueType} を登録する。
92       */
93      protected void registerWithDefaultValueTypes() {
94  
95          // 文字列用の設定
96          register(String.class, new StringType());
97  
98          // ブール値用の設定
99          register(Boolean.class, new BooleanType(false));
100         register(boolean.class, new BooleanType(true));
101 
102         // 数値用の設定
103         register(Short.class, new ShortType(false));
104         register(short.class, new ShortType(true));
105         register(Integer.class, new IntegerType(false));
106         register(int.class, new IntegerType(true));
107         register(Long.class, new LongType(false));
108         register(long.class, new LongType(true));
109         register(Float.class, new FloatType(false));
110         register(float.class, new FloatType(true));
111         register(Double.class, new DoubleType(false));
112         register(double.class, new DoubleType(true));
113         register(BigDecimal.class, new BigDecimalType());
114 
115         // SQLの時制用の設定
116         // java.util.Date型は、ValueTypeRegistry 内で別途処理される。
117         register(Time.class, new SqlTimeType());
118         register(java.sql.Date.class, new SqlDateType());
119         register(Timestamp.class, new SqlTimestampType());
120 
121         // JSR-310の時勢用の設定
122         // タイムゾーンを持たない時制のみサポート
123         register(LocalTime.class, new LocalTimeType());
124         register(LocalDate.class, new LocalDateType());
125         register(LocalDateTime.class, new LocalDateTimeType());
126 
127         // その他の型の設定
128         register(UUID.class, new UUIDType());
129 
130     }

132     /**
133      * プロパティメタ情報に対する値の変換処理を取得する。
134      * @param propertyMeta プロパティメタ情報
135      * @return 対応する {@link ValueType}の実装。
136      * @throws ValueTypeNotFoundException 対応する {@link ValueType} が見つからない場合。
137      */
138     public ValueType<?> findValueType(@NonNull PropertyMeta propertyMeta) {
139 
140         Optional<Convert> convertAnno = propertyMeta.getAnnotation(Convert.class);
141         if(convertAnno.isPresent()) {
142             return getValueType(propertyMeta, convertAnno.get());
143         }
144 
145         final Class<?> propertyType = propertyMeta.getPropertyType();
146 
147         if(propertyMeta.isLob()) {
148             return getLobType(propertyMeta);
149         }
150 
151         if(typeMap.containsKey(propertyType)) {
152             return typeMap.get(propertyType);
153         }
154 
155         if(propertyType.isEnum()) {
156             return getEnumType(propertyMeta);
157         }
158 
159         if(Date.class.isAssignableFrom(propertyType)) {
160             return getUtilDateType(propertyMeta);
161         }
162 
163         throw new ValueTypeNotFoundException(propertyMeta, messageFormatter.create("typeValue.notFound")
164                 .paramWithClass("entityClass", propertyMeta.getDeclaringClass())
165                 .param("property", propertyMeta.getName())
166                 .paramWithClass("propertyType", propertyType)
167                 .format());
168 
169     }
170 
171     /**
172      * アノテーション {@link Convert} から {@link ValueType} を取得する。
173      * @param propertyMeta 対象となるプロパティメタ情報
174      * @param convertAnno 変換規則を指定するアノテーション
175      * @return {@link ValueType}のインスタンス。
176      */
177     protected ValueType<?> getValueType(final PropertyMeta propertyMeta, final Convert convertAnno) {
178 
179         if(convertAnno.name().isEmpty()) {
180             return BeanUtils.instantiateClass(convertAnno.type());
181         } else {
182             return applicationContext.getBean(convertAnno.name(), convertAnno.type());
183         }
184     }
185 
186     /**
187      * ラージオブジェクト用の{@link ValueType} を取得する。
188      * @param propertyMeta 対象となるプロパティメタ情報
189      * @return {@link ValueType}のインスタンス。
190      */
191     protected ValueType<?> getLobType(final PropertyMeta propertyMeta) {
192 
193         final Class<?> propertyType = propertyMeta.getPropertyType();
194         if(String.class.isAssignableFrom(propertyType)) {
195             return new LobStringType(lobHandler);
196 
197         } else if (byte[].class.isAssignableFrom(propertyType)) {
198             return new LobByteArrayType(lobHandler);
199         }
200 
201         throw new ValueTypeNotFoundException(propertyMeta, messageFormatter.create("typeValue.notFoundLob")
202                 .paramWithClass("entityClass", propertyMeta.getDeclaringClass())
203                 .param("property", propertyMeta.getName())
204                 .paramWithClass("propertyType", propertyType)
205                 .format());
206 
207     }
208 
209     /**
210      * 列挙型用の{@link ValueType} を取得する。
211      * @param propertyMeta 対象となるプロパティメタ情報
212      * @return {@link ValueType}のインスタンス。
213      */
214     @SuppressWarnings({"rawtypes", "unchecked"})
215     protected ValueType<?> getEnumType(final PropertyMeta propertyMeta) {
216 
217         Optional<Enumerated> enumeratedAnno = propertyMeta.getAnnotation(Enumerated.class);
218         Class<?> propertyType = propertyMeta.getPropertyType();
219 
220         if(enumeratedAnno.isPresent()) {
221             final Enumerated.EnumType enumType = enumeratedAnno.get().value();
222             if(enumType == Enumerated.EnumType.ORDINAL) {
223                 return new EnumOrdinalType(propertyType, messageFormatter);
224             } else if(enumType == Enumerated.EnumType.STRING) {
225                 return new EnumStringType(propertyType, messageFormatter);
226             }
227         }
228 
229         // デフォルトの場合
230         return new EnumStringType(propertyType, messageFormatter);
231 
232     }
233 
234     /**
235      * 時制の型が不明な {@link java.util.Date} の ValueType} を取得する。
236      * @param propertyMeta 対象となるプロパティメタ情報
237      * @return {@link ValueType}のインスタンス。
238      */
239     protected ValueType<?> getUtilDateType(final PropertyMeta propertyMeta) {
240 
241         Optional<Temporal> temporalAnno = propertyMeta.getAnnotation(Temporal.class);
242 
243         if(temporalAnno.isPresent()) {
244             final Temporal.TemporalType temporalType = temporalAnno.get().value();
245             if(temporalType == Temporal.TemporalType.TIMESTAMP) {
246                 return new UtilDateType(new SqlTimestampType());
247             } else if(temporalType == Temporal.TemporalType.DATE) {
248                 return new UtilDateType(new SqlDateType());
249 
250             } else if(temporalType == Temporal.TemporalType.TIME) {
251                 return new UtilDateType(new SqlTimeType());
252 
253             }
254         }
255 
256         // デフォルトの場合
257         return new UtilDateType(new SqlTimestampType());
258 
259     }

261     /**
262      * {@link ValueType} を登録します。
263      *
264      * @param <T> 関連付ける型
265      * @param type 関連付けるクラスタイプ
266      * @param valueType {@link ValueType}の実装
267      */
268     public <T> void register(@NonNull Class<T> type, @NonNull ValueType<T> valueType) {
269         this.typeMap.put(type, valueType);
270     }
