80      /**
81       * クラスタイプで関連付けられた{@link ValueType}のマップ
82       */
83      protected Map<Class<?>, ValueType<?>> typeMap = new ConcurrentHashMap<>();
84  
85      /**
86       * {@inheritDoc}
87       *
88       * <p>初期化処理として、標準の {@link ValueType} を登録する({@link #registerWithDefaultValueTypes()})。
89       */
90      @Override
91      public void afterPropertiesSet() throws Exception {
92          registerWithDefaultValueTypes();
93      }

95      /**
96       * 標準の {@link ValueType} を登録する。
97       */
98      protected void registerWithDefaultValueTypes() {
99  
100         // 文字列用の設定
101         register(String.class, new StringType());
102 
103         // ブール値用の設定
104         register(Boolean.class, new BooleanType(false));
105         register(boolean.class, new BooleanType(true));
106 
107         // 数値用の設定
108         register(Short.class, new ShortType(false));
109         register(short.class, new ShortType(true));
110         register(Integer.class, new IntegerType(false));
111         register(int.class, new IntegerType(true));
112         register(Long.class, new LongType(false));
113         register(long.class, new LongType(true));
114         register(Float.class, new FloatType(false));
115         register(float.class, new FloatType(true));
116         register(Double.class, new DoubleType(false));
117         register(double.class, new DoubleType(true));
118         register(BigDecimal.class, new BigDecimalType());
119 
120         // SQLの時制用の設定
121         // java.util.Date型は、ValueTypeRegistry 内で別途処理される。
122         register(Time.class, new SqlTimeType());
123         register(java.sql.Date.class, new SqlDateType());
124         register(Timestamp.class, new SqlTimestampType());
125 
126         // JSR-310の時勢用の設定
127         // タイムゾーンを持たない時制のみサポート
128         register(LocalTime.class, new LocalTimeType());
129         register(LocalDate.class, new LocalDateType());
130         register(LocalDateTime.class, new LocalDateTimeType());
131 
132         // その他の型の設定
133         register(UUID.class, new UUIDType());
134         register(Blob.class, new BLobType());
135         register(Clob.class, new CLobType());
136 
137     }

141      * @param classType 判定対象のクラスタイプ。
142      * @return 登録済みのクラスタイプのとき {@literal true} を返します。

145

147         return typeMap.containsKey(classType);
148 
149     }

151     /**
152      * 登録済みのクラスタイプ対する {@link ValueType} を取得します。
153      * @param classType クラスタイプ。
154      * @return {@link ValueType} を返します。ただし、登録済みでない場合は、{@literal null} を返します。
155      */
156     public ValueType<?> findValueType(@NonNull Class<?> classType) {
157         return typeMap.get(classType);
158     }

160     /**
161      * ストアドのプロパティ情報に対する {@link ValueType} を取得します。
162      * @param propertyMeta
163      * @return 対応する {@link ValueType}の実装を返します。
164      * @throws ValueTypeNotFoundException 対応する {@link ValueType} が見つからない場合。
165      */
166     public ValueType<?> findValueType(@NonNull StoredPropertyMeta propertyMeta) {
167         final Class<?> propertyType = propertyMeta.getPropertyType();
168         if(typeMap.containsKey(propertyType)) {
169             return typeMap.get(propertyType);
170         }
171 
172         throw new ValueTypeNotFoundException(propertyMeta, messageFormatter.create("typeValue.notFound")
173                 .paramWithClass("entityClass", propertyMeta.getDeclaringClass())
174                 .param("property", propertyMeta.getName())
175                 .paramWithClass("propertyType", propertyType)
176                 .format());
177 
178     }
179 
180     /**
181      * エンティティのプロパティメタ情報に対する {@link ValueType} を取得します。
182      * @param propertyMeta プロパティメタ情報
183      * @return 対応する {@link ValueType}の実装を返します。
184      * @throws ValueTypeNotFoundException 対応する {@link ValueType} が見つからない場合。
185      */
186     public ValueType<?> findValueType(@NonNull PropertyMeta propertyMeta) {
187 
188         Optional<Convert> convertAnno = propertyMeta.getAnnotation(Convert.class);
189         if(convertAnno.isPresent()) {
190             return getValueType(propertyMeta, convertAnno.get());
191         }
192 
193         final Class<?> propertyType = propertyMeta.getPropertyType();
194 
195         if(propertyMeta.isLob()) {
196             return getLobType(propertyMeta);
197         }
198 
199         if(typeMap.containsKey(propertyType)) {
200             return typeMap.get(propertyType);
201         }
202 
203         if(propertyType.isEnum()) {
204             return getEnumType(propertyMeta);
205         }
206 
207         if(Date.class.isAssignableFrom(propertyType)) {
208             return getUtilDateType(propertyMeta);
209         }
210 
211         throw new ValueTypeNotFoundException(propertyMeta, messageFormatter.create("typeValue.notFound")
212                 .paramWithClass("entityClass", propertyMeta.getDeclaringClass())
213                 .param("property", propertyMeta.getName())
214                 .paramWithClass("propertyType", propertyType)
215                 .format());
216 
217     }

219     /**
220      * アノテーション {@link Convert} から {@link ValueType} を取得する。
221      * @param propertyMeta 対象となるプロパティメタ情報
222      * @param convertAnno 変換規則を指定するアノテーション
223      * @return {@link ValueType}のインスタンス。
224      */
225     protected ValueType<?> getValueType(final PropertyMeta propertyMeta, final Convert convertAnno) {
226 
227         if(convertAnno.name().isEmpty()) {
228             return BeanUtils.instantiateClass(convertAnno.type());
229         } else {
230             return applicationContext.getBean(convertAnno.name(), convertAnno.type());
231         }
232     }
233 
234     /**
235      * ラージオブジェクト用の{@link ValueType} を取得する。
236      * @param propertyMeta 対象となるプロパティメタ情報
237      * @return {@link ValueType}のインスタンス。
238      */
239     protected ValueType<?> getLobType(final PropertyMeta propertyMeta) {
240 
241         final Class<?> propertyType = propertyMeta.getPropertyType();
242 
243         if(Clob.class.isAssignableFrom(propertyType)) {
244             return new CLobType();
245 
246         } else if(Blob.class.isAssignableFrom(propertyType)) {
247             return new BLobType();
248 
249         } else if(String.class.isAssignableFrom(propertyType)) {
250             return new LobStringType(lobHandler);
251 
252         } else if (byte[].class.isAssignableFrom(propertyType)) {
253             return new LobByteArrayType(lobHandler);
254         }
255 
256         throw new ValueTypeNotFoundException(propertyMeta, messageFormatter.create("typeValue.notFoundLob")
257                 .paramWithClass("entityClass", propertyMeta.getDeclaringClass())
258                 .param("property", propertyMeta.getName())
259                 .paramWithClass("propertyType", propertyType)
260                 .format());
261 
262     }

264     /**
265      * 列挙型用の{@link ValueType} を取得する。
266      * @param propertyMeta 対象となるプロパティメタ情報
267      * @return {@link ValueType}のインスタンス。
268      */
269     @SuppressWarnings({"rawtypes", "unchecked"})
270     protected ValueType<?> getEnumType(final PropertyMeta propertyMeta) {
271 
272         Optional<Enumerated> enumeratedAnno = propertyMeta.getAnnotation(Enumerated.class);
273         Class<?> propertyType = propertyMeta.getPropertyType();
274 
275         if(enumeratedAnno.isPresent()) {
276             final Enumerated.EnumType enumType = enumeratedAnno.get().value();
277             if(enumType == Enumerated.EnumType.ORDINAL) {
278                 return new EnumOrdinalType(propertyType, messageFormatter);
279 
280             } else if(enumType == Enumerated.EnumType.STRING) {
281                 return new EnumStringType(propertyType, messageFormatter);
282 
283             }
284         }
285 
286         // デフォルトの場合
287         return new EnumStringType(propertyType, messageFormatter);
288 
289     }

291     /**
292      * 時制の型が不明な {@link java.util.Date} の ValueType} を取得する。
293      * @param propertyMeta 対象となるプロパティメタ情報
294      * @return {@link ValueType}のインスタンス。
295      */
296     protected ValueType<?> getUtilDateType(final PropertyMeta propertyMeta) {
297 
298         Optional<Temporal> temporalAnno = propertyMeta.getAnnotation(Temporal.class);
299 
300         if(temporalAnno.isPresent()) {
301             final Temporal.TemporalType temporalType = temporalAnno.get().value();
302             if(temporalType == Temporal.TemporalType.TIMESTAMP) {
303                 return new UtilDateType(new SqlTimestampType());
304 
305             } else if(temporalType == Temporal.TemporalType.DATE) {
306                 return new UtilDateType(new SqlDateType());
307 
308             } else if(temporalType == Temporal.TemporalType.TIME) {
309                 return new UtilDateType(new SqlTimeType());
310 
311             }
312         }
313 
314         // デフォルトの場合
315         return new UtilDateType(new SqlTimestampType());
316 
317     }

319     /**
320      * {@link ValueType} を登録します。
321      *
322      * @param <T> 関連付ける型
323      * @param type 関連付けるクラスタイプ
324      * @param valueType {@link ValueType}の実装
325      */
326     public <T> void register(@NonNull Class<T> type, @NonNull ValueType<T> valueType) {
327         this.typeMap.put(type, valueType);
328     }
