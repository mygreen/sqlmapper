1   package com.github.mygreen.sqlmapper.core.dialect;
2   
3   import javax.sql.DataSource;
4   
5   import org.springframework.jdbc.support.incrementer.DataFieldMaxValueIncrementer;
6   import org.springframework.jdbc.support.incrementer.HsqlSequenceMaxValueIncrementer;
7   
8   import com.github.mygreen.sqlmapper.core.annotation.GeneratedValue.GenerationType;
9   
10  /**
11   * HSQL用の方言の定義。
12   *
13   *
14   * @author T.TSUCHIE
15   *
16   */
17  public class HsqlDialect extends DialectBase {
18  
19      /**
20       * {@inheritDoc}
21       *
22       * @return {@literal "hsql"} を返します。
23       */
24      @Override
25      public String getName() {
26          return "hsql";
27      }
28  
29      /**
30       * {@inheritDoc}
31       *
32       * <ul>
33       *  <li>{@link GenerationType#IDENTITY} : {@literal true}</li>
34       *  <li>{@link GenerationType#SEQUENCE} : {@literal true}</li>
35       *  <li>{@link GenerationType#TABLE} : {@literal true}</li>
36       *  <li>{@link GenerationType#UUID} : {@literal true}</li>
37       * </ul>
38       */
39      @Override
40      public boolean supportsGenerationType(GenerationType generationType) {
41          switch(generationType) {
42              case IDENTITY:
43                  return true;
44              case SEQUENCE:
45                  return true;
46              case TABLE:
47                  return true;
48              case UUID:
49                  return true;
50              default:
51                  return false;
52          }
53      }
54  
55      /**
56       * {@inheritDoc}
57       *
58       * @return {@link HsqlSequenceMaxValueIncrementer} のインスタンスを返します。
59       */
60      @Override
61      public DataFieldMaxValueIncrementer getSequenceIncrementer(DataSource dataSource, String sequenceName) {
62          return new HsqlSequenceMaxValueIncrementer(dataSource, sequenceName);
63      }
64  
65  }
