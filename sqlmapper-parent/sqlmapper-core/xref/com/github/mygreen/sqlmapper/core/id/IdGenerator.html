1   package com.github.mygreen.sqlmapper.core.id;
2   
3   import org.springframework.dao.DataIntegrityViolationException;
4   
5   /**
6    * IDを自動生成する処理のインタフェースです。
7    *
8    * @version 0.3
9    * @author T.TSUCHIE
10   *
11   */
12  public interface IdGenerator {
13  
14      /**
15       * サポートしているクラスタイプかどうか
16       * @param type 検査対象のクラスタイプ
17       * @return trueのときサポート対象
18       */
19      boolean isSupportedType(Class<?> type);
20  
21      /**
22       * サポートしているクラスのタイプ一覧を取得する
23       * @return サポートしているクラスのタイプ一覧
24       */
25      Class<?>[] getSupportedTypes();
26  
27      /**
28       * IDを新たに生成します。
29       * @param context 生成対象のIDの情報
30       * @return 生成したID
31       * @throws DataIntegrityViolationException IDの生成に失敗した場合にスローされます。
32       */
33      Object generateValue(IdGenerationContext context);
34  
35      /**
36       * 指定した個数分のIDを新たに生成します。
37       * @param context 生成対象のIDの情報
38       * @param num 生成するIDの個数
39       * @return DataIntegrityViolationException IDの生成に失敗した場合にスローされます。
40       */
41      default Object[] generateValues(final IdGenerationContext context, final int num) {
42          Object[] values = new Object[num];
43          for(int i=0; i < num; i++) {
44              values[i] = generateValue(context);
45          }
46          return values;
47      }
48  
49  }
