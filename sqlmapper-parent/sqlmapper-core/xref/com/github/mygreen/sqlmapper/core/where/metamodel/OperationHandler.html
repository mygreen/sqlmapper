23  /**
24   * 演算子に対する処理を行うためのテンプレートクラス。
25   *
26   *
27   * @author T.TSUCHIE
28   *
29   * @param <T> 処理対象の演算子
30   */
31  public abstract class OperationHandler<T extends Operator> {
32  
33      /**
34       * 演算子に対するテンプレートのマップ
35       * マップの値ととなるテンプレートは{@link MessageFormat}の形式。
36       */
37      protected Map<T, String> templateMap = new HashMap<>();
38  
39      public OperationHandler() {
40          init();
41      }
42  
43      /**
44       * 初期化処理
45       */
46      protected abstract void init();
47  
48      /**
49       * 演算子に対する処理を行います。
50       * @param operator 演算子
51       * @param expr 演算子と非演算子を含む処理対象の式の情報
52       * @param visitor Visitor
53       * @param context このンテキスト
54       */
55      public abstract void handle(T operator, Operation<?> expr, Visitor<VisitorContext> visitor, VisitorContext context);
56  
57      /**
58       * テンプレートを追加します。
59       * @param op 演算子
60       * @param template テンプレート（{@link MessageFormat}の形式。）
61       * @return 既に演算子に対するテンプレートが追加されている場合は、古いテンプレートの値を返します。
62       */
63      public String addTemplate(T op, String template) {
64          return templateMap.put(op, template);
65      }
66  
67      /**
68       * 演算子に対応するテンプレートを取得します。
69       * @param op 演算子
70       * @return 対応するテンプレートが存在しない場合は、nullを返します。
71       */
72      public String getTemplate(T op) {
73          return templateMap.get(op);
74      }
75  
76      /**
77       * テンプレートを使用してフォーマットします。
78       * @param op 演算子
79       * @param args 引数
80       * @return フォーマットした値
81       */
82      public String formatWithTemplate(T op, Object... args) {
83          String template = templateMap.get(op);
84          return MessageFormat.format(template, args);
85      }
86  
87      /**
88       * プロパティが確定しているのとき定数の処理。
89       * @param propertyPath プロパティパス
90       * @param expr 定数
91       * @param context コンテキスト
92       */
93      @SuppressWarnings({"rawtypes", "unchecked"})
94      protected void visitConstantWithPropertyPath(PropertyPath<?> propertyPath, Constant<?> expr, VisitorContext context) {
95  
96          Path<?> rootPath = propertyPath.getPathMeta().findRootPath();
97          Class<?> rootClassType = rootPath.getType();
98          String propertyName = propertyPath.getPathMeta().getElement();
99          Optional<PropertyMeta> propertyMeta = context.getEntityMetaMap().get(rootClassType).findPropertyMeta(propertyName);
100         if(propertyMeta.isEmpty()) {
101             throw new IllegalQueryException("unknwon property : " + propertyName);
102         }
103 
104         ValueType valueType = propertyMeta.get().getValueType();
105 
106         // 値はプレースホルダーを追加
107         if(expr.isExpandable()) {
108             // 展開可能な複数の要素の場合
109             Collection<?> values = (Collection<?>)expr.getValue();
110             for(Object value : values) {
111                 context.addParamValue(valueType.getSqlParameterValue(value));
112             }
113             context.appendSql("(")
114                 .append(QueryUtils.repeat("?", ",", values.size()))
115                 .append(")");
116         } else {
117             context.addParamValue(valueType.getSqlParameterValue(expr.getValue()));
118             context.appendSql("?");
119         }
120 
121     }
122 
123     /**
124      * 各処理に振り分ける
125      * @param parentOperator 親ノードの演算子
126      * @param expr 評価対象の式
127      * @param visitor Visitor
128      * @param context コンテキスト
129      */
130     protected void invoke(Operator parentOperator, Expression<?> expr, Visitor<VisitorContext> visitor, VisitorContext context) {
131 
132         if(expr instanceof Operation) {
133             Operation<?> operation = (Operation<?>)expr;
134             // /子ノードが演算子の場合、括弧で囲むか判定する。
135             if(OperationUtils.isEnclosedParenthesis(parentOperator, operation.getOperator())) {
136                 context.appendSql("(");
137                 visitor.visit(operation, context);
138                 context.appendSql(")");
139             } else {
140                 visitor.visit(operation, context);
141             }
142 
143         } else if(expr instanceof Constant) {
144             visitor.visit((Constant<?>)expr, context);
145         } else if (expr instanceof Path) {
146             visitor.visit((Path<?>)expr, context);
147         } else if(expr instanceof SubQueryExpression) {
148             context.appendSql("(");
149             visitor.visit((SubQueryExpression<?>)expr, context);
150             context.appendSql(")");
151         } else {
152             throw new IllegalArgumentException("not support Expression instance of " + expr.getClass());
153         }
154 
155     }
