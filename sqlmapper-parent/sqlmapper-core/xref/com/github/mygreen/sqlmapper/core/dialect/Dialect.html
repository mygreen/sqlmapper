1   package com.github.mygreen.sqlmapper.core.dialect;
2   
3   import java.sql.ResultSet;
4   import java.util.Map;
5   
6   import javax.sql.DataSource;
7   
8   import org.springframework.jdbc.support.incrementer.DataFieldMaxValueIncrementer;
9   
10  import com.github.mygreen.splate.SqlTemplateEngine;
11  import com.github.mygreen.sqlmapper.core.annotation.GeneratedValue.GenerationType;
12  import com.github.mygreen.sqlmapper.core.query.SelectForUpdateType;
13  import com.github.mygreen.sqlmapper.core.type.ValueType;
14  import com.github.mygreen.sqlmapper.core.where.metamodel.OperationHandler;
15  import com.github.mygreen.sqlmapper.metamodel.operator.Operator;
16  
17  /**
18   * SQLの方言を定義します。
19   *
20   * @version 0.3
21   * @author T.TSUCHIE
22   *
23   */
24  public interface Dialect {
25  
26      /**
27       * 方言の名称を取得します。
28       * <p>この値は、{@link SqlTemplateEngine#setSuffixName(String)} にも使用されます。</p>
29       *
30       * @return 方言の名称。
31       */
32      String getName();
33  
34      /**
35       * {@link GenerationType#AUTO} が指定された場合の、 デフォルトの {@link GenerationType}を返します。
36       * @return {@link GenerationType} を返します。
37       */
38      GenerationType getDefaultGenerationType();
39  
40      /**
41       * サポートする主キーの生成戦略を判定します。
42       * @param generationType 主キーの生成戦略。
43       * @return {@literal true}のときサポートします。
44       */
45      boolean supportsGenerationType(GenerationType generationType);
46  
47      /**
48       * シーケンスをインクリメントする処理を取得します。
49       * @param dataSource データソース
50       * @param sequenceName シーケンス名
51       * @return DBに対応したインクリメントする処理の実装を返します。シーケンスをサポートしない場合は{@literal null}を返します。
52       */
53      DataFieldMaxValueIncrementer getSequenceIncrementer(DataSource dataSource, String sequenceName);
54  
55      /**
56       * 対応するクラスタイプに対する値タイプを返します。
57       * Oracleなどのようにbooleanが存在しない場合は対応する{@link ValueType} に切り替えたりします。
58       *
59       * @param valueType 値タイプ
60       * @return 値タイプ
61       */
62      ValueType<?> getValueType(ValueType<?> valueType);
63  
64      /**
65       * 件数取得用のSQLのSELECT句を取得します。
66       * @return SELECT句
67       */
68      String getCountSql();
69  
70      /**
71       * 件数取得用のSQLに変換します。
72       * @param sql SQL
73       * @return 件数取得用SQL
74       * @throws IllegalArgumentException 引数{@literal sql}が空の場合にスローされます。
75       */
76      String convertGetCountSql(String sql);
77  
78      /**
79       * ヒントコメントを返します。
80       * <p>ヒント句をサポートしていないDBの場合は空文字を返します。</p>
81       *
82       * @param hint ヒント
83       * @return ヒントコメント
84       */
85      String getHintComment(String hint);
86  
87      /**
88       * LIMIT句用SQLに変換します。
89       * @param sql SQL
90       * @param offset オフセット。省略する場合は {@literal -1}を指定します。
91       * @param limit リミット。省略する場合は {@literal -1} を指定します。
92       * @return LIMIT句用SQL。
93       * @throws IllegalArgumentException 引数{@literal offset} と {@literal limit} の値の両方が0より小さい場合にスローされます。
94       */
95      String convertLimitSql(String sql, int offset, int limit);
96  
97      /**
98       * SELECT文で<code>FOR UPDATE</code>をサポートしていれば<code>true</code>を返します。
99       *
100      * @param type SELECT ～ FOR UPDATEのタイプ
101      * @return SELECT文で<code>FOR UPDATE</code>をサポートしていれば<code>true</code>
102      */
103     boolean supportsSelectForUpdate(SelectForUpdateType type);
104 
105     /**
106      * SELECT文に付加する<code>FOR UPDATE NOWAIT</code>相当のSQLを返します。
107      *
108      * @param type SELECT ～ FOR UPDATEのタイプ
109      * @param waitSeconds <code>type</code>に{@link SelectForUpdateType#WAIT} が指定された場合の待機時間(秒単位)
110      * @return SELECT文に付加する<code>FOR UPDATE</code>句のSQL
111      */
112     String getForUpdateSql(SelectForUpdateType type, int waitSeconds);
113 
114     /**
115      * プロシージャの呼び出しで {@link ResultSet} に対してパラメータが必要かどうかを判定します。
116      * @return {@literal true}のとき、{@link ResultSet} に対してパラメータが必要です。
117      */
118     boolean needsParameterForResultSet();
119 
120     /**
121      * メタモデルの条件式を評価する処理のマップを返します。
122      * <p>デフォルトの実装から変更するものみ設定します。
123      *
124      * @since 0.3
125      * @return メタモデルの条件式を評価する処理のマップ
126      */
127     Map<Class<?>, OperationHandler<? extends Operator>> getOperationHandlerMap();
128 }
