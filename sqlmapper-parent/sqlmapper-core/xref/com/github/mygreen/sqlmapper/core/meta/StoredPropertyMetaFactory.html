30   * ストアドプロシージャ／ファンクション
31   *
32   * @since 0.3
33   * @author T.TSUCHIE
34   *
35   */
36  public class StoredPropertyMetaFactory {
37  
38      @Getter
39      @Setter
40      @Autowired
41      private ValueTypeRegistry valueTypeRegistry;
42  
43      @Getter
44      @Setter
45      @Autowired
46      private NamingRule namingRule;
47  
48      /**
49       * フィールド情報を元に、ストアド用のエンティティ形式のパラメータのプロパティ情報を作成します。
50       *
51       * @param field フィールド情報
52       * @return プロパティ情報
53       */
54      public StoredPropertyMeta create(final Field field) {
55  
56          final StoredPropertyMeta propertyMeta = new StoredPropertyMeta(field.getName(), field.getType());
57          doField(propertyMeta, field);
58  
59          // フィールドに対するgetter/setterメソッドを設定します。
60          final Class<?> declaringClass = field.getDeclaringClass();
61          for(Method method : declaringClass.getMethods()) {
62              ReflectionUtils.makeAccessible(method);
63  
64              int modifiers = method.getModifiers();
65              if(Modifier.isStatic(modifiers)) {
66                  continue;
67              }
68  
69              if(ClassUtils.isSetterMethod(method)) {
70                  doSetterMethod(propertyMeta, method);
71  
72              } else if(ClassUtils.isGetterMethod(method) || ClassUtils.isBooleanGetterMethod(method)) {
73                  doGetterMethod(propertyMeta, method);
74              }
75          }
76  
77          if(propertyMeta.isIn() || propertyMeta.isOut() || propertyMeta.isInOut()) {
78  
79              // プロパティに対する型変換を設定します。
80              ValueType<?> valueType = valueTypeRegistry.findValueType(propertyMeta);
81  
82              //TODO: Oracleなどの純粋なタイプ
83              propertyMeta.setValueType(valueType);
84              propertyMeta.setSingleValue(true);
85  
86              // パラメータ名の設定
87              Optional<String> paramName = propertyMeta.getAnnotation(In.class).map(a -> a.name())
88                      .or(() -> propertyMeta.getAnnotation(Out.class).map(a -> a.name()))
89                      .or(() -> propertyMeta.getAnnotation(InOut.class).map(a -> a.name()));
90              propertyMeta.setParamName(paramName.map(p -> StringUtils.hasLength(p) ? p : namingRule.propertyToStoredParam(propertyMeta.getName()))
91                      .orElseGet(() -> namingRule.propertyToStoredParam(propertyMeta.getName())));
92  
93  
94          } else if(propertyMeta.isResultSet()) {
95              // ResultSetの場合
96              if(valueTypeRegistry.isRegisteredType(propertyMeta.getPropertyType())) {
97                  // 通常の値である場合
98                  ValueType<?> valueType = valueTypeRegistry.findValueType(propertyMeta);
99  
100                 //TODO: Oracleなどの純粋なタイプ
101                 propertyMeta.setValueType(valueType);
102                 propertyMeta.setSingleValue(true);
103 
104             } else {
105                 // JavaBeanの場合
106                 // JavaBeanのプロパティの抽出は、StoredParamMetaFactoryで行う。
107                 propertyMeta.setSingleValue(false);
108 
109                 // 総称型の設定
110                 doComponentType(propertyMeta, field);
111             }
112 
113             // パラメータ名の設定
114             Optional<String> paramName = propertyMeta.getAnnotation(ResultSet.class).map(a -> a.name());
115             propertyMeta.setParamName(paramName.map(p -> StringUtils.hasLength(p) ? p : namingRule.propertyToStoredParam(propertyMeta.getName()))
116                     .orElseGet(() -> namingRule.propertyToStoredParam(propertyMeta.getName())));
117 
118 
119         } else {
120             // ResultSetのネストしたプロパティの場合
121             ValueType<?> valueType = valueTypeRegistry.findValueType(propertyMeta);
122             propertyMeta.setValueType(valueType);
123             propertyMeta.setSingleValue(true);
124 
125             // パラメータ名の設定
126             Optional<String> paramName = propertyMeta.getAnnotation(Column.class).map(a -> a.name());
127             propertyMeta.setParamName(paramName.map(p -> StringUtils.hasLength(p) ? p : namingRule.propertyToColumn(propertyMeta.getName()))
128                     .orElseGet(() -> namingRule.propertyToColumn(propertyMeta.getName())));
129 
130         }
131 
132         return propertyMeta;
133 
134     }
135 
136     /**
137      * プロパティのメタ情報に対する処理を実行します。
138      * @param propertyMeta プロパティのメタ情報
139      * @param field フィールド情報
140      */
141     private void doField(final StoredPropertyMeta propertyMeta, final Field field) {
142 
143         propertyMeta.setField(field);
144 
145         final Annotation[] annos = field.getAnnotations();
146         for(Annotation anno : annos) {
147             if(!isSupportedAnnotation(anno)) {
148                 continue;
149             }
150 
151             final Class<? extends Annotation> annoClass = anno.annotationType();
152             propertyMeta.addAnnotation(annoClass, anno);
153         }
154     }
155 
156     /**
157      * サポートするアノテーションか判定する。
158      * <p>確実に重複するJava標準のアノテーションは除外するようにします。</p>
159      *
160      * @param anno 判定対象のアノテーション
161      * @return tureのときサポートします。
162      */
163     private boolean isSupportedAnnotation(final Annotation anno) {
164 
165         final String name = anno.annotationType().getName();
166         if(name.startsWith("java.lang.annotation.")) {
167             return false;
168         }
169 
170         return true;
171     }
172 
173     /**
174      * setterメソッドの情報を処理する。
175      * @param propertyMeta プロパティのメタ情報
176      * @param method setterメソッド
177      */
178     private void doSetterMethod(final StoredPropertyMeta propertyMeta, final Method method) {
179 
180         final String methodName = method.getName();
181         final String propertyName = NameUtils.uncapitalize(methodName.substring(3));
182 
183         if(!propertyMeta.getName().equals(propertyName)) {
184             // プロパティ名が一致しない場合はスキップする
185             return;
186         }
187 
188         propertyMeta.setWriteMethod(method);
189 
190     }
191 
192     /**
193      * getterメソッドの情報を処理する。
194      * @param propertyMeta プロパティのメタ情報
195      * @param method getterメソッド
196      */
197     private void doGetterMethod(final StoredPropertyMeta propertyMeta, final Method method) {
198 
199         final String methodName = method.getName();
200         final String propertyName;
201         if(methodName.startsWith("get")) {
202             propertyName = NameUtils.uncapitalize(methodName.substring(3));
203         } else {
204             // 「is」から始まる場合
205             propertyName = NameUtils.uncapitalize(methodName.substring(2));
206         }
207 
208         if(!propertyMeta.getName().equals(propertyName)) {
209             // プロパティ名が一致しない場合はスキップする
210             return;
211         }
212 
213         propertyMeta.setReadMethod(method);
214 
215     }
216 
217     /**
218      * JavaBean型の場合、総称型を設定します。
219      * @param propertyMeta プロパティのメタ情報
220      * @param field フィールド情報
221      */
222     private void doComponentType(final StoredPropertyMeta propertyMeta, final Field field) {
223 
224         Class<?> fieldType = field.getType();
225         if(Collection.class.isAssignableFrom(fieldType)) {
226             ParameterizedType type = (ParameterizedType) field.getGenericType();
227             propertyMeta.setComponentType((Class<?>)type.getActualTypeArguments()[0]);
228 
229         } else if(fieldType.isArray()) {
230             propertyMeta.setComponentType(fieldType.getComponentType());
231 
232         }
233 
234     }
