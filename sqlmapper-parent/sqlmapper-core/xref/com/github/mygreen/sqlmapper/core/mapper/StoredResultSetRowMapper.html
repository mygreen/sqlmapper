18  /**
19   * ストアドプロシージャ／ファンクションのResultSetをBeanにマッピングします。
20   *
21   *
22   * @since 0.3
23   * @param <T> マッピング先のクラスタイプ
24   *
25   * @author T.TSUCHIE
26   */
27  @Slf4j
28  @RequiredArgsConstructor
29  public class StoredResultSetRowMapper<T> implements RowMapper<T> {
30  
31      private final StoredPropertyMeta propertyMeta;
32  
33      @SuppressWarnings("unchecked")
34      @Override
35      public T mapRow(final ResultSet rs, final int rowNum) throws SQLException {
36  
37          // マッピングするエンティティのインスタンスの準備
38          final T bean = BeanUtils.instantiateClass((Class<T>)propertyMeta.getComponentType().get());
39  
40          final ResultSetMetaData rsmd = rs.getMetaData();
41          final int columnCount = rsmd.getColumnCount();
42          for(int i=0; i < columnCount; i++) {
43              String columnName = rsmd.getColumnName(i+1);
44  
45              Optional<PropertyMeta> nestedPropertyMeta = propertyMeta.findNestedColumnPropertyMeta(columnName);
46              if(nestedPropertyMeta.isEmpty()) {
47                  if(log.isDebugEnabled()) {
48                      log.debug("not found nestedPropertyMeta for column name '{}' in {}.", columnName, propertyMeta.getDeclaringClass().getSimpleName());
49                  }
50                  continue;
51              }
52  
53              Object propertyValue = nestedPropertyMeta.get().getValueType().getValue(rs, i+1);
54              PropertyValueInvoker.setEmbeddedPropertyValue(nestedPropertyMeta.get(), bean, propertyValue);
55          }
56  
57          return bean;
58      }
59  
60  }
