175             entities.add(entity);

185

265     @Test
266     void testAutoAnyDelete() {
267 
268         MInheritanceTestEntity m_ = MInheritanceTestEntity.testInheritance;
269 
270         long beforeCount = sqlMapper.selectFrom(m_)
271                     .where(m_.createAt.goe(Timestamp.valueOf("2021-01-03 00:00:00")))
272                     .getCount();
273 
274         int count = txNew().execute(action -> {
275             return sqlMapper.deleteFrom(m_)
276                     .where(m_.createAt.goe(Timestamp.valueOf("2021-01-03 00:00:00")))
277                     .execute();
278         });
279 
280         assertThat(count).isEqualTo(beforeCount);
281 
282         sqlMapper.selectFrom(m_).getResultList().forEach(entity -> {
283             assertThat(entity.getCreateAt()).isBefore(Timestamp.valueOf("2021-01-03 00:00:00"));
284         });
