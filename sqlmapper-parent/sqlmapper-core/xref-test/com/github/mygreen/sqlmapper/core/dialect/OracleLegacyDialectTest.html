25      @Test
26      void testConvertLimitSql() {
27          String select = "select * from sample";
28          assertThatThrownBy(() -> convertLimitSql(select, -1, -1)).isInstanceOf(IllegalArgumentException.class);
29          assertThat(convertLimitSql(select, -1, 5)).isEqualTo("select * from ( select temp_.*, rownum rownumber_ from ( select * from sample ) temp_ ) where rownumber_ <= 5");
30          assertThat(convertLimitSql(select, 10, -1)).isEqualTo("select * from ( select temp_.*, rownum rownumber_ from ( select * from sample ) temp_ ) where rownumber_ >= 11");
31          assertThat(convertLimitSql(select, 10, 5)).isEqualTo("select * from ( select temp_.*, rownum rownumber_ from ( select * from sample ) temp_ ) where rownumber_ between 11 and 15");
32  
33      }
