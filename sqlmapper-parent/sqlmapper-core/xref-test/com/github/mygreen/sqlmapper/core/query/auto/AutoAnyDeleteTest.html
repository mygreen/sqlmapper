26  @ExtendWith(SpringExtension.class)
27  @ContextConfiguration(classes=H2TestConfig.class)
28  public class AutoAnyDeleteTest extends QueryTestSupport {
29  
30      @Autowired
31      SqlMapper sqlMapper;
32  
33      @BeforeEach
34      void beforeMethod() {
35          resetData();
36          executeSqlFileAndCommit("insert_data_customer.sql");
37      }
38  
39      @Test
40      void testAnyDeleteAll() {
41  
42          MCustomer m_ = MCustomer.customer;
43  
44          long beforeCount = sqlMapper.selectFrom(m_)
45                  .getCount();
46  
47          int count = txNew().execute(action -> {
48              return sqlMapper.deleteFrom(m_)
49                  .execute();
50          });
51  
52          assertThat(count).isEqualTo(beforeCount);
53  
54          long afterCount = sqlMapper.selectFrom(m_)
55                  .getCount();
56  
57          assertThat(afterCount).isEqualTo(0);
58  
59      }
60  
61      @Test
62      void testWhere() {
63  
64          MCustomer m_ = MCustomer.customer;
65  
66  
67          long beforeCount = sqlMapper.selectFrom(m_)
68                  .where(m_.genderType.eq(GenderType.MALE))
69                  .getCount();
70  
71          int count = txNew().execute(action -> {
72              return sqlMapper.deleteFrom(m_)
73                  .where(m_.genderType.eq(GenderType.MALE))
74                  .execute();
75          });
76  
77          assertThat(count).isEqualTo(beforeCount);
78  
79          sqlMapper.selectFrom(m_).getResultList().forEach(entity -> {
80              assertThat(entity).hasFieldOrPropertyWithValue("genderType", GenderType.FEMALE);
81          });
82  
83      }
