51      @Test
52      void testConvertLimitSql() {
53          String select = "select * from sample";
54          assertThatThrownBy(() -> convertLimitSql(select, -1, -1)).isInstanceOf(IllegalArgumentException.class);
55          assertThat(convertLimitSql(select, -1, 5)).isEqualTo("select * from sample fetch first 5 rows only");
56          assertThat(convertLimitSql(select, 10, -1)).isEqualTo("select * from sample offset 10 rows");
57          assertThat(convertLimitSql(select, 10, 5)).isEqualTo("select * from sample offset 10 rows fetch first 5 rows only");
58      }
