6   import com.github.mygreen.sqlmapper.metamodel.Visitor;

11  import lombok.NonNull;

13  /**
14   * {@link Date} を値に持つ式の演算操作を表現します。
15   *
16   *
17   * @author T.TSUCHIE
18   *
19   */
20  public class UtilDateOperation extends UtilDateExpression implements Operation<Date> {
21  
22      private final OperationMixin<Date> opMixin;
23  
24      public UtilDateOperation(OperationMixin<Date> mixin) {
25          super(mixin);
26          this.opMixin = mixin;
27      }
28  
29      public UtilDateOperation(@NonNull Operator op, Expression<?>... args) {
30          this(new OperationMixin<>(Date.class, op, args));
31      }
32  
33      @Override
34      public <C> void accept(Visitor<C> visitor, C context) {
35          visitor.visit(opMixin, context);
36      }
37  
38      @Override
39      public Operator getOperator() {
40          return opMixin.getOperator();
41      }
42  
43      @Override
44      public Expression<?> getArg(int index) {
45          return opMixin.getArg(index);
46      }
47  
48      @Override
49      public List<Expression<?>> getArgs() {
50          return opMixin.getArgs();
51      }
52  }
