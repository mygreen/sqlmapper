8   /**
9    * エンティティのメタモデルのベースクラス。
10   * <p>エンティティのメタモデルは基本的にこのクラスを実装して作成します。
11   *
12   *
13   * @author T.TSUCHIE
14   *
15   * @param <T> エンティティのクラスタイプ
16   */
17  public abstract class EntityPathBase<T> implements EntityPath<T> {
18  
19      /**
20       * エンティティタイプ
21       */
22      @Getter
23      private final Class<? extends T> type;
24  
25      /**
26       * エンティティのメタ情報
27       */
28      @Getter
29      private final PathMeta pathMeta;
30  
31      /**
32       * プロパティのインスタンスのマップ
33       */
34      private final Map<String, PropertyPath<?>> propertyPathMap = new HashMap<>();
35  
36      /**
37       * 埋め込みプロパティ（エンティティ）のインスタンスのマップ
38       */
39      private final Map<String, EntityPath<?>> embeddedPropertyPathMap = new HashMap<>();
40  
41      /**
42       * エンティティのメタモデルのインスタンスを作成します。
43       *
44       * @param type エンティティのタイプ。
45       * @param name エンティティ名。
46       */
47      public EntityPathBase(final Class<? extends T> type, final String name) {
48          this.type = type;
49          this.pathMeta = new PathMeta(name, PathType.ROOT);
50      }
51  
52      /**
53       * 親を持つエンティティのメタモデルのインスタンスを作成します。
54       *
55       * @param type エンティティのタイプ
56       * @param parent 親の情報
57       * @param name エンティティ名
58       */
59      public EntityPathBase(final Class<? extends T> type, final EntityPathBase<?> parent, final String name) {
60          this.type = type;
61          this.pathMeta = new PathMeta(parent, name, PathType.EMBEDDED);
62  
63          // 親の埋め込み情報に追加する
64          parent.embeddedPropertyPathMap.put(name, this);
65      }
66  
67      @Override
68      public PropertyPath<?> getPropertyPath(final String propertyName) {
69          return propertyPathMap.get(propertyName);
70      }
71  
72      @Override
73      public PropertyPath<?> findPropertyPath(final String propertyName) {
74          PropertyPath<?> propertyPath = propertyPathMap.get(propertyName);
75          if(propertyPath != null) {
76              return propertyPath;
77          }
78  
79          // 埋め込みプロパティの中を探す
80          for(EntityPath<?> entityPath : embeddedPropertyPathMap.values()) {
81              propertyPath = entityPath.getPropertyPath(propertyName);
82              if(propertyPath != null) {
83                  return propertyPath;
84              }
85          }
86          return null;
87      }
88  
89      /**
90       * 文字列型のプロパティを作成します。
91       * @param property プロパティ名
92       * @return 文字列型のプロパティ。
93       */
94      protected StringPath createString(final String property) {
95          StringPath propertyPath = new StringPath(this, property);
96          propertyPathMap.put(property, propertyPath);
97          return propertyPath;
98      }
99
