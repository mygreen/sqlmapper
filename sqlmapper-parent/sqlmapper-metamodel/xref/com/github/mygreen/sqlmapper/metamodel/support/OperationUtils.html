10   * 演算操作に関するユーティリティ
11   *
12   *
13   * @author T.TSUCHIE

15   */
16  public class OperationUtils {
17  
18      /**
19       * 式の評価結果を括弧で囲むか判定する。
20       * @param parentOp 親ノードの演算子
21       * @param childOp 子ノードの演算子
22       * @return 括弧で囲むときは {@literal true} を返す。
23       */
24      public static boolean isEnclosedParenthesis(Operator parentOp, Operator childOp) {
25  
26          if(parentOp.getPriority() < childOp.getPriority()) {
27              // 親ノードの演算子の優先度が高い(値が小さい)とき
28              return true;
29          } else if(parentOp == BooleanOp.OR && childOp == BooleanOp.AND) {
30              // 親ノードがORで子ノードがANDのとき
31              // 実際は括弧で囲まなくてもよいが見やすさのために囲む
32              return true;
33          } else if(parentOp instanceof BooleanOp && childOp == ComparisionOp.BETWEEN) {
34              // 親ノードがOR/ANDで子ノードがBETWEENのとき
35              // 実際は括弧で囲まなくてもよいが見やすさのために囲む
36              return true;
37          } else if(childOp instanceof ArithmeticOp) {
38              // 子ノードが算術演算子の場合、必ず囲む。
39              return true;
40          }
41  
42          return false;
43      }
44  
45      /**
46       * 式ノードをデバッグ情報形式の文字列に変換する。
47       * @param exp 式ノード
48       * @return デバッグ情報
49       */
50      public static String toDebugString(final Expression<?> exp) {
51  
52          DebugVisitor visitor = new DebugVisitor();
53          DebugVisitorContext context = new DebugVisitorContext();
54          exp.accept(visitor, context);
55  
56          return context.getCriteria();
57  
58      }
59  }
