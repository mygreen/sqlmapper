5   /**
6    * ブーリアン型のプロパティを表現します。
7    *
8    * @author T.TSUCHIE
9    *
10   */
11  public class BooleanPath extends BooleanExpression implements PropertyPath<Boolean> {
12  
13      private final PathMixin<Boolean> pathMixin;
14  
15      /**
16       * プロパティの情報を指定してインスタンスを作成する。
17       * @param mixin プロパティの情報
18       */
19      protected BooleanPath(PathMixin<Boolean> mixin) {
20          super(mixin);
21          this.pathMixin = mixin;
22      }
23  
24      /**
25       * プロパティが所属するエンティティの情報とプロパティ名を指定してインスタンスを作成する。
26       * @param parent プロパティが属するエンティティのパス情報。
27       * @param propertyName プロパティ名
28       */
29      public BooleanPath(Path<?> parent, String propertyName) {
30          this(new PathMixin<>(Boolean.class, PathMetaUtils.forProperty(parent, propertyName)));
31      }
32  
33      @Override
34      public <C> void accept(Visitor<C> visitor, C context) {
35          visitor.visit(this, context);
36      }
37  
38      @Override
39      public PathMeta getPathMeta() {
40          return pathMixin.getPathMeta();
41      }
42  
43  }
