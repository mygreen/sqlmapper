1   package com.github.mygreen.sqlmapper.metamodel.support;

3   import com.github.mygreen.sqlmapper.metamodel.EntityPath;
4   import com.github.mygreen.sqlmapper.metamodel.PropertyPath;
5   import com.github.mygreen.sqlmapper.metamodel.expression.SubQueryExpression;
6   import com.github.mygreen.sqlmapper.metamodel.operation.SubQueryMeta;
7   import com.github.mygreen.sqlmapper.metamodel.operation.SubQueryOperation;
8   
9   /**
10   * サブクエリ式を組み立てるときのヘルパークラス。
11   *
12   * @author T.TSUCHIE
13   *
14   */
15  public class SubQueryHelper {
16  
17      /**
18       * 抽出対象のエンティティを指定して、サブクエリのインスタンスを作成します。
19       *
20       * @param entityPath 抽出対象のテーブルのエンティティ
21       * @return サブクエリの式
22       */
23      @SuppressWarnings("rawtypes")
24      public static SubQueryExpression<?> from(final EntityPath<?> entityPath) {
25          SubQueryMeta queryMeta = new SubQueryMeta();
26          queryMeta.setEntityPath(entityPath);
27          return new SubQueryOperation(queryMeta);
28      }
29  
30      /**
31       * 抽出対象のエンティティ（テーブル）とプロパティ（カラム）指定して、サブクエリのインスタンスを作成します。
32       *
33       * @param <T> 抽出対象のプロパティ（カラム）の型
34       * @param entityPath 抽出対象のテーブルのエンティティ
35       * @param include 抽出対象のカラムのプロパティ
36       * @return サブクエリの式
37       */
38      public static <T> SubQueryExpression<T> from(final EntityPath<?> entityPath, final PropertyPath<T> include) {
39          SubQueryMeta queryMeta = new SubQueryMeta();
40          queryMeta.setEntityPath(entityPath);
41          queryMeta.addInclude(include);
42          return new SubQueryOperation<>(queryMeta);
43      }
44  
45  
46  }
