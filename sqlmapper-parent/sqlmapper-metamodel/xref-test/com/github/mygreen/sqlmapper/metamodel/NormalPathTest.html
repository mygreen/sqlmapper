26      @DisplayName("chek meta info.")
27      @Test
28      void testPathMeta() {
29  
30          // エンティティのチェック
31          MSampleEntity entity = MSampleEntity.sampleEntity;
32  
33          assertThat(entity.getType()).isEqualTo(SampleEntity.class);
34          assertThat(entity.getPathMeta().getParent()).isNull();
35          assertThat(entity.getPathMeta().getElement()).isEqualTo("sampleEntity");
36          assertThat(entity.getPathMeta().getType()).isEqualTo(PathType.ROOT);
37  
38          // プロパティのチェック
39          {
40              PropertyPath<String> prop = entity.name;
41              assertThat(prop.getType()).isEqualTo(String.class);
42              assertThat(prop.getPathMeta().getParent()).isEqualTo(entity);
43              assertThat(prop.getPathMeta().getElement()).isEqualTo("name");
44              assertThat(prop.getPathMeta().getType()).isEqualTo(PathType.PROPERTY);
45  
46          }
47  
48      }
49  
50      @Test
51      void testBuildCondition() {
52  
53          MSampleEntity entity = MSampleEntity.sampleEntity;
54          Predicate exp = entity.name.lower().contains("yamada")
55                  .and(entity.age.add(10).gt(20))
56                  .and(entity.role.in(Role.Admin, Role.Normal))
57                  .and(entity.updateAt.after(SqlTimestampExpression.currentTimestamp()))
58                  .and(entity.deleted.isFalse())
59                  .and(entity.salary.goe(new BigDecimal("1000000")))
60                  ;
61  
62          String resultString = exp.toString();
63          assertThat(resultString).isEqualTo("contains(lower(sampleEntity.name), 'yamada') and (sampleEntity.age + 10) > 20 and sampleEntity.role in (Admin, Normal) and sampleEntity.updateAt > current_timestamp and sampleEntity.deleted = false and sampleEntity.salary >= 1000000");
64  
65      }
66  
67      @Test
68      void testBuildOrder() {
69          MSampleEntity entity = MSampleEntity.sampleEntity;
70  
71          List<OrderSpecifier> orders = List.of(entity.name.desc(), entity.role.asc());
72          String resultString = orders.stream()
73                  .map(o -> o.toString())
74                  .collect(Collectors.joining(", "));
75  
76          assertThat(resultString).isEqualTo("sampleEntity.name DESC, sampleEntity.role ASC");
77  
78      }
79  
80      @Test
81      void testCustomFunction() {
82          MSampleEntity entity = MSampleEntity.sampleEntity;
83  
84          Predicate exp = entity.name.function("custom_format($left, 'yyyMMdd')").returnString()
85                  .function("custom_is_valid($left, ?)", "TEST").returnBoolean();
86  
87          String resultString = exp.toString();
88          assertThat(resultString).isEqualTo("custom_is_valid(custom_format(sampleEntity.name, 'yyyMMdd'), 'TEST')");
89  
90      }
