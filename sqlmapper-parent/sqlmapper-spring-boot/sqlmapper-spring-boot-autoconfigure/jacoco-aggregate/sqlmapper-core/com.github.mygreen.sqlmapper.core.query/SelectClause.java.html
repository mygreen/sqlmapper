package com.github.mygreen.sqlmapper.core.query;

/**
 * select句を組み立てるクラスです。
 *
 * @author higa
 *
 */
public class SelectClause {

    /**
     * SQLです。
     */
    private StringBuilder sql;

    /**
     * {@link SelectClause}を作成します。
     *
     */
    public SelectClause() {
        this(300);
    }

    /**
     * {@link SelectClause}を作成します。
     *
     * @param capacity
     *            初期容量
     */
    public SelectClause(int capacity) {
        sql = new StringBuilder(capacity);
    }

    /**
     * SQLの長さを返します。
     *
     * @return SQLの長さ
     */
    public int getLength() {
        return sql.length();
    }

    /**
     * SQLに変換します。
     *
     * @return SQL
     */
    public String toSql() {
        return sql.toString();
    }

    /**
     * select句を追加します。
     *
     * @param tableAlias テーブル別名
     * @param columnName カラム名
     */
    public void addSql(String tableAlias, String columnName) {
        if (sql.length() > 0) {
            sql.append(", ");
        }

        sql.append(tableAlias + "." + columnName);
    }

    /**
     * select句を追加します。
     *
     * @param selectItem セレクト項目
     */
    public void addSql(String selectItem) {
        if (sql.length() > 0) {
            sql.append(", ");
        }
        sql.append(selectItem);
    }

}
