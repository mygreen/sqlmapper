11  /**
12   * {@literal CONCAT} 関数を処理します。
13   *
14   * @since 0.3
15   * @author T.TSUCHIE
16   *
17   */
18  public class ConcatFunction implements SqlFunction {
19  
20      @Override
21      public void handle(List<Expression<?>> args, Visitor<VisitorContext> visitor, VisitorContext context,
22              ExpressionEvaluator evaluator) {
23  
24          // 左辺の評価
25          Expression<?> left = args.get(0);
26          VisitorContext leftContext = new VisitorContext(context);
27          evaluator.evaluate(left, visitor, leftContext);
28  
29          // 右辺の評価
30          Expression<?> right = args.get(1);
31          VisitorContext rightContext = new VisitorContext(context);
32          evaluator.evaluate(right, visitor, rightContext);
33  
34          // 評価した結果を親のコンテキストに追加する
35          context.appendSql("concat(")
36              .append(leftContext.getCriteria())
37              .append(", ")
38              .append(rightContext.getCriteria())
39              .append(")");
40  
41          context.addParamValues(leftContext.getParamValues());
42          context.addParamValues(rightContext.getParamValues());
43  
44      }
45  }
