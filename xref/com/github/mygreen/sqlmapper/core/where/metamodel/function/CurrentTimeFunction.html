12  /**
13   * {@literal CURRENT_TIME} 関数を処理します。
14   *
15   * @since 0.3
16   * @author T.TSUCHIE
17   *
18   */
19  public class CurrentTimeFunction implements SqlFunction {
20  
21      @Override
22      public void handle(List<Expression<?>> args, Visitor<VisitorContext> visitor, VisitorContext context,
23              ExpressionEvaluator invoker) {
24  
25          context.appendSql("current_time");
26          doPrecision(args, context);
27  
28      }
29  
30      /**
31       * 時間の制度が指定されていれば処理を行う。
32       * @param expr 演算子の式
33       * @param context コンテキスト
34       */
35      @SuppressWarnings("unchecked")
36      private void doPrecision(List<Expression<?>> args, VisitorContext context) {
37  
38          if(args.isEmpty()) {
39              return;
40          }
41  
42          int precision = (int)((Constant<Integer>)args.get(0)).getValue();
43          context.appendSql("(").append(precision).append(")");
44  
45      }
46  
47  }
