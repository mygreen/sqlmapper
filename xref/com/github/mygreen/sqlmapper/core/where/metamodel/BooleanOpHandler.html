8   /**
9    * ブール演算子({@link BooleanOp})に対する処理を定義します。
10   *
11   *
12   * @author T.TSUCHIE
13   *
14   */
15  public class BooleanOpHandler extends OperationHandler<BooleanOp> {
16  
17      @Override
18      protected void init() {
19          addTemplate(BooleanOp.AND, "{0} and {1}");
20          addTemplate(BooleanOp.OR, "{0} or {1}");
21      }
22  
23      @Override
24      public void handle(BooleanOp operator, Operation<?> expr, Visitor<VisitorContext> visitor, VisitorContext context) {
25  
26          Expression<?> left = expr.getArg(0);
27          Expression<?> right = expr.getArg(1);
28  
29          VisitorContext leftContext = new VisitorContext(context);
30          VisitorContext rightContext = new VisitorContext(context);
31  
32          // 左辺の評価
33          invoke(operator, left, visitor, leftContext);
34          invoke(operator, right, visitor, rightContext);
35  
36          // 評価した結果を親のコンテキストに追加する
37          String sql = formatWithTemplate(operator, leftContext.getCriteria(), rightContext.getCriteria());
38          context.appendSql(sql);
39  
40          context.addParamValues(leftContext.getParamValues());
41          context.addParamValues(rightContext.getParamValues());
42  
43      }
44  
45  
46  }
