89      /**
90       * プロパティのメタ情報を作成します。
91       * @param field フィールド
92       * @param entityMeta エンティティのメタ情報
93       * @return プロパティのメタ情報
94       */
95      public PropertyMeta create(final Field field, final EntityMeta entityMeta) {
96  
97          final Class<?> declaringClass = field.getDeclaringClass();
98          final PropertyMeta propertyMeta = new PropertyMeta(field.getName(), field.getType());
99          doField(propertyMeta, field);
100 
101         // フィールドに対するgetter/setterメソッドを設定します。
102         for(Method method : declaringClass.getMethods()) {
103             ReflectionUtils.makeAccessible(method);
104 
105             int modifiers = method.getModifiers();
106             if(Modifier.isStatic(modifiers)) {
107                 continue;
108             }
109 
110             if(ClassUtils.isSetterMethod(method)) {
111                 doSetterMethod(propertyMeta, method);
112 
113             } else if(ClassUtils.isGetterMethod(method) || ClassUtils.isBooleanGetterMethod(method)) {
114                 doGetterMethod(propertyMeta, method);
115             }
116         }
117 
118         // 永続化対象のプロパティはカラム情報を設定します。
119         if(!propertyMeta.isEmbedded() && !propertyMeta.isTransient()) {
120 
121             doColumnMeta(propertyMeta);
122             doIdGenerator(propertyMeta, entityMeta);
123 
124             // プロパティに対する型変換を設定します。
125             ValueType<?> valueType = valueTypeRegistry.findValueType(propertyMeta);
126 
127             // OracleなどBoolean型を純粋にサポートしていない場合は、int型に変換するタイプに変換する。
128             valueType = dialect.getValueType(valueType);
129             propertyMeta.setValueType(valueType);
130 
131 
132             validateColumnProperty(declaringClass, propertyMeta);
133 
134         }
135 
136         return propertyMeta;
137 
138     }

140     /**
141      * プロパティのメタ情報に対する処理を実行します。
142      * @param propertyMeta プロパティのメタ情報
143      * @param field フィールド情報
144      */
145     private void doField(final PropertyMeta propertyMeta, final Field field) {
146 
147         propertyMeta.setField(field);
148 
149         final Annotation[] annos = field.getAnnotations();
150         for(Annotation anno : annos) {
151             if(!isSupportedAnnotation(anno)) {
152                 continue;
153             }
154 
155             final Class<? extends Annotation> annoClass = anno.annotationType();
156             propertyMeta.addAnnotation(annoClass, anno);
157         }
158     }

160     /**
161      * サポートするアノテーションか判定する。
162      * <p>確実に重複するJava標準のアノテーションは除外するようにします。</p>
163      *
164      * @param anno 判定対象のアノテーション
165      * @return tureのときサポートします。
166      */
167     private boolean isSupportedAnnotation(final Annotation anno) {
168 
169         final String name = anno.annotationType().getName();
170         if(name.startsWith("java.lang.annotation.")) {
171             return false;
172         }
173 
174         return true;
175     }

177     /**
178      * setterメソッドの情報を処理する。
179      * @param propertyMeta プロパティのメタ情報
180      * @param method setterメソッド
181      */
182     private void doSetterMethod(final PropertyMeta propertyMeta, final Method method) {
183 
184         final String methodName = method.getName();
185         final String propertyName = NameUtils.uncapitalize(methodName.substring(3));
186 
187         if(!propertyMeta.getName().equals(propertyName)) {
188             // プロパティ名が一致しない場合はスキップする
189             return;
190         }
191 
192         propertyMeta.setWriteMethod(method);
193 
194     }
195 
196     /**
197      * getterメソッドの情報を処理する。
198      * @param propertyMeta プロパティのメタ情報
199      * @param method getterメソッド
200      */
201     private void doGetterMethod(final PropertyMeta propertyMeta, final Method method) {
202 
203         final String methodName = method.getName();
204         final String propertyName;
205         if(methodName.startsWith("get")) {
206             propertyName = NameUtils.uncapitalize(methodName.substring(3));
207         } else {
208             // 「is」から始まる場合
209             propertyName = NameUtils.uncapitalize(methodName.substring(2));
210         }
211 
212         if(!propertyMeta.getName().equals(propertyName)) {
213             // プロパティ名が一致しない場合はスキップする
214             return;
215         }
216 
217         propertyMeta.setReadMethod(method);
218 
219     }

221     /**
222      * カラム情報を処理する。
223      * @param propertyMeta プロパティのメタ情報
224      */
225     private void doColumnMeta(final PropertyMeta propertyMeta) {
226 
227         final ColumnMeta columnMeta = new ColumnMeta();
228 
229         final String defaultColumnName = namingRule.propertyToColumn(propertyMeta.getName());
230 
231         Optional<Column> annoColumn = propertyMeta.getAnnotation(Column.class);
232         if(annoColumn.isPresent()) {
233             if(!annoColumn.get().name().isEmpty()) {
234                 columnMeta.setName(annoColumn.get().name());
235             } else {
236                 columnMeta.setName(defaultColumnName);
237             }
238 
239             columnMeta.setInsertable(annoColumn.get().insertable());
240             columnMeta.setUpdatable(annoColumn.get().insertable());
241 
242         } else {
243             columnMeta.setName(defaultColumnName);
244         }
245 
246         propertyMeta.setColumnMeta(columnMeta);
247 
248     }

250     /**
251      * 主キーの生成情報を処理します。
252      * @param propertyMeta プロパティのメタ情報
253      * @param entityMeta エンティティのメタ情報
254      */
255     private void doIdGenerator(final PropertyMeta propertyMeta, final EntityMeta entityMeta) {
256 
257         if(!propertyMeta.isId()) {
258             return;
259         }
260 
261         Optional<GeneratedValue> annoGeneratedValue = propertyMeta.getAnnotation(GeneratedValue.class);
262         if(annoGeneratedValue.isEmpty()) {
263             return;
264         }
265 
266         final Class<?> propertyType = propertyMeta.getPropertyType();
267 
268         GenerationType generationType = annoGeneratedValue.get().strategy();
269         if(generationType == GenerationType.AUTO) {
270             generationType = dialect.getDefaultGenerationType();
271         }
272 
273         final IdGenerator idGenerator;
274         if(generationType == GenerationType.IDENTITY) {
275             IdentityIdGenerator identityIdGenerator = new IdentityIdGenerator(propertyType);
276             if(!annoGeneratedValue.get().format().isEmpty()) {
277                 identityIdGenerator.setFormatter(new DecimalFormat(annoGeneratedValue.get().format()));
278             }
279             idGenerator = identityIdGenerator;
280 
281         } else if(generationType == GenerationType.SEQUENCE) {
282             Optional<SequenceGenerator> annoSequenceGenerator = propertyMeta.getAnnotation(SequenceGenerator.class);
283             final String sequenceName;
284             if(annoSequenceGenerator.isPresent()) {
285                 sequenceName = NameUtils.tableFullName(annoSequenceGenerator.get().sequenceName(),
286                         annoSequenceGenerator.get().catalog(),
287                         annoSequenceGenerator.get().schema());
288             } else {
289                 sequenceName = entityMeta.getTableMeta().getName() + "_" + propertyMeta.getColumnMeta().getName();
290             }
291             SequenceIdGenerator sequenceIdGenerator = new SequenceIdGenerator(
292                     dialect.getSequenceIncrementer(dataSource, sequenceName), propertyType);
293 
294             if(!annoGeneratedValue.get().format().isEmpty()) {
295                 sequenceIdGenerator.setFormatter(new DecimalFormat(annoGeneratedValue.get().format()));
296             }
297 
298             idGenerator = sequenceIdGenerator;
299 
300         } else if(generationType == GenerationType.TABLE) {
301             Optional<TableGenerator> annoTableGenerator = propertyMeta.getAnnotation(TableGenerator.class);
302 
303             final TableIdContext tableIdContext = new TableIdContext();
304             tableIdContext.setTable(tableIdGeneratorProperties.getTable());
305             tableIdContext.setSchema(tableIdGeneratorProperties.getSchema());
306             tableIdContext.setCatalog(tableIdGeneratorProperties.getCatalog());
307             tableIdContext.setPkColumn(tableIdGeneratorProperties.getPkColumn());
308             tableIdContext.setValueColumn(tableIdGeneratorProperties.getValueColumn());
309             tableIdContext.setAllocationSize(tableIdGeneratorProperties.getAllocationSize());
310             tableIdContext.setInitialValue(tableIdGeneratorProperties.getInitialValue());
311 
312             String sequenceName = entityMeta.getTableMeta().getName() + "_" + propertyMeta.getColumnMeta().getName();
313 
314             annoTableGenerator.ifPresent(a -> {
315                 if(!a.table().isEmpty()) {
316                     tableIdContext.setTable(a.table());
317                 }
318 
319                 if(!a.schema().isEmpty()) {
320                     tableIdContext.setSchema(a.schema());
321                 }
322 
323                 if(!a.catalog().isEmpty()) {
324                     tableIdContext.setCatalog(a.catalog());
325                 }
326 
327                 if(!a.pkColumn().isEmpty()) {
328                     tableIdContext.setPkColumn(a.pkColumn());
329                 }
330 
331                 if(!a.valueColumn().isEmpty()) {
332                     tableIdContext.setValueColumn(a.valueColumn());
333                 }
334 
335                 if(a.initialValue() >= 0L) {
336                     tableIdContext.setInitialValue(a.initialValue());
337                 } else {
338                     throw new InvalidEntityException(entityMeta.getEntityType(), messageFormatter.create("property.anno.attr.min")
339                             .paramWithClass("classType", entityMeta.getEntityType())
340                             .param("property", propertyMeta.getName())
341                             .paramWithAnno("anno", TableGenerator.class)
342                             .param("attrName", "initialValue")
343                             .param("attrValue", a.initialValue())
344                             .param("min", 0)
345                             .format());
346                 }
347 
348                 if(a.allocationSize() >= 1L) {
349                     tableIdContext.setAllocationSize(a.allocationSize());
350                 } else {
351                     throw new InvalidEntityException(entityMeta.getEntityType(), messageFormatter.create("property.anno.attr.min")
352                             .paramWithClass("classType", entityMeta.getEntityType())
353                             .param("property", propertyMeta.getName())
354                             .paramWithAnno("anno", TableGenerator.class)
355                             .param("attrName", "allocationSize")
356                             .param("attrValue", a.allocationSize())
357                             .param("min", 1)
358                             .format());
359                 }
360 
361             });
362 
363 
364             if(annoTableGenerator.isPresent() && annoTableGenerator.get().pkColumn().isEmpty()) {
365                 sequenceName = annoTableGenerator.get().pkColumn();
366             }
367 
368             TableIdGenerator tableIdGenerator = new TableIdGenerator(
369                     new TableIdIncrementer(jdbcTemplate, tableIdContext),
370                     propertyMeta.getPropertyType(), sequenceName);
371 
372 
373             if(!annoGeneratedValue.get().format().isEmpty()) {
374                 try {
375                     tableIdGenerator.setFormatter(new DecimalFormat(annoGeneratedValue.get().format()));
376                 } catch(IllegalArgumentException e) {
377                     throw new InvalidEntityException(entityMeta.getEntityType(), messageFormatter.create("property.anno.attr.wrongFormat")
378                             .paramWithClass("classType", entityMeta.getEntityType())
379                             .param("property", propertyMeta.getName())
380                             .paramWithAnno("anno", TableGenerator.class)
381                             .param("attrName", "format")
382                             .param("attrValue", annoGeneratedValue.get().format())
383                             .format(), e);
384                 }
385             }
386 
387             idGenerator = tableIdGenerator;
388 
389 
390         } else if(generationType == GenerationType.UUID) {
391             idGenerator = new UUIDGenerator(propertyType);
392 
393         } else {
394             throw new InvalidEntityException(entityMeta.getEntityType(), messageFormatter.create("property.anno.attr.notSupportValue")
395                     .paramWithClass("classType", entityMeta.getClass())
396                     .param("property", propertyMeta.getName())
397                     .paramWithAnno("anno", GeneratedValue.class)
398                     .param("attrName", "strategy")
399                     .paramWithEnum("attrValue", generationType)
400                     .format());
401         }
402 
403         if(!idGenerator.isSupportedType(propertyType)) {
404             throw new InvalidEntityException(entityMeta.getClass(), messageFormatter.create("property.anno.notSupportTypeList")
405                     .paramWithClass("classType", entityMeta.getEntityType())
406                     .param("property", propertyMeta.getName())
407                     .paramWithAnno("anno", GeneratedValue.class)
408                     .paramWithClass("actualType", propertyType)
409                     .paramWithClass("expectedTypeList", idGenerator.getSupportedTypes())
410                     .format());
411         }
412 
413         propertyMeta.setIdGenerator(idGenerator);
414         propertyMeta.setIdGeneratonType(generationType);
415         //TODO: SpringからBeanを取得するする
416 
417     }

419     /**
420      * カラムとなるプロパティの整合性のチェック。
421      * アノテーションとクラスタイプのチェック
422      *
423      * @param declaringClass プロパティが定義されているクラス
424      * @param propertyMeta チェック対象のプロパティ
425      */
426     private void validateColumnProperty(final Class<?> declaringClass, final PropertyMeta propertyMeta) {
427 
428         final Class<?> propertyType = propertyMeta.getPropertyType();
429 
430         // 主キーのタイプチェック
431         if(propertyMeta.isId()
432                 && propertyType != String.class
433                 && propertyType != UUID.class
434                 && propertyType != Integer.class && propertyType != int.class
435                 && propertyType != Long.class && propertyType != long.class) {
436 
437             throw new InvalidEntityException(declaringClass, messageFormatter.create("property.anno.notSupportTypeList")
438                     .paramWithClass("entityType", declaringClass)
439                     .param("propperty", propertyMeta.getName())
440                     .paramWithAnno("anno", Id.class)
441                     .paramWithClass("actualType", propertyType)
442                     .paramWithClass("expectedTypeList", String.class, Integer.class, int.class, Long.class, long.class)
443                     .format());
444         }
445 
446         // 主キーでないのに値の生成用のアノテーションが付与されている場合
447         if(!propertyMeta.isId() && propertyMeta.hasAnnotation(GeneratedValue.class)) {
448 
449             throw new InvalidEntityException(declaringClass, messageFormatter.create("property.anno.notIdWithGeneratedValue")
450                     .paramWithClass("entityType", declaringClass)
451                     .param("propperty", propertyMeta.getName())
452                     .paramWithAnno("anno", GeneratedValue.class)
453                     .format());
454         }
455 
456 
457         // 列挙型のタイプチェック
458         if(propertyMeta.hasAnnotation(Enumerated.class) && !propertyType.isEnum()) {
459 
460             throw new InvalidEntityException(declaringClass, messageFormatter.create("property.anno.notSupportType")
461                     .paramWithClass("entityType", declaringClass)
462                     .param("propperty", propertyMeta.getName())
463                     .paramWithAnno("anno", Enumerated.class)
464                     .paramWithClass("actualType", propertyType)
465                     .paramWithClass("expectedType", Enum.class)
466                     .format());
467         }
468 
469         // バージョンキーのタイプチェック
470         if(propertyMeta.hasAnnotation(Version.class)
471                 && propertyType != Integer.class && propertyType != int.class
472                 && propertyType != Long.class && propertyType != long.class) {
473 
474             throw new InvalidEntityException(declaringClass, messageFormatter.create("property.anno.notSupportTypeList")
475                     .paramWithClass("entityType", declaringClass)
476                     .param("propperty", propertyMeta.getName())
477                     .paramWithAnno("anno", Version.class)
478                     .paramWithClass("actualType", propertyType)
479                     .paramWithClass("expectedTypeList", Integer.class, int.class, Long.class, long.class)
480                     .format());
481 
482         }
483 
484         // 時制のタイプチェック
485         if(!propertyMeta.hasAnnotation(Temporal.class)
486                 && propertyType == java.util.Date.class) {
487 
488             // 時制の型が不明なプロパティに対して、@Temporalが付与されていない場合
489             throw new InvalidEntityException(declaringClass, messageFormatter.create("property.anno.requiredAnnoTemporal")
490                     .paramWithClass("entityType", declaringClass)
491                     .param("propperty", propertyMeta.getName())
492                     .paramWithAnno("anno", Temporal.class)
493                     .format());
494         }
495 
496     }
