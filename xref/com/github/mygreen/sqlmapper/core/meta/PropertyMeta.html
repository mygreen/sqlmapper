28  /**
29   * プロパティのメタ情報です。
30   *
31   * @author T.TSUCHIE
32   * @version 0.3
33   *
34   */
35  public class PropertyMeta extends PropertyBase {
36  
37      /**
38       * 埋め込み型の主キーの子プロパティかどうか。
39       */
40      @Setter
41      private boolean embeddedableId = false;
42  
43      /**
44       * 埋め込みプロパティの情報
45       * <p>key=プロパティ名</p>
46       */
47      private LinkedCaseInsensitiveMap<PropertyMeta> embeddedablePropertyMetaMap = new LinkedCaseInsensitiveMap<>();
48  
49      /**
50       * 埋め込みのときの親のプロパティ情報
51       */
52      private Optional<PropertyMeta> parent = Optional.empty();
53  
54      /**
55       * カラムのメタ情報
56       */
57      @Getter
58      @Setter
59      private ColumnMeta columnMeta;
60  
61      /**
62       * 値の変換処理
63       */
64      @Getter
65      @Setter
66      private ValueType<?> valueType;
67  
68      /**
69       * IDの生成タイプ
70       */
71      @Getter
72      private Optional<GeneratedValue.GenerationType> idGenerationType = Optional.empty();
73  
74      /**
75       * IDの生成処理
76       */
77      @Getter
78      private Optional<IdGenerator> idGenerator = Optional.empty();
79  
80      /**
81       * 生成対象の識別子の情報。
82       * <p>ID生成時に渡す際の情報として使用するので、効率化のためにここで事前に作成して保持しておく。
83       */
84      @Getter
85      private Optional<IdGenerationContext> idGenerationContext = Optional.empty();
86  
87      /**
88       * プロパティのインスタンス情報を作成します。
89       * @param name プロパティ名
90       * @param propertyType プロパティのクラスタイプ
91       */
92      public PropertyMeta(String name, Class<?> propertyType) {
93          super(name, propertyType);
94      }
95  
96      /**
97       * カラム用のプロパティかどうか判定する。
98       * @return カラム情報を持つときtrueを返す。
99       */
100     public boolean isColumn() {
101         return columnMeta != null;
102     }
103 
104     /**
105      * 埋め込みプロパティ情報を追加する
106      * @param embeddedablePropertyMeta 埋め込みプロパティ
107      */
108     public void addEmbeddedablePropertyMeta(@NonNull PropertyMeta embeddedablePropertyMeta) {
109 
110         embeddedablePropertyMeta.parent = Optional.of(this);
111 
112         // 親が埋め込み主キーの時
113         if(hasAnnotation(EmbeddedId.class) && !isTransient()) {
114             embeddedablePropertyMeta.embeddedableId = true;
115         }
116 
117         this.embeddedablePropertyMetaMap.put(embeddedablePropertyMeta.getName(), embeddedablePropertyMeta);
118 
119     }
120 
121     /**
122      * 埋め込み用のクラスのプロパティかどか判定する。
123      * @return 埋め込み用のクラスのプロパティの場合trueを変す。
124      */
125     public boolean hasParent() {
126         return parent.isPresent();
127     }
128 
129     /**
130      * 埋め込み用のクラスのプロパティの親情報を取得する。
131      * @return 親情のプロパティ情報
132      * @throws NoSuchElementException 親が存在しないときにスローされます。
133      */
134     public PropertyMeta getParent() {
135         return parent.get();
136     }
137 
138     /**
139      * 埋め込みプロパティの一覧を取得する。
140      * @return
141      */
142     public Collection<PropertyMeta> getEmbeddedablePopertyMetaList() {
143         return embeddedablePropertyMetaMap.values();
144     }
145 
146     /**
147      * 主キーかどうか判定する。
148      * <p>アノテーション {@link Id}を付与されているかどうかで判定する。</p>
149      * <p>また、親が{@link EmbeddedId} を付与された埋め込みIDの場合は、子も主キーとなるためtrueを返す。</p>
150      *
151      * @return 主キーの場合は {@literal true}  を返す。
152      */
153     public boolean isId() {
154 
155         return hasAnnotation(Id.class) || embeddedableId;
156     }
157 
158     /**
159      * 埋め込み用のプロパティかどうか判定する。
160      * <p>埋め込みプロパティの子の場合は、falseを返す。
161      * @return 埋め込みの場合trueを返す。
162      */
163     public boolean isEmbedded() {
164         return hasAnnotation(EmbeddedId.class);
165     }
166

172         this.idGenerationType = Optional.ofNullable(generationType);

180         this.idGenerator = Optional.ofNullable(idGenerator);

188         this.idGenerationContext = Optional.ofNullable(idGenerationContext);
189     }

191     /**
192      * 永続化対象外かどうか判定する。
193      * <p>永続化対象外とは、アノテーション {@link Transient}が付与されているか、
194      * または、フィールドに修飾子 {@literal transient} が付与されているかどうかで判定します。
195      * @return 永続化対象外のとき {@literal true} を返す。
196      */
197     public boolean isTransient() {
198         return hasAnnotation(Transient.class)
199                 || field.map(f -> Modifier.isTransient(f.getModifiers())).orElse(false);
200     }
201
