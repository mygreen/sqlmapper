3   import java.util.List;
4   import java.util.UUID;
5   
6   import org.springframework.dao.DataIntegrityViolationException;
7   
8   import com.github.mygreen.sqlmapper.core.annotation.GeneratedValue.GenerationType;
9   
10  import lombok.RequiredArgsConstructor;
11  
12  /**
13   * {@link GenerationType#UUID}方式でIDの値を自動生成するIDジェネレータです。
14   * <p>サポートするIDのクラスタイプは、{@link UUID} / {@link String} です。
15   *
16   * @author T.TSUCHIE
17   *
18   */
19  @RequiredArgsConstructor
20  public class UUIDGenerator implements IdGenerator {
21  
22      /**
23       * サポートしているクラスタイプ
24       */
25      private static final List<Class<?>> SUPPORTED_TYPE_LIST = List.of(UUID.class, String.class);
26  
27      /**
28       * 生成するIDのクラスタイプ
29       */
30      private final Class<?> requiredType;
31  
32      @Override
33      public boolean isSupportedType(Class<?> type) {
34          return SUPPORTED_TYPE_LIST.contains(type);
35      }
36  
37      @Override
38      public Class<?>[] getSupportedTypes() {
39          return SUPPORTED_TYPE_LIST.toArray(new Class[SUPPORTED_TYPE_LIST.size()]);
40      }
41  
42      @Override
43      public Object generateValue() {
44  
45          UUID value = UUID.randomUUID();
46  
47          if(requiredType == UUID.class) {
48              return value;
49  
50          } else if(requiredType == String.class) {
51              return value.toString();
52          }
53  
54          throw new DataIntegrityViolationException("not supported java type : " + requiredType.getName());
55      }
56  }
