1   package com.github.mygreen.sqlmapper.core.id;
2   
3   import java.util.List;
4   
5   import org.springframework.dao.DataIntegrityViolationException;
6   
7   import com.github.mygreen.sqlmapper.core.util.NumberConvertUtils;
8   
9   import lombok.RequiredArgsConstructor;
10  
11  /**
12   * IDENTITYによる生成は、実際にはJdbcTemplateで行います。
13   * <p>このクラスでは、{@link #generateValue(Number)} による変換だけ行います。
14   *
15   * @version 0.3
16   * @author T.TSUCHIE
17   *
18   */
19  @RequiredArgsConstructor
20  public class IdentityIdGenerator implements IdGenerator {
21  
22      /**
23       * サポートしているクラスタイプ
24       */
25      private static final List<Class<?>> SUPPORTED_TYPE_LIST = List.of(
26              long.class, Long.class, int.class, Integer.class);
27  
28      /**
29       * 生成する識別子のタイプ
30       */
31      private final Class<?> requiredType;
32  
33      @Override
34      public boolean isSupportedType(Class<?> type) {
35          return SUPPORTED_TYPE_LIST.contains(type);
36      }
37  
38      @Override
39      public Class<?>[] getSupportedTypes() {
40          return SUPPORTED_TYPE_LIST.toArray(new Class[SUPPORTED_TYPE_LIST.size()]);
41      }
42  
43      /**
44       * {@inheritDoc}
45       * @throws UnsupportedOperationException このメソッドを呼び出したときに必ずスローされます。
46       */
47      @Override
48      public Object generateValue(final IdGenerationContext context) {
49          throw new UnsupportedOperationException("this method is not supported.");
50      }
51  
52      /**
53       * 生成したIDをプロパティのクラス型に変換する。
54       * @param value 生成したID
55       * @return プロパティのクラス型に変換したID。
56       * @throws DataIntegrityViolationException サポート対象外のクラスタイプの場合にスローされます。
57       */
58      public Object generateValue(final Number value) {
59  
60          if(requiredType == long.class || requiredType == Long.class
61                  || requiredType == int.class || requiredType == Integer.class) {
62              return NumberConvertUtils.convertNumber(requiredType, value);
63  
64          }
65  
66          throw new DataIntegrityViolationException("not supported java type : " + requiredType.getName());
67  
68      }
69  }
