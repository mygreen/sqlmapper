1   package com.github.mygreen.sqlmapper.core.type.standard;
2   
3   import java.sql.Date;
4   import java.sql.ResultSet;
5   import java.sql.SQLException;
6   import java.sql.Types;
7   import java.text.SimpleDateFormat;
8   
9   import com.github.mygreen.sqlmapper.core.annotation.Temporal.TemporalType;
10  import com.github.mygreen.sqlmapper.core.dialect.Dialect;
11  
12  import lombok.Getter;
13  
14  /**
15   * {@link Date} 型のマッピングを処理します。
16   *
17   *
18   * @author T.TSUCHIE
19   *
20   */
21  public class SqlDateType implements SqlTemporalType<Date> {
22  
23      /**
24       * SQLに直接埋め込む時にフォーマットする書式
25       */
26      @Getter
27      private final String pattern;
28  
29      /**
30       * インスタンスを作成します。
31       * <p>SQLに直接埋め込む時にフォーマットする書式は、{@literal yyyy-MM-dd} が指定されます。
32       */
33      public SqlDateType() {
34          this("yyyy-MM-dd");
35      }
36  
37      /**
38       * 書式を指定してインスタンスを作成します。
39       * @param pattern SQLに直接埋め込む時にフォーマットする書式。
40       */
41      public SqlDateType(final String pattern) {
42          this.pattern = pattern;
43      }
44  
45      /**
46       * {@inheritDoc}
47       *
48       * @return {@link TemporalType#DATE} を返します。
49       */
50      @Override
51      public TemporalType getTemporalType() {
52          return TemporalType.DATE;
53      }
54  
55      @Override
56      public Date convertTo(java.util.Date utilDate) {
57          return new Date(utilDate.getTime());
58      }
59  
60      @Override
61      public Date getValue(ResultSet rs, int columnIndex) throws SQLException {
62  
63          Date value = rs.getDate(columnIndex);
64          if(rs.wasNull()) {
65              return null;
66          }
67  
68          return value;
69      }
70  
71      @Override
72      public Object getSqlParameterValue(Date value) {
73          return value;
74      }
75  
76      /**
77       * {@inheritDoc}
78       *
79       * 引数が{@literal null}の場合は{@literal null}を返します。
80       * @return コンストラクタで指定した書式でフォーマットした値を返します。
81       */
82      @Override
83      public String getEmbeddedValue(Date value) {
84          return value != null ? new SimpleDateFormat(pattern).format(value) : null;
85      }
86  
87      @Override
88      public int getSqlType(Dialect dialect) {
89          return Types.DATE;
90      }
91  
92  }
