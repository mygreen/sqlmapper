3   import java.math.BigDecimal;
4   import java.sql.ResultSet;
5   import java.sql.SQLException;
6   
7   import com.github.mygreen.sqlmapper.core.type.ValueType;
8   
9   import lombok.RequiredArgsConstructor;
10  
11  /**
12   * {@link BigDecimal} 型のマッピングを処理します。
13   *
14   *
15   * @author T.TSUCHIE
16   *
17   */
18  @RequiredArgsConstructor
19  public class BigDecimalType implements ValueType<BigDecimal> {
20  
21      @Override
22      public BigDecimal getValue(ResultSet rs, int columnIndex) throws SQLException {
23  
24          BigDecimal value = rs.getBigDecimal(columnIndex);
25          if(value == null) {
26              return null;
27          }
28  
29          return value;
30      }
31  
32      @Override
33      public Object getSqlParameterValue(BigDecimal value) {
34          return value;
35      }
36  
37      @Override
38      public String getEmbeddedValue(BigDecimal value) {
39          return value != null ? value.toPlainString() : null;
40      }
41  
42  }
