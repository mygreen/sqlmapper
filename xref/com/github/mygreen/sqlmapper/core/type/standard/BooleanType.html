1   package com.github.mygreen.sqlmapper.core.type.standard;

3   import java.sql.ResultSet;
4   import java.sql.SQLException;
5   import java.sql.Types;
6   
7   import com.github.mygreen.sqlmapper.core.dialect.Dialect;
8   import com.github.mygreen.sqlmapper.core.type.ValueType;
9   
10  import lombok.Getter;
11  import lombok.RequiredArgsConstructor;
12  
13  /**
14   * boolean 型及びそのラッパー型のマッピングを処理します。
15   *
16   *
17   * @author T.TSUCHIE
18   *
19   */
20  @RequiredArgsConstructor
21  public class BooleanType implements ValueType<Boolean> {
22  
23      /**
24       * プリミティブ型かどうか
25       */
26      @Getter
27      private final boolean forPrimitive;
28  
29      @Override
30      public Boolean getValue(ResultSet rs, int columnIndex) throws SQLException {
31  
32          Boolean value = rs.getBoolean(columnIndex);
33          if(rs.wasNull() && !forPrimitive) {
34              return null;
35          }
36  
37          return value;
38      }
39  
40      /**
41       * {@inheritDoc}
42       * @return 引数で指定した値が{@literal null} かつ、プリミティブ型にマッピングする際は {@literal false} を返します。
43       */
44      @Override
45      public Object getSqlParameterValue(Boolean value) {
46  
47          if(value == null && forPrimitive) {
48              value = false;
49          }
50  
51          return value;
52      }
53  
54      @Override
55      public int getSqlType(Dialect dialect) {
56          return Types.BOOLEAN;
57      }
58  }
