1   package com.github.mygreen.sqlmapper.core.type.standard;

3   import java.sql.ResultSet;
4   import java.sql.SQLException;
5   import java.sql.Types;
6   import java.util.UUID;
7   
8   import com.github.mygreen.sqlmapper.core.dialect.Dialect;
9   import com.github.mygreen.sqlmapper.core.type.ValueType;
10  
11  /**
12   * 文字列型を{@link UUID} 型のマッピングを処理します。
13   * <p>DB側がUUID型に対応していない場合、永続化する際に文字列に変換して対応します。
14   *
15   *
16   * @author T.TSUCHIE
17   *
18   */
19  public class UUIDType implements ValueType<UUID> {
20  
21      @Override
22      public UUID getValue(ResultSet rs, int columnIndex) throws SQLException {
23          Object value = rs.getObject(columnIndex);
24          return value != null ? UUID.fromString(value.toString()) : null;
25      }
26  
27      @Override
28      public Object getSqlParameterValue(UUID value) {
29          /*
30           * UUIDのオブジェクト型を使用する場合
31           * ・PostgreSQLの場合は、OTHERで型指定が必要。
32           * ・H2DBの場合は、逆にOTHER指定でエラーが出るため、何も指定しない。
33           */
34          return value != null ? value.toString() : null;
35      }
36  
37      @Override
38      public int getSqlType(Dialect dialect) {
39          return Types.VARCHAR;
40      }
41  }
