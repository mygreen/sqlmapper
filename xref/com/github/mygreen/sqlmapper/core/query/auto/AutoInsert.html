1   package com.github.mygreen.sqlmapper.core.query.auto;
2   
3   import org.springframework.dao.DuplicateKeyException;
4   
5   import com.github.mygreen.sqlmapper.core.query.IllegalOperateException;
6   import com.github.mygreen.sqlmapper.metamodel.PropertyPath;
7   
8   /**
9    * 挿入を行うSQLを自動生成するクエリです。
10   *
11   *
12   * @author T.TSUCHIE
13   *
14   * @param <T> 処理対象となるエンティティの型
15   */
16  public interface AutoInsert<T> {
17  
18      /**
19       * 指定のプロパティのみを挿入対象とします。
20       * <p>アノテーション {@literal @Column(insertable = false)} が設定されているプロパティは対象外となります。</p>
21       *
22       * @param properties 挿入対象のプロパティ情報。
23       * @return 自身のインスタンス。
24       * @throws IllegalOperateException エンティティに存在しないプロパティ名を指定した場合にスローされます。
25       */
26      AutoInsert<T> includes(PropertyPath<?>... properties);
27  
28      /**
29       * 指定のプロパティを挿入対象から除外します。
30       *
31       * @param properties 除外対象のプロパティ情報。
32       * @return 自身のインスタンス。
33       */
34      AutoInsert<T> excludes(PropertyPath<?>... properties);
35  
36      /**
37       * クエリを実行します。
38       * @return 挿入した行数。
39       * @throws DuplicateKeyException 主キーなどのが一意制約に違反したとき。
40       */
41      int execute();
42  
43  }
