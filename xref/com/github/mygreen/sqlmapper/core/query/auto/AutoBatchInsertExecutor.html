25  /**
26   * バッチ挿入を行うSQLを自動生成するクエリを実行します。
27   * {@link AutoBatchInsertImpl}のクエリ実行処理の移譲先です。
28   *
29   *
30   * @author T.TSUCHIE
31   *
32   */
33  public class AutoBatchInsertExecutor {
34  
35      /**
36       * バージョンプロパティの初期値
37       */
38      public static final long INITIAL_VERSION = AutoInsertExecutor.INITIAL_VERSION;
39  
40      /**
41       * クエリ情報
42       */
43      private final AutoBatchInsertImpl<?> query;
44  
45      /**
46       * 設定情報
47       */
48      private final SqlMapperContext context;
49  
50      /**
51       * クエリのパラメータ - エンティティごとの設定
52       */
53      private MapSqlParameterSource[] batchParams;
54  
55      /**
56       * 挿入するカラム名
57       */
58      private List<String> usingColumnNames = new ArrayList<>();
59  
60      /**
61       * IDENTITYによる主キーの自動生成を使用するカラム名
62       */
63      private List<String> usingIdentityKeyColumnNames = new ArrayList<>();
64  
65      /**
66       * レコードの挿入操作を行う処理
67       */
68      private SimpleJdbcInsert insertOperation;
69  
70      /**
71       * シーケンスやテーブルによる主キーの生成したキー
72       * <p>key=カラム名、value=全レコード分の生成したキーの値</p>
73       */
74      private Map<String, Object[]> generatedKeysMap = new HashMap<String, Object[]>();
75  
76      /**
77       * 組み立てたクエリ情報を指定するコンストラクタ。
78       * @param query クエリ情報
79       */
80      public AutoBatchInsertExecutor(AutoBatchInsertImpl<?> query) {
81          this.query = query;
82          this.context = query.getContext();
83      }
84  
85      /**
86       * クエリ実行の準備を行います。
87       */
88      private void prepare() {
89  
90          this.batchParams = new MapSqlParameterSource[query.getEntitySize()];
91  
92          prepareSqlParam();
93          prepareInsertOperation();
94      }
95  
96      /**
97       * クエリ実行時のパラメータを準備します。
98       */
99      @SuppressWarnings({"rawtypes", "unchecked"})
100     private void prepareSqlParam() {
101 
102         final int dataSize = query.getEntitySize();
103 
104         for(PropertyMeta propertyMeta : query.getEntityMeta().getAllColumnPropertyMeta()) {
105 
106             final String propertyName = propertyMeta.getName();
107             if(!propertyMeta.getColumnMeta().isInsertable()) {
108                 continue;
109             }
110 
111             if(query.getExcludesProperties().contains(propertyName)) {
112                 continue;
113             }
114 
115             if(!query.getIncludesProperties().isEmpty() && !query.getIncludesProperties().contains(propertyName)) {
116                 continue;
117             }
118 
119             final String columnName = propertyMeta.getColumnMeta().getName();
120 
121             // 各レコードのパラメータを作成する。
122             for(int i=0; i < dataSize; i++) {
123                 final MapSqlParameterSource paramSource = QueryUtils.get(batchParams, i);
124                 Object propertyValue = PropertyValueInvoker.getEmbeddedPropertyValue(propertyMeta, query.getEntity(i));
125 
126                 Optional<GenerationType> generationType = propertyMeta.getIdGenerationType();
127                 if(propertyMeta.isId() && generationType.isPresent()) {
128                     if(generationType.get() == GenerationType.IDENTITY) {
129                         if(i == 0) {
130                             //IDENTITYの場合は、クエリ実行後に取得するため、対象のカラム情報を１回だけ
131                             usingIdentityKeyColumnNames.add(columnName);
132                         }
133                         continue;
134                     } else {
135                         propertyValue = getNextVal(propertyMeta.getIdGenerator().get(), propertyMeta.getColumnMeta().getName(), i);
136                         PropertyValueInvoker.setEmbeddedPropertyValue(propertyMeta, query.getEntity(i), propertyValue);
137                     }
138                 }
139 
140                 if(propertyValue == null && propertyMeta.isVersion()) {
141                     // バージョンキーが設定されていない場合、初期値設定する
142                     propertyValue = NumberConvertUtils.convertNumber(propertyMeta.getPropertyType(), INITIAL_VERSION);
143                     PropertyValueInvoker.setEmbeddedPropertyValue(propertyMeta, query.getEntity(i), propertyValue);
144 
145                 }
146 
147                 // クエリのパラメータの組み立て
148                 ValueType valueType = propertyMeta.getValueType();
149                 paramSource.addValue(columnName, valueType.getSqlParameterValue(propertyValue));
150 
151             }
152 
153             // IDENTITYの主キーでない場合は通常カラムとして追加
154             if(!usingIdentityKeyColumnNames.contains(columnName)) {
155                 usingColumnNames.add(columnName);
156             }
157 
158         }
159     }
160 
161     /**
162      * 主キーを生成する
163      * @param generator 主キーの生成処理
164      * @param columnName 生成対象のカラム名
165      * @param index 生成対象のレコードのインデックス
166      * @return 生成した主キーの値
167      */
168     private Object getNextVal(final IdGenerator generator, final String columnName, final int index) {
169 
170         // 1レコードの主キーをまとめてキーを生成しておき、キャッシュしておく。
171         Object[] generatedKeys = generatedKeysMap.computeIfAbsent(columnName, v ->
172                 context.getIdGeneratorTransactionTemplate().execute(action -> {
173                     return generator.generateValues(query.getEntities().length);
174                 }));
175 
176          return generatedKeys[index];
177     }
178 
179     /**
180      * SQLの実行をする処理を組み立てます
181      */
182     private void prepareInsertOperation() {
183 
184         this.insertOperation = new SimpleJdbcInsert(context.getJdbcTemplate())
185                 .withTableName(query.getEntityMeta().getTableMeta().getFullName())
186                 .usingColumns(QueryUtils.toArray(usingColumnNames));
187 
188         if(!usingIdentityKeyColumnNames.isEmpty()) {
189             insertOperation.usingGeneratedKeyColumns(QueryUtils.toArray(usingIdentityKeyColumnNames));
190         }
191     }
192 
193     /**
194      * 挿入処理を実行します。
195      * @return 挿入したレコード件数を返します。
196      * @throws DuplicateKeyException 主キーなどのが一意制約に違反したとき。
197      */
198     public int[] execute() {
199 
200         prepare();
201 
202         if(this.usingIdentityKeyColumnNames.isEmpty()) {
203             // 主キーがIDENTITYによる生成でない場合
204             return insertOperation.executeBatch(batchParams);
205 
206         } else {
207             // １件ずつ処理する
208             int dataSize = query.getEntities().length;
209             int[] res = new int[dataSize];
210             for(int i=0; i < dataSize; i++) {
211 
212                 final KeyHolder keyHolder = insertOperation.executeAndReturnKeyHolder(batchParams[i]);
213                 // 生成した主キーをエンティティに設定する
214                 for(Map.Entry<String, Object> entry : keyHolder.getKeys().entrySet()) {
215 
216                     if(!usingIdentityKeyColumnNames.contains(entry.getKey())) {
217                         continue;
218                     }
219 
220                     PropertyMeta propertyMeta = query.getEntityMeta().getColumnPropertyMeta(entry.getKey()).orElseThrow();
221                     IdentityIdGenerator idGenerator = (IdentityIdGenerator) propertyMeta.getIdGenerator().get();
222                     Object propertyValue = idGenerator.generateValue((Number)entry.getValue());
223                     PropertyValueInvoker.setEmbeddedPropertyValue(propertyMeta, query.getEntity(i), propertyValue);
224 
225                 }
226 
227                 res[i] = 1;
228 
229             }
230 
231             return res;
232         }
233     }
