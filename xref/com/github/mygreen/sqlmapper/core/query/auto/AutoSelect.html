1   package com.github.mygreen.sqlmapper.core.query.auto;
2   
3   import java.util.List;
4   import java.util.Optional;
5   import java.util.stream.Stream;
6   
7   import org.springframework.dao.IncorrectResultSizeDataAccessException;
8   
9   import com.github.mygreen.sqlmapper.core.query.IllegalOperateException;
10  import com.github.mygreen.sqlmapper.core.query.JoinAssociation;
11  import com.github.mygreen.sqlmapper.core.query.JoinCondition;
12  import com.github.mygreen.sqlmapper.metamodel.EntityPath;
13  import com.github.mygreen.sqlmapper.metamodel.OrderSpecifier;
14  import com.github.mygreen.sqlmapper.metamodel.Predicate;
15  import com.github.mygreen.sqlmapper.metamodel.PropertyPath;
16  
17  /**
18   * 抽出を行うSQLを自動生成するクエリです。
19   *
20   * @author T.TSUCHIE
21   *
22   * @param <T> 処理対象となるエンティティの型
23   */
24  public interface AutoSelect<T> {
25  
26      /**
27       * ヒントを設定します。
28       * @param hint ヒント
29       * @return このインスタンス自身
30       */
31      AutoSelect<T> hint(String hint);
32  
33      /**
34       * 抽出する行数を指定します。
35       * @param limit 行数
36       * @return このインスタンス自身
37       */
38      AutoSelect<T> limit(int limit);
39  
40      /**
41       * 抽出するデータの開始位置を指定します。
42       * @param offset 開始位置。
43       * @return このインスタンス自身
44       */
45      AutoSelect<T> offset(int offset);
46  
47      /**
48       * 指定のプロパティのみを挿入対象とします。
49       * <p>アノテーション {@literal @Column(insertable = false)} が設定されているプロパティは対象外となります。</p>
50       *
51       * @param properties 挿入対象のプロパティ情報。
52       * @return 自身のインスタンス。
53       * @throws IllegalOperateException エンティティに存在しないプロパティ名を指定した場合にスローされます。
54       */
55      AutoSelect<T> includes(PropertyPath<?>... properties);
56  
57      /**
58       * 指定のプロパティを挿入対象から除外します。
59       *
60       * @param properties 除外対象のプロパティ情報。
61       * @return 自身のインスタンス。
62       */
63      AutoSelect<T> excludes(PropertyPath<?>... properties);
64  
65      /**
66       * FROM句で指定したテーブルと内部結合（{@literal INNERT JOIN}）する条件を指定します。
67       *
68       * @param <ENTITY> 結合先のテーブルのエンティティタイプ
69       * @param toEntityPath 結合先テーブルのエンティティ情報
70       * @param conditioner 結合条件の組み立て
71       * @return 自身のインスタンス
72       */
73      <ENTITY extends EntityPath<?>> AutoSelect<T> innerJoin(ENTITY toEntityPath,
74              JoinCondition.Conditioner<ENTITY> conditioner);
75  
76      /**
77       * FROM句で指定したテーブルと左外部結合（{@literal LEFT OUTER JOIN}）する条件を指定します。
78       *
79       * @param <ENTITY> 結合先のテーブルのエンティティタイプ
80       * @param toEntityPath 結合先テーブルのエンティティ情報
81       * @param conditioner 結合条件の組み立て
82       * @return 自身のインスタンス
83       * @throws IllegalOperateException 既に同じ組み合わせのエンティティ（テーブル）を指定しているときにスローされます。
84       */
85      <ENTITY extends EntityPath<?>> AutoSelect<T> leftJoin(ENTITY toEntityPath,
86              JoinCondition.Conditioner<ENTITY> conditioner);
87  
88      /**
89       * テーブル結合の際に複数のテーブルのエンティティの構成定義を指定します。
90       *
91       * @param <E1> エンティティタイプ1
92       * @param <E2> エンティティタイプ2
93       * @param entityPath1 エンティティ情報1
94       * @param entityPath2 エンティティ情報2
95       * @param associator エンティティの構成定義
96       * @return 自身のインスタンス
97       * @throws IllegalOperateException 既に同じ組み合わせのエンティティの構成定義を指定しているときにスローされます。
98       */
99      <E1, E2> AutoSelect<T> associate(EntityPath<E1> entityPath1, EntityPath<E2> entityPath2,
100             JoinAssociation.Associator<E1, E2> associator);
101 
102 
103     /**
104      * 検索条件を指定します。
105      * @param where 検索条件。
106      * @return 自身のインスタンス。
107      */
108     AutoSelect<T> where(Predicate where);
109 
110     /**
111      * ソート順を指定します。
112      * @param orders ソートするロパティの並び順情報
113      * @return 自身のインスタンス。
114      */
115     AutoSelect<T> orderBy(OrderSpecifier... orders);
116 
117     /**
118      * WHERE句の条件にIdプロパティ(主キー)を指定します。
119      *
120      * @param idPropertyValues IDプロパティの値。エンティティに定義している順で指定する必要があります。
121      * @return 自身のインスタンス。
122      * @throws IllegalOperateException 指定したIDの個数とエンティティの個数と一致しないときにスローされます。
123      */
124     AutoSelect<T> id(Object... idPropertyValues);
125 
126     /**
127      * WHERE句の条件にバージョンプロパティを指定します。
128      *
129      * @param versionPropertyValue バージョンプロパティの値。
130      * @return 自身のインスタンス
131      * @throws IllegalOperateException エンティティにバージョンキーが定義されていないときにスローされます。
132      */
133     AutoSelect<T> version(Object versionPropertyValue);
134 
135     /**
136      * {@literal FOR UPDATE} を追加します。
137      * @return このインスタンス自身。
138      * @throws IllegalOperateException DBMSがこの操作をサポートしていない場合にスローされます。
139      */
140     AutoSelect<T> forUpdate();
141 
142     /**
143      * {@literal FOR UPDATE NOWAIT} を追加します。
144      * @return このインスタンス自身。
145      * @throws IllegalOperateException DBMSがこの操作をサポートしていない場合にスローされます。
146      */
147     AutoSelect<T> forUpdateNoWait();
148 
149     /**
150      * {@literal FOR UPDATE WAIT} を追加します。
151      * @param seconds  ロックを獲得できるまでの最大待機時間(秒単位)
152      * @return このインスタンス自身。
153      * @throws IllegalOperateException DBMSがこの操作をサポートしていない場合にスローされます。
154      */
155     AutoSelect<T> forUpdateWait(int seconds);
156 
157     /**
158      * SQLが返す結果セットの行数を返します。
159      * @return SQLが返す結果セットの行数
160      */
161     long getCount();
162 
163     /**
164      * 検索してベースオブジェクトを返します。
165      *
166      * @return ベースオブジェクト。
167      * @throws IncorrectResultSizeDataAccessException 1件も見つからない場合、2件以上見つかった場合にスローされます。
168      */
169     T getSingleResult();
170 
171     /**
172      * 検索してベースオブジェクトを返します。
173      *
174      * @return ベースオブジェクト。1件も対象がないときは空を返します。
175      */
176     Optional<T> getOptionalResult();
177 
178     /**
179      * 検索してベースオブジェクトを返します。
180      *
181      * @return 検索してヒットした複数のベースオブジェクト。1件も対象がないときは空のリストを返します。
182      */
183     List<T> getResultList();
184 
185     /**
186      * 問い合わせ結果を{@link Stream} で取得します。
187      * 問い合わせ結果全体のリストを作成しないため、問い合わせ結果が膨大になる場合でもメモリ消費量を抑えることが出来ます。
188      *
189      * @return 問い合わせの結果。
190      */
191     Stream<T> getResultStream();
192 
193 }
