1   package com.github.mygreen.sqlmapper.core.query.auto;
2   
3   import org.springframework.dao.OptimisticLockingFailureException;
4   
5   import com.github.mygreen.sqlmapper.core.query.IllegalOperateException;
6   import com.github.mygreen.sqlmapper.metamodel.PropertyPath;
7   
8   /**
9    * バッチ更新を行うSQLを自動生成するクエリです。
10   *
11   * @version 0.3
12   * @author T.TSUCHIE
13   *
14   * @param <T> 処理対象となるエンティティの型
15   */
16  public interface AutoBatchUpdate<T> {
17  
18      /**
19       * クエリタイムアウトの秒数を設定します。
20       * <p>{@literal -1} を指定するとJDBC ドライバーのデフォルト値を使用します。
21       *
22       * @since 0.3
23       * @param seconds クエリタイムアウトの秒数
24       * @return 自身のインスタンス。
25       */
26      AutoBatchUpdate<T> queryTimeout(int seconds);
27  
28      /**
29       * バージョンプロパティを通常の更新対象に含め、バージョンチェックの対象外とします。
30       * <p>
31       * このメソッドが呼び出されると、<code>update</code>文の<code>where</code>句にはバージョンのチェックが含まれなくなり、
32       * バージョンプロパティは通常のプロパティと同じように更新対象に含められます。
33       * </p>
34       *
35       * @return このインスタンス自身
36       */
37      AutoBatchUpdate<T> includesVersion();
38  
39      /**
40       * バージョンチェックを行った場合に、更新行数が0行でも {@link OptimisticLockingFailureException} をスローしないようにします。
41       * @return このインスタンス自身
42       */
43      AutoBatchUpdate<T> suppresOptimisticLockException();
44  
45      /**
46       * 指定のプロパティのみを更新対象とします。
47       * <p>ID(主キー)は自動的に更新対象外となります。</p>
48       * <p>アノテーション {@literal @Column(updatable = false)} が設定されているプロパティは自動的に対象外となります。</p>
49       *
50       * @param properties 更新対象のプロパティ情報。
51       * @return 自身のインスタンス。
52       * @throws IllegalOperateException エンティティに存在しないプロパティ名を指定した場合にスローされます。
53       */
54      AutoBatchUpdate<T> includes(PropertyPath<?>... properties);
55  
56      /**
57       * 指定のプロパティを更新対象から除外します。
58       * <p>ID(主キー)は自動的に更新対象外となります。</p>
59       * <p>アノテーション {@literal @Column(updatable = false)} が設定されているプロパティは自動的に対象外となります。</p>
60       *
61       * @param properties 除外対象のプロパティ名。
62       * @return 自身のインスタンス。
63       * @throws IllegalOperateException エンティティに存在しないプロパティ名を指定した場合にスローされます。
64       */
65      AutoBatchUpdate<T> excludes(PropertyPath<?>... properties);
66  
67      /**
68       * 更新クエリを実行します。
69       * @return 更新したレコード件数を返します。
70       */
71      int[] execute();
72  }
