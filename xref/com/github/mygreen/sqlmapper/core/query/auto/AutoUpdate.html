1   package com.github.mygreen.sqlmapper.core.query.auto;
2   
3   import java.util.Map;
4   
5   import org.springframework.dao.OptimisticLockingFailureException;
6   
7   import com.github.mygreen.sqlmapper.core.query.IllegalOperateException;
8   import com.github.mygreen.sqlmapper.metamodel.PropertyPath;
9   
10  /**
11   * 更新を行うSQLを自動生成するクエリです。
12   *
13   *
14   * @author T.TSUCHIE
15   *
16   * @param <T> 処理対象となるエンティティの型
17   */
18  public interface AutoUpdate<T> {
19  
20      /**
21       * バージョンプロパティを通常の更新対象に含め、バージョンチェックの対象外とします。
22       * <p>
23       * このメソッドが呼び出されると、<code>update</code>文の<code>where</code>句にはバージョンのチェックが含まれなくなり、
24       * バージョンプロパティは通常のプロパティと同じように更新対象に含められます ({@link #excludesNull()}や{@link #changedFrom(Object)}等も同じように適用されます)。
25       * </p>
26       *
27       * @return このインスタンス自身
28       */
29      AutoUpdate<T> includesVersion();
30  
31      /**
32       * <code>null</code>値のプロパティを更新対象から除外します。
33       * @return このインスタンス自身
34       */
35      AutoUpdate<T> excludesNull();
36  
37      /**
38       * バージョンチェックを行った場合に、更新行数が0行でも {@link OptimisticLockingFailureException} をスローしないようにします。
39       * @return このインスタンス自身
40       */
41      AutoUpdate<T> suppresOptimisticLockException();
42  
43      /**
44       * 指定のプロパティのみを挿入対象とします。
45       * <p>アノテーション {@literal @Column(updatable = false)} が設定されているプロパティは対象外となります。</p>
46       *
47       * @param properties 更新対象のプロパティ情報。
48       * @return 自身のインスタンス。
49       * @throws IllegalOperateException エンティティに存在しないプロパティ名を指定した場合にスローされます。
50       */
51      AutoUpdate<T> includes(PropertyPath<?>... properties);
52  
53      /**
54       * 指定のプロパティを更新対象から除外します。
55       *
56       * @param properties 除外対象のプロパティ名。
57       * @return 自身のインスタンス。
58       * @throws IllegalOperateException エンティティに存在しないプロパティ名を指定した場合にスローされます。
59       */
60      AutoUpdate<T> excludes(PropertyPath<?>... properties);
61  
62      /**
63       * beforeから変更のあったプロパティだけを更新対象とします
64       * @param beforeEntity 変更前の状態を持つエンティティ
65       * @return このインスタンス自身
66       */
67      AutoUpdate<T> changedFrom(T beforeEntity);
68  
69      /**
70       * beforeから変更のあったプロパティだけを更新対象とします。
71       * <p>引数 {@literal beforeStates} のサイズが {@literal 0} のときは何もしません。
72       * @param beforeStates 変更前の状態を持つマップ。（key=プロパティ名、value=プロパティ値）
73       * @return このインスタンス自身。
74       */
75      AutoUpdate<T> changedFrom(Map<String, Object> beforeStates);
76  
77      /**
78       * 更新クエリを実行します。
79       * @return 更新したレコード件数です。更新対象のプロパティ（カラム）がない場合は {@literal 0} を返します。
80       * @throws OptimisticLockingFailureException 楽観的排他制御を行う場合に該当するレコードが存在しない場合にスローされます。
81       */
82      int execute();
83  
84  }
