3   /**
4    * order by句を組み立てるクラスです。
5    *
6    *
7    * @author T.TSUCHIE
8    *
9    */
10  public class OrderByClause {
11  
12      private final StringBuilder sql;
13  
14      public OrderByClause() {
15          this.sql = new StringBuilder();
16      }
17  
18      /**
19       * SQLの長さを返します。
20       * @return SQLの長さ
21       */
22      public int getLength() {
23          return sql.length();
24      }
25  
26      /**
27       * SQLに変換します。
28       * @return 組み立てたSQL
29       */
30      public String toSql() {
31          return sql.toString();
32      }
33  
34      /**
35       * order by句を追加します。
36       * @param orderBys order by句
37       */
38      public void addSql(final String... orderBys) {
39          if(sql.length() == 0) {
40              sql.append(" order by ");
41          }
42          sql.append(String.join(", ", orderBys));
43  
44      }
45  
46  }
