1   package com.github.mygreen.sqlmapper.core.dialect;
2   
3   import javax.sql.DataSource;
4   
5   import org.springframework.jdbc.support.incrementer.DataFieldMaxValueIncrementer;
6   
7   import com.github.mygreen.splate.SqlTemplateEngine;
8   import com.github.mygreen.sqlmapper.core.annotation.GeneratedValue.GenerationType;
9   import com.github.mygreen.sqlmapper.core.query.SelectForUpdateType;
10  import com.github.mygreen.sqlmapper.core.type.ValueType;
11  
12  /**
13   * SQLの方言を定義します。
14   *
15   *
16   * @author T.TSUCHIE
17   *
18   */
19  public interface Dialect {
20  
21      /**
22       * 方言の名称を取得します。
23       * <p>この値は、{@link SqlTemplateEngine#setSuffixName(String)} にも使用されます。</p>
24       *
25       * @return 方言の名称。
26       */
27      String getName();
28  
29      /**
30       * {@link GenerationType#AUTO} が指定された場合の、 デフォルトの {@link GenerationType}を返します。
31       * @return {@link GenerationType} を返します。
32       */
33      GenerationType getDefaultGenerationType();
34  
35      /**
36       * サポートする主キーの生成戦略を判定します。
37       * @param generationType 主キーの生成戦略。
38       * @return {@literal true}のときサポートします。
39       */
40      boolean isSupportedGenerationType(GenerationType generationType);
41  
42      /**
43       * シーケンスをインクリメントする処理を取得します。
44       * @param dataSource データソース
45       * @param sequenceName シーケンス名
46       * @return DBに対応したインクリメントする処理の実装を返します。シーケンスをサポートしない場合は{@literal null}を返します。
47       */
48      DataFieldMaxValueIncrementer getSequenceIncrementer(DataSource dataSource, String sequenceName);
49  
50      /**
51       * 対応するクラスタイプに対する値タイプを返します。
52       * Oracleなどのようにbooleanが存在しない場合は対応する{@link ValueType} に切り替えたりします。
53       *
54       * @param valueType 値タイプ
55       * @return 値タイプ
56       */
57      ValueType<?> getValueType(ValueType<?> valueType);
58  
59      /**
60       * 件数取得用のSQLのSELECT句を取得します。
61       * @return SELECT句
62       */
63      String getCountSql();
64  
65      /**
66       * ヒントコメントを返します。
67       * <p>ヒント句をサポートしていないDBの場合は空文字を返します。</p>
68       *
69       * @param hint ヒント
70       * @return ヒントコメント
71       */
72      String getHintComment(String hint);
73  
74      /**
75       * LIMIT用SQLに変換します。
76       * @param sql SQL
77       * @param offset オフセット。省略する場合は {@literal -1}を指定します。
78       * @param limit リミット。省略する場合は {@literal -1} を指定します。
79       * @return LIMIT用SQL。
80       */
81      String convertLimitSql(String sql, int offset, int limit);
82  
83      /**
84       * SELECT文で<code>FOR UPDATE</code>をサポートしていれば<code>true</code>を返します。
85       *
86       * @param type SELECT ～ FOR UPDATEのタイプ
87       * @return SELECT文で<code>FOR UPDATE</code>をサポートしていれば<code>true</code>
88       */
89      boolean isSupportedSelectForUpdate(SelectForUpdateType type);
90  
91      /**
92       * SELECT文に付加する<code>FOR UPDATE NOWAIT</code>相当のSQLを返します。
93       *
94       * @param type SELECT ～ FOR UPDATEのタイプ
95       * @param waitSeconds <code>type</code>に{@link SelectForUpdateType#WAIT} が指定された場合の待機時間(秒単位)
96       * @return SELECT文に付加する<code>FOR UPDATE</code>句のSQL
97       */
98      String getForUpdateSql(SelectForUpdateType type, int waitSeconds);
99  }
