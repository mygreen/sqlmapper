3   import org.springframework.lang.Nullable;

5   import com.github.mygreen.sqlmapper.core.annotation.GeneratedValue.GenerationType;
6   import com.github.mygreen.sqlmapper.core.type.ValueType;
7   import com.github.mygreen.sqlmapper.core.type.standard.BooleanType;
8   import com.github.mygreen.sqlmapper.core.type.standard.NumberableBooleanType;
9   
10  /**
11   * 古いOracleDBの方言です。
12   * <p>Oracle11以前が対象です。
13   *
14   *
15   * @author T.TSUCHIE
16   *
17   */
18  public class OracleLegacyDialect extends OracleDialect {
19  
20      private final NumberableBooleanType objectiveBooleanType = new NumberableBooleanType(false);
21  
22      private final NumberableBooleanType primitiveBooleanType = new NumberableBooleanType(true);
23  
24      /**
25       * {@inheritDoc}
26       *
27       * @return <ul>
28       *  <li>{@link GenerationType#IDENTITY} : {@literal false}</li>
29       *  <li>{@link GenerationType#SEQUENCE} : {@literal true}</li>
30       *  <li>{@link GenerationType#TABLE} : {@literal true}</li>
31       *  <li>その他 : {@literal false}</li>
32       * </ul>
33       */
34      @Override
35      public boolean isSupportedGenerationType(GenerationType generationType) {
36          switch(generationType) {
37              case IDENTITY:
38                  return false;
39              case SEQUENCE:
40                  return true;
41              case TABLE:
42                  return true;
43              default:
44                  return false;
45          }
46      }
47  
48      /**
49       * {@inheritDoc}
50       *
51       * @return 与えられた値が {@literal boolean/Boolean}のとき、整数型に変換する {@link NumberableBooleanType} に変換します。
52       */
53      @Override
54      public ValueType<?> getValueType(@Nullable ValueType<?> valueType) {
55          if(valueType == null) {
56              return null;
57          }
58  
59          if(valueType instanceof BooleanType) {
60              if(((BooleanType)valueType).isForPrimitive()) {
61                  return primitiveBooleanType;
62              } else {
63                  return objectiveBooleanType;
64              }
65          }
66          return valueType;
67      }
68  
69      /**
70       * {@inheritDoc}
71       *
72       * @return {@literal ROWNUMBER}を使用し、疑似的にLIMIT句を表現します。
73       */
74      @Override
75      public String convertLimitSql(String sql, int offset, int limit) {
76          StringBuilder buf = new StringBuilder(sql.length() + 100);
77          sql = sql.trim();
78          String lowerSql = sql.toLowerCase();
79          boolean isForUpdate = false;
80          if (lowerSql.endsWith(" for update")) {
81              sql = sql.substring(0, sql.length() - 11);
82              isForUpdate = true;
83          }
84          buf.append("select * from ( select temp_.*, rownum rownumber_ from ( ");
85          buf.append(sql);
86          buf.append(" ) temp_ ) where");
87          boolean hasOffset = offset > 0;
88          if (hasOffset) {
89              buf.append(" rownumber_ > ");
90              buf.append(offset);
91          }
92          if (limit > 0) {
93              if (hasOffset) {
94                  buf.append(" and");
95              }
96              buf.append(" rownumber_ <= ");
97              buf.append(offset + limit);
98          }
99          if (isForUpdate) {
100             buf.append(" for update");
101         }
102         return buf.toString();
103     }

105 }
