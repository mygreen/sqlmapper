1   package com.github.mygreen.sqlmapper.metamodel.expression;

3   import com.github.mygreen.sqlmapper.metamodel.Predicate;
4   import com.github.mygreen.sqlmapper.metamodel.operation.BooleanOperation;
5   import com.github.mygreen.sqlmapper.metamodel.operator.BooleanOp;
6   import com.github.mygreen.sqlmapper.metamodel.operator.UnaryOp;
7   
8   /**
9    * ブーリアンによる式を表現します。
10   *
11   * @author T.TSUCHIE
12   *
13   */
14  public abstract class BooleanExpression extends ComparableExpression<Boolean> implements Predicate {
15  
16      public BooleanExpression(Expression<Boolean> mixin) {
17          super(mixin);
18      }
19  
20      /**
21       * {@literal 左辺 == true} として比較する式を作成します。
22       * @return {@literal 左辺 = TRUE}
23       */
24      public BooleanExpression isTrue() {
25          return eq(Constant.createBoolean(true));
26      }
27  
28      /**
29       * {@literal 左辺 == false} として比較する式を作成します。
30       * @return {@literal 左辺 = FALSE}
31       */
32      public BooleanExpression isFalse() {
33          return eq(Constant.createBoolean(false));
34      }
35  
36      @Override
37      public BooleanExpression not() {
38          return new BooleanOperation(UnaryOp.NOT, mixin);
39      }
40  
41      /**
42       * 右辺を論理席({@literal 左辺 AND 右辺})で評価します。
43       * @param right 右辺
44       * @return {@literal 左辺 AND 右辺}
45       */
46      public BooleanExpression and(final Predicate right) {
47          if(right == null) {
48              // 右辺がnullの場合何もしない
49              return this;
50          }
51          return new BooleanOperation(BooleanOp.AND, mixin, right);
52      }
53  
54      /**
55       * 右辺を論理和( {@literal 左辺 OR 右辺}) で評価します。
56       * @param right 右辺
57       * @return {@literal 左辺 OR 右辺}
58       */
59      public BooleanExpression or(final Predicate right) {
60          if(right == null) {
61              // 右辺がnullの場合何もしない
62              return this;
63          }
64          return new BooleanOperation(BooleanOp.OR, mixin, right);
65      }
66  
67      /**
68       * 引数で指定した全ての和({@literal OR})に対して積({@literal AND})を取ります。
69       * <p>例：{@literal 左辺 AND (A OR B OR C ...)}
70       *
71       * @param predicates 和(OR)を取る対象の式
72       * @return {@literal 左辺 AND (右辺1 OR 右辺2 OR 右辺3 ...)}
73       */
74      public BooleanExpression andAnyOf(Predicate... predicates) {
75          if(predicates == null || predicates.length == 0) {
76              // 右辺がnullの場合何もしない
77              return this;
78  
79          }
80  
81          // 引数の述語をORで繋げる
82          Predicate right = null;
83          for(Predicate b : predicates) {
84              if(b != null) {
85                  right = (right == null) ? b : new BooleanOperation(BooleanOp.OR, right, b);
86              }
87          }
88  
89          return and(right);
90      }
91  
92      /**
93       * 引数で指定した全ての積({@literal AND})に対して和({@literal OR})を取ります。
94       * <p>例：{@literal 左辺 OR (A AND B AND C ...)}
95       *
96       * @param predicates 和(AND)を取る対象の式
97       * @return {@literal 左辺 OR (右辺1 AND 右辺2 AND 右辺3 ...)}
98       */
99      public BooleanExpression orAllOf(final Predicate... predicates) {
100         if(predicates == null || predicates.length == 0) {
101             // 右辺がnullの場合何もしない
102             return this;
103         }
104 
105         // 引数の述語をANDで繋げる
106         Predicate right = null;
107         for(Predicate b : predicates) {
108             if(b != null) {
109                 right = (right == null) ? b : new BooleanOperation(BooleanOp.AND, right, b);
110             }
111         }
112 
113         return or(right);
114     }
115 
116 }
