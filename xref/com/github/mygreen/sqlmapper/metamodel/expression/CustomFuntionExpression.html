33  /**
34   * 任意の関数式を表現します。
35   *
36   * @since 0.3
37   * @author T.TSUCHIE
38   *
39   */
40  public abstract class CustomFuntionExpression<T> implements Expression<T> {
41  
42      protected final Expression<?> mixin;
43  
44      /**
45       * 関数のクエリ
46       */
47      @Getter
48      protected final String query;
49  
50      /**
51       * 関数の引数
52       */
53      @Getter
54      protected final List<Expression<?>> args;
55  
56      /**
57       * SQL関数をパースしてトークンに分解した結果
58       */
59      @Getter
60      protected final List<Token> tokens;
61  
62      public CustomFuntionExpression(Expression<?> mixin, String query, Object... args) {
63          if(query == null || query.isEmpty()) {
64              throw new IllegalArgumentException("query should be not empty.");
65          }
66  
67          /*
68           * クエリ中のプレースホルダーと引数の個数を比較する。
69           * プレースホルダーをカウントするときには、文字列句('abc')などを考慮する必要があるが、
70           * そこまで複雑なことはしないと思うので実施しない。
71           */
72          List<Token> tokens = new SqlFunctionParser().parse(query);
73          int countPlaceHolder = (int) tokens.stream()
74                  .filter(t -> t.type == TokenType.BIND_VARIABLE)
75                  .count();
76          int sizeArgs = (args == null ? 0 : args.length);
77          if(countPlaceHolder != sizeArgs) {
78              throw new IllegalArgumentException(String.format("'%s' is not match place holder count '%d'. ", query, countPlaceHolder));
79          }
80  
81          this.mixin = mixin;
82          this.query = query;
83          this.tokens = tokens;
84  
85          if(sizeArgs == 0) {
86              this.args = Collections.emptyList();
87          } else {
88              this.args = Arrays.stream(args)
89                      .map(this::convertExpression)
90                      .collect(Collectors.toList());
91          }
92  
93      }
94  
95      /**
96       * 値を式オブジェクトに変換する。
97       * @param value 値
98       * @return 変換した値
99       */
100     @SuppressWarnings("rawtypes")
101     private Expression<?> convertExpression(Object value) {
102         if(value instanceof Expression) {
103             return (Expression)value;
104         } else if(value instanceof Collection) {
105             return Constant.createCollection((Collection)value);
106         } else {
107             return Constant.create(value);
108         }
109     }
