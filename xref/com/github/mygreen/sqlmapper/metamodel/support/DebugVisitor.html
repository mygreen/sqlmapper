62      protected void initOperationTemplate() {
63  
64          // BooleanOp
65          operationTemplateMap.put(BooleanOp.AND, "{0} and {1}");

68          // UnaryOp

70          operationTemplateMap.put(UnaryOp.IS_NULL, "{0} is null");
71          operationTemplateMap.put(UnaryOp.IS_NOT_NULL, "{0} is not null");
72          operationTemplateMap.put(UnaryOp.EXISTS, "exists {0}");
73          operationTemplateMap.put(UnaryOp.NOT_EXISTS, "not exists {0}");
74  
75          // ComparisionOp
76          operationTemplateMap.put(ComparisionOp.EQ, "{0} = {1}");
77          operationTemplateMap.put(ComparisionOp.NE, "{0} <> {1}");
78          operationTemplateMap.put(ComparisionOp.IN, "{0} in {1}");
79          operationTemplateMap.put(ComparisionOp.NOT_IN, "{0} not in {1}");
80          operationTemplateMap.put(ComparisionOp.BETWEEN, "{0} between {1} and {2}");
81          operationTemplateMap.put(ComparisionOp.GOE, "{0} >= {1}");
82          operationTemplateMap.put(ComparisionOp.GT, "{0} > {1}");
83          operationTemplateMap.put(ComparisionOp.LOE, "{0} <= {1}");

111     @Override
112     public void visit(final Operation<?> expr, final DebugVisitorContext context) {
113 
114         final Operator operator = expr.getOperator();
115         final String template = operationTemplateMap.get(operator);
116         if(template != null) {
117             // 引数を評価します。
118             List<Object> evals = new ArrayList<>();
119             for(Expression<?> arg : expr.getArgs()) {
120                 DebugVisitorContext argContext =  new DebugVisitorContext();
121                 invoke(operator, arg, argContext);
122                 evals.add(argContext.getCriteria());
123             }
124 
125             String formatExpr = MessageFormat.format(template, evals.toArray(new Object[evals.size()]));
126             context.append(formatExpr);
127 
128         } else if(operator == FunctionOp.CURRENT_DATE
129                 || operator == FunctionOp.CURRENT_TIME
130                 || operator == FunctionOp.CURRENT_TIMESTAMP) {
131             context.append(operator.name().toLowerCase());
132             // 精度が存在するとき評価する
133             if(!expr.getArgs().isEmpty()) {
134                 context.append("(");
135                 invoke(operator, expr.getArg(0), context);
136                 context.append(")");
137 
138             }
139 
140         } else if(operator == FunctionOp.CUSTOM) {
141 
142             // 左辺の評価
143             Expression<?> left = expr.getArg(0);
144             DebugVisitorContext leftContext = new DebugVisitorContext();
145             invoke(operator, left, leftContext);
146 
147             // トークンを置換していく
148             CustomFunctionOperation op = (CustomFunctionOperation) expr.getArg(1);
149             @SuppressWarnings("unchecked")
150             List<Token> tokens = op.getTokens();
151             for(Token token : tokens) {
152                 if(token.type == TokenType.SQL) {
153                     context.append(token.value);
154 
155                 } else if(token.type == TokenType.LEFT_VARIABLE) {
156                     context.append(leftContext.getCriteria());
157 
158                 } else if(token.type == TokenType.BIND_VARIABLE) {
159                     int varIndex = token.bindBariableIndex;
160                     Expression<?> arg = op.getArg(varIndex);
161 
162                     DebugVisitorContext argContext = new DebugVisitorContext();
163                     invoke(operator, arg, argContext);
164 
165                     context.append(argContext.getCriteria());
166 
167                 } else {
168                     // unknown token
169                 }
170 
171             }
172 
173         }
174     }

176     @Override
177     public void visit(final Constant<?> expr, final DebugVisitorContext context) {
178 
179         final Object constant = expr.getValue();
180 
181         if(constant == null) {
182             context.append("null");
183 
184         } else if(expr.isExpandable()) {
185             context.append("(");
186             StringBuilder joined = new StringBuilder();
187             for(Object value : (Collection<?>)constant) {
188                 if(joined.length() > 0) {
189                     joined.append(", ");
190                 }
191                 joined.append(formatConstantValue(value));
192             }
193             context.append(joined.toString())
194                 .append(")");
195 
196         } else {
197             context.append(formatConstantValue(constant));
198         }
199 
200     }

202     /**
203      * 定数の値をフォーマットします。
204      * @param value フォーマット対象の値
205      * @return フォーマットした値
206      */
207     private String formatConstantValue(final Object value) {
208         if(value == null) {
209             return "null";
210         }
211 
212         if(value instanceof CharSequence) {
213             return "'" + value.toString() + "'";
214 
215         } else if(value instanceof Date) {
216             return new SimpleDateFormat("yyyy-MM-dd HH:mm:ss").format(value);
217         } else if(value instanceof java.sql.Date) {
218             return new SimpleDateFormat("yyyy-MM-dd").format(value);
219         } else if(value instanceof Time) {
220             return new SimpleDateFormat("HH:mm:ss").format(value);
221         } else if(value instanceof Timestamp) {
222             return new SimpleDateFormat("yyyy-MM-dd HH:mm:ss.SSS").format(value);
223         } else if(value instanceof LocalDateTime) {
224             return FORMATTER_LOCAL_DATETIME.format((LocalDateTime)value);
225         } else if(value instanceof LocalDate) {
226             return FORMATTER_LOCAL_DATE.format((LocalDate)value);
227         } else if(value instanceof LocalTime) {
228             return FORMATTER_LOCAL_TIME.format((LocalTime)value);
229         }
230 
231         return value.toString();
232     }

234     @Override
235     public void visit(final Path<?> expr, final DebugVisitorContext context) {
236 
237         PathMeta pathMeta = expr.getPathMeta();
238         String pathName = pathMeta.getElement();
239         Path<?> parent = null;
240 
241         // 親をたどりパス名を追加していく。
242         // <親のパス>.<子のパス>
243         do {
244             parent = pathMeta.getParent();
245             if(parent == null) {
246                 break;
247             }
248             pathMeta = parent.getPathMeta();
249             // 親の名前を先頭に追加していく。
250             pathName = pathMeta.getElement() + "." + pathName;
251         } while(parent.getPathMeta().getType() != PathType.ROOT);
252 
253         context.append(pathName);
254 
255     }

257     @Override
258     public void visit(final SubQueryExpression<?> expr, final DebugVisitorContext context) {
259 
260         final SubQueryMeta queryMeta = expr.getQueryMeta();
261 
262         context.append("select");
263 
264         // 抽出カラムの組み立て
265         if(!queryMeta.getIncludesProperties().isEmpty()) {
266             StringBuilder columnQuery = new StringBuilder();
267             for(PropertyPath<?> propertyPath : queryMeta.getIncludesProperties()) {
268                 if(columnQuery.length() > 0) {
269                     columnQuery.append(", ");
270                 }
271 
272                 // プロパティのパスの評価
273                 DebugVisitorContext propertyPathContext = new DebugVisitorContext();
274                 propertyPath.accept(this, propertyPathContext);
275                 columnQuery.append(propertyPathContext.getCriteria());
276             }
277             context.append(columnQuery.toString());
278         } else {
279             context.append(" *");
280         }
281 
282         // from句の組み立て
283         context.append(" from ").append(queryMeta.getEntityPath().getPathMeta().getElement());
284 
285         // where句の組み立て
286         if(queryMeta.getWhere() != null) {
287             // 条件式の評価
288             DebugVisitorContext whereContext = new DebugVisitorContext();
289             queryMeta.getWhere().accept(this, whereContext);
290 
291             context.append(" whrere ")
292                 .append(whereContext.getCriteria());
293         }
294 
295         // 並び順の組み立て
296         if(!queryMeta.getOrders().isEmpty()) {
297             StringBuilder orderQuery = new StringBuilder();
298             for(OrderSpecifier order : queryMeta.getOrders()) {
299                 if(orderQuery.length() > 0) {
300                     orderQuery.append(", ");
301                 }
302 
303                 // 並び順のパスを評価する
304                 DebugVisitorContext orderContext = new DebugVisitorContext();
305                 order.getPath().accept(this, orderContext);
306                 orderQuery.append(orderContext.getCriteria())
307                     .append(" ")
308                     .append(order.getOrder().name());
309             }
310             context.append(" order by ")
311                 .append(orderQuery.toString());
312         }
313 
314         // limit句の組み立て
315         if(queryMeta.getLimit() > 0 || queryMeta.getLimit() == 0 && queryMeta.getOffset() > 0) {
316             if (queryMeta.getOffset() > 0) {
317                 context.append(" limit ")
318                     .append(queryMeta.getLimit())
319                     .append(" offset ")
320                     .append(queryMeta.getOffset());
321             } else {
322                 context.append(" limit ")
323                     .append(queryMeta.getLimit());
324             }
325         }
326 
327     }

329     protected void invoke(final Operator parentOperator, final Expression<?> expr, final DebugVisitorContext context) {
330 
331         if(expr instanceof Operation) {
332             Operation<?> operation = (Operation<?>)expr;
333             // /子ノードが演算子の場合、括弧で囲むか判定する。
334             if(OperationUtils.isEnclosedParenthesis(parentOperator, operation.getOperator())) {
335                 context.append("(");
336                 visit(operation, context);
337                 context.append(")");
338             } else {
339                 visit(operation, context);
340             }
341 
342         } else if(expr instanceof Constant) {
343             visit((Constant<?>)expr, context);
344         } else if (expr instanceof Path) {
345             visit((Path<?>)expr, context);
346         } else if(expr instanceof SubQueryExpression) {
347             context.append("(");
348             visit((SubQueryExpression<?>)expr, context);
349             context.append(")");
350         } else {
351             throw new IllegalArgumentException("not support Expression instance of " + expr.getClass());
352         }
353 
354     }
