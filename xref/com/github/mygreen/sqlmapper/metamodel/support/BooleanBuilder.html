1   package com.github.mygreen.sqlmapper.metamodel.support;

3   import com.github.mygreen.sqlmapper.metamodel.Predicate;
4   import com.github.mygreen.sqlmapper.metamodel.Visitor;
5   import com.github.mygreen.sqlmapper.metamodel.operation.BooleanOperation;
6   import com.github.mygreen.sqlmapper.metamodel.operation.PredicateOperation;
7   import com.github.mygreen.sqlmapper.metamodel.operator.BooleanOp;
8   
9   import lombok.Getter;
10  
11  /**
12   * {@link Predicate} 式を組み立てるためのヘルパークラス。
13   *
14   *
15   * <pre class="highlight"><code class="java">
16   * QEmployee employee = QEmployee.employee;
17   * BooleanBuilder builder = new BooleanBuilder();
18   * for (String name : names) {
19   *     builder.or(employee.name.eq(name));
20   * }
21   * </code></pre>
22   *
23   *
24   * @author T.TSUCHIE
25   *
26   */
27  public class BooleanBuilder implements Predicate {
28  
29      /**
30       * 現在の式
31       */
32      @Getter
33      private Predicate predicate;
34  
35      /**
36       * {@link BooleanBuilder} のインスタンスを作成します。
37       */
38      public BooleanBuilder() {
39  
40      }
41  
42      /**
43       * 初期式を指定して、{@link BooleanBuilder} のインスタンスを作成します。
44       * @param initial 諸域式
45       */
46      public BooleanBuilder(Predicate initial) {
47          this.predicate = initial;
48      }
49  
50      @Override
51      public Class<? extends Boolean> getType() {
52          return Boolean.class;
53      }
54  
55      /**
56       * {@inheritDoc}
57       * <p>評価する式がない場合は何もしません。
58       */
59      @Override
60      public <C> void accept(Visitor<C> visitor, C context) {
61          if(predicate != null) {
62              predicate.accept(visitor, context);
63          }
64      }
65  
66      /**
67       * {@inheritDoc}
68       *
69       * @return 否定する式がない場合は {@literal null} を返す。
70       */
71      @Override
72      public Predicate not() {
73  
74          if(predicate != null) {
75              return predicate.not();
76          }
77  
78          return null;
79  
80      }
81  
82      /**
83       * 式を持つかどうか判定します。
84       * @return 式を持つとき、{@literal true}を返します。
85       */
86      public boolean hasValue() {
87          return predicate != null;
88      }
89  
90      /**
91       * 右辺を論理積( {@literal 左辺 AND 右辺})で評価します。
92       * <p>左辺が存在しない場合は何もしません。
93       *
94       * @param right 右辺。nullの場合、何もしません。
95       * @return 自身のインスタンス
96       */
97      public BooleanBuilder and(Predicate right) {
98  
99          if(right != null) {
100             if(predicate == null) {
101                 this.predicate = right;
102             } else {
103                 predicate = new PredicateOperation(BooleanOp.AND, predicate, right);
104             }
105         }
106 
107         return this;
108 
109     }
110 
111     /**
112      * 右辺を論理和( {@literal 左辺 OR 右辺})で評価します。
113      * <p>左辺が存在しない場合は何もしません。
114      *
115      * @param right 右辺。nullの場合、何もしません。
116      * @return 自身のインスタンス
117      */
118     public BooleanBuilder or(Predicate right) {
119 
120         if(right != null) {
121             if(predicate == null) {
122                 this.predicate = right;
123             } else {
124                 predicate = new PredicateOperation(BooleanOp.OR, predicate, right);
125             }
126         }
127 
128         return this;
129 
130     }
131 
132     /**
133      * 引数で指定した全ての和({@literal OR})に対して積({@literal AND})を取ります。
134      * <p>例：{@literal 左辺 AND (A OR B OR C ...)}
135      *
136      * @param predicates 和(OR)を取る対象の式
137      * @return {@literal 左辺 AND (右辺1 OR 右辺2 OR 右辺3 ...)}
138      */
139     public BooleanBuilder andAnyOf(Predicate... predicates) {
140 
141         if(predicates == null || predicates.length == 0) {
142             // 右辺がnullの場合何もしない
143             return this;
144 
145         }
146 
147         // 引数の述語をORで繋げる
148         Predicate right = null;
149         for(Predicate b : predicates) {
150             if(b != null) {
151                 right = (right == null) ? b : new BooleanOperation(BooleanOp.OR, right, b);
152             }
153         }
154 
155         return and(right);
156 
157     }
158 
159     /**
160      * 引数で指定した全ての積({@literal AND})に対して和({@literal OR})を取ります。
161      * <p>例：{@literal 左辺 OR (A AND B AND C ...)}
162      *
163      * @param predicates 和(AND)を取る対象の式
164      * @return {@literal 左辺 OR (右辺1 AND 右辺2 AND 右辺3 ...)}
165      */
166     public BooleanBuilder orAllOf(Predicate... predicates) {
167 
168         if(predicates == null || predicates.length == 0) {
169             // 右辺がnullの場合何もしない
170             return this;
171         }
172 
173         // 引数の述語をANDで繋げる
174         Predicate right = null;
175         for(Predicate b : predicates) {
176             if(b != null) {
177                 right = (right == null) ? b : new BooleanOperation(BooleanOp.AND, right, b);
178             }
179         }
180 
181         return or(right);
182 
183     }
184 
185 }
