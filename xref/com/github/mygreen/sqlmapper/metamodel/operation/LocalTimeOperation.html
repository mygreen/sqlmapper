6   import com.github.mygreen.sqlmapper.metamodel.Visitor;

14  /**
15   * {@link LocalTime} を値に持つ式の演算操作を表現します。
16   *
17   *
18   * @author T.TSUCHIE
19   *
20   */
21  public class LocalTimeOperation extends LocalTimeExpression implements Operation<LocalTime> {
22  
23      private final OperationMixin<LocalTime> opMixin;
24  
25      public LocalTimeOperation(OperationMixin<LocalTime> mixin) {
26          super(mixin);
27          this.opMixin = mixin;
28      }
29  
30      public LocalTimeOperation(@NonNull Operator op, Expression<?>... args) {
31          this(new OperationMixin<>(LocalTime.class, op, args));
32      }
33  
34      @Override
35      public <C> void accept(Visitor<C> visitor, C context) {
36          visitor.visit(opMixin, context);
37      }
38  
39      @Override
40      public Operator getOperator() {
41          return opMixin.getOperator();
42      }
43  
44      @Override
45      public Expression<?> getArg(int index) {
46          return opMixin.getArg(index);
47      }
48  
49      @Override
50      public List<Expression<?>> getArgs() {
51          return opMixin.getArgs();
52      }
53  }
