13  /**
14   * APT処理のユーティリティクラスです。
15   *
16   *
17   * @author T.TSUCHIE
18   *
19   */
20  public class AptUtils {
21  
22      /**
23       * 要素が非finalなインスタンスフィールド(非staticなフィールド)か判定します。
24       * @param element 判定対象の要素
25       * @return 非finalなインスタンスフィールド(非staticなフィールド)のとき{@literal true}を返します。
26       */
27      public static boolean isInstanceField(final Element element) {
28  
29          // フィールドかどうか
30          if(element.getKind() != ElementKind.FIELD) {
31              return false;
32          }
33  
34          // 非static かどうか
35          if(element.getModifiers().contains(Modifier.STATIC)) {
36              return false;
37          }
38  
39          // 非finalかどうか
40          if(element.getModifiers().contains(Modifier.FINAL)) {
41              return false;
42          }
43  
44          return true;
45  
46      }
47  
48      /**
49       * クラス名とパッケージ名のセパレータを取得します。
50       * @param entityModel エンティティモデル情報
51       * @return 内部クラスのとき {@literal "$"} を返し、それ以外の時は {@literal "."} を返します。
52       */
53      public static String getPackageClassNameSeparator(final EntityMetamodel entityModel) {
54          return entityModel.getType().isStaticInnerClass() ? "$" : ".";
55      }
56  
57      /**
58       * 継承しているクラス情報を抽出する。
59       * @param type 現在のクラス情報
60       * @param typeUtils ユーティリティ。
61       * @param superTypes 抽出したクラス情報。
62       */
63      public static void extractSuperClassTypes(final TypeMirror type, final Types typeUtils, final List<TypeMirror> superTypes) {
64  
65          if(type.toString().equals(Object.class.getCanonicalName())) {
66              return;
67          }
68  
69          List<? extends TypeMirror> list = typeUtils.directSupertypes(type);
70          for(TypeMirror superType : list) {
71              if(!superTypes.contains(superType)) {
72                  superTypes.add(superType);
73              }
74  
75              // 再帰的に見ていく
76              extractSuperClassTypes(superType, typeUtils, superTypes);
77          }
78  
79  
80      }
81  
82  }
